76bool SmtTimer::isBarrierReached()
77{
78 // A thread has to give up m_lock when calling the global barrier
79 // When this happens, we don't want anyone to continue
80 if (in_sync)
81 return false;
82
83 bool any_in_barrier = false;
84 for(smtthread_id_t thread_num = 0; thread_num < m_threads.size(); ++thread_num)
85 {
86 SmtThread *thread = m_threads[thread_num];
87
88 if (thread->running && !thread->in_barrier)
89 {
90 return false;
91 }
92 else if (thread->in_barrier)
93 {
94 any_in_barrier = true;
95 }
96 }
97
98 return any_in_barrier;
99}

101bool SmtTimer::barrierRelease(bool release_all, smtthread_id_t thread_id)
102{
103 bool release_me = false;
104 execute_thread = thread_id;
105
106 // Release those threads for which we want more instructions
107
108 unsigned int num_running = 0;
109 for(smtthread_id_t thread_num = 0; thread_num < m_threads.size(); ++thread_num)
110 {
111 SmtThread *thread = m_threads[thread_num];
112
113 if (!thread->in_barrier)
114 {
115 if (thread->running)
116 ++num_running;
117
118 // Thread wasn't stalled, no need to wake it up
119 continue;
120 }
121
122 if (threadHasEnoughInstructions(thread_num))
123 {
124 // We have enough instructions for now, keep the thread stalled
125 continue;
126 }
127
128 if (thread->running)
129 ++num_running;
130
131 // We want more instructions from this thread, release it
132 if (thread_num == thread_id)
133 {
134 release_me = true;
135 }
136 else
137 {
138 thread->in_barrier = false;
139 thread->cond.signal();
140 }
141
142 if (!release_all)
143 {
144 execute_thread = thread_num;
145 break;
146 }
147 }
148
149
150 UInt64 limit = 1;
151
152 while (num_running == 0)
153 {
154 // All threads have enough instructions. We still need to release someone to make forward progress.
155 // Find the thread with the (approximately) lowest number of surplus instructions and release it
156
157 for(smtthread_id_t thread_num = 0; thread_num < m_threads.size(); ++thread_num)
158 {
159 SmtThread *thread = m_threads[thread_num];
160
161 if (thread->in_barrier && thread->running && threadNumSurplusInstructions(thread_num) < limit)
162 {
163 ++num_running;
164
165 if (thread_num == thread_id)
166 {
167 release_me = true;
168 }
169 else
170 {
171 thread->in_barrier = false;
172 thread->cond.signal();
173 }
174
175 if (!release_all)
176 {
177 execute_thread = thread_num;
178 break;
179 }
180 }
181 }
182
183 limit <<= 1;
184 }
185
186 return release_me;
187}

190bool SmtTimer::barrier(smtthread_id_t thread_id)
191{
192 if (!enabled)
193 return false;
194
195 SmtThread *smtthread = m_threads[thread_id];
196 const Thread *thread = smtthread->thread;
197 LOG_ASSERT_ERROR(thread != NULL, "How can SMT barrier be called when there is not thread running?");
198
199 smtthread->in_barrier = true;
200 CLOG("smtbarrier", "Entry core %d thread %d", smtthread->core->getId(), thread->getId());
201
202 if (!isBarrierReached())
203 {
204 smtthread->cond.wait(m_lock);
205 }
206 else
207 {
208 bool release_me = barrierRelease(false, thread_id);
209
210 if (release_me)
211 smtthread->in_barrier = false;
212 else
213 smtthread->cond.wait(m_lock);
214 }
215
216 CLOG("smtbarrier", "Exit core %d thread %d (execute = %s)",
217 smtthread->core->getId(), thread->getId(), thread_id == execute_thread ? "true" : "false");
218
219 // Will we call execute() ?
220 return thread_id == execute_thread;
221}

286void SmtTimer::threadResume(HooksManager::ThreadResume *argument)
287{
288 ScopedLock sl(m_lock);
289
290 // When a thread wakes up, it will usually call simulate() before it processes the SYNC_INSTRUCTION
291 // that updates their thread->now to the wakeup time. In this intervening period,
292 // we should not return any latency from simulate() since that will be huge (this->now has advanced,
293 // because the other threads have ran, leaving this thread who was asleep behind; but thread->now
294 // is still set to the time the thread went to sleep).
295 // Instead, we listen for HookType::HOOK_THREAD_RESUME (called from the MCP, before the thread gets its
296 // wakeup message and way before the offending simulate() call) and set thread->in_wakeup, which signals
297 // simulate() to not return any latency until RobSmtTimer::synchronize() was called which updates
298 // thread->now to the correct wakeup time.
299
300 smtthread_id_t smtthread_id = findSmtThreadFromThread(argument->thread_id);
301 if (smtthread_id != INVALID_SMTTHREAD_ID)
302 {
303 m_threads[smtthread_id]->in_wakeup = true;
304 m_threads[smtthread_id]->running = true;
305 notifyNumActiveThreadsChange();
306 }
307}

309void SmtTimer::threadMigrate(HooksManager::ThreadMigrate *argument)
310{
311 ScopedLock sl(m_lock);
312
313 // Same as in wakeup case: thread will send a SpawnInstruction, but we don't want to send huge latencies
314 // to simulate() calls that precede it
315
316 // If this thread was previously running on this core, mark it as gone
317 smtthread_id_t smtthread_id = findSmtThreadFromThread(argument->thread_id);
318 if (smtthread_id != INVALID_SMTTHREAD_ID)
319 {
320 SmtThread *thread = m_threads[smtthread_id];
321 thread->running = false;
322 thread->thread = NULL;
323 notifyNumActiveThreadsChange();
324 if (thread->in_barrier)
325 {
326 thread->in_barrier = false;
327 thread->cond.signal();
328 }
329 }
330
331 // If the new core is one of ours, mark it as running this thread
332 smtthread_id = findSmtThreadFromCore(argument->core_id);
333 if (smtthread_id != INVALID_SMTTHREAD_ID)
334 {
335 #ifdef DEBUG_PERCYCLE
336 std::cout<<"** ["<<argument->thread_id<<"] threadMigrate"<<std::endl;
337 #endif
338 m_threads[smtthread_id]->in_wakeup = true;
339 m_threads[smtthread_id]->running = true;
340 m_threads[smtthread_id]->thread = Sim()->getThreadManager()->getThreadFromID(argument->thread_id);
341 notifyNumActiveThreadsChange();
342 }

345void SmtTimer::simulate(smtthread_id_t thread_id)
346{
347 SmtThread *thread = m_threads[thread_id];
348
349 if (threadNumSurplusInstructions(thread_id) > 128)
350 {
351 if (barrier(thread_id))
352 {
353 execute();
354
355 // We executed this cycle, make sure no-one else (potentially released through
356 // sync->periodic->reschedule->migrate) executes again.
357 execute_thread = INVALID_THREAD_ID;
358
359 ClockSkewMinimizationClient *client = thread->core->getClockSkewMinimizationClient();
360 if (client)
361 {
362 in_sync = true;
363 m_lock.release();
364 client->synchronize();
365 m_lock.acquire();
366 in_sync = false;
367 }
368
369 barrierRelease(true, thread_id);
370 }
371 }
