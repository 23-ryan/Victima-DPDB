5// Carbon Computer Simulator
6//
7// This simulator models future multi-core computers with thousands of cores.
8// It runs on today's x86 multicores and will scale as more and more cores
9// and better inter-core communications mechanisms become available.
10// The simulator provides a platform for research in processor architecture,
11// compilers, network interconnect topologies, and some OS.
12//
13// The simulator runs on top of Intel's Pin dynamic binary instrumention engine.
14// Application code in the absence of instrumentation runs more or less
15// natively and is thus high performance. When instrumentation is used, models
16// can be hot-swapped or dynamically enabled and disabled as desired so that
17// performance tracks the level of simulation detail needed.

157void handleCheckScheduled(THREADID threadIndex)
158{
159 Thread* thread = localStore[threadIndex].thread;
160 Core* core = thread->getCore();
161
162 if (core == NULL)
163 {
164 // This thread isn't scheduled. Normally it shouldn't execute any code,
165 // but just after thread start we do get here.
166 // Wait here until we're scheduled on one of the cores.
167 SubsecondTime time = SubsecondTime::Zero();
168 thread->reschedule(time, core);
169
170 // We should be on a core now, set its performance model to our local time
171 core = thread->getCore();
172 // If the core already has a later time, we have to wait
173 time = std::max(time, core->getPerformanceModel()->getElapsedTime());
174 core->getPerformanceModel()->queuePseudoInstruction(new SpawnInstruction(time));
175 }
176}

188VOID traceCallback(TRACE trace, void *v)
189{
190 InstMode::inst_mode_t inst_mode = INSTR_GET_MODE(trace);
191
192 BBL bbl_head = TRACE_BblHead(trace);
193 INS ins_head = BBL_InsHead(bbl_head);
194
195 #ifdef PINPLAY_SUPPORTED
196 // Make sure all version switches happen before any PinPlay instrumentation, to avoid PinPlay seeing some instructions twice
197 CALL_ORDER call_order = (CALL_ORDER)(pinplay_engine.PinPlayFirstBeforeCallOrder() - 1);
198 #else
199 CALL_ORDER call_order = CALL_ORDER_DEFAULT;
200 #endif
201
202 // Write the resulting mode to REG_INST_Gx for use by INS_InsertVersionCase
203 INS_InsertCall(ins_head, IPOINT_BEFORE, (AFUNPTR)getInstMode, IARG_RETURN_REGS, g_toolregs[TOOLREG_TEMP], IARG_CALL_ORDER, call_order, IARG_END);
204
205 // Add version switch cases for all possible target versions (no test to switch to self)
206 #define SWITCH_VERSION(v) if (inst_mode != (v)) INS_InsertVersionCase(ins_head, g_toolregs[TOOLREG_TEMP], v, v, IARG_CALL_ORDER, call_order, IARG_END);
207 SWITCH_VERSION(InstMode::DETAILED)
208 SWITCH_VERSION(InstMode::CACHE_ONLY)
209 SWITCH_VERSION(InstMode::FAST_FORWARD)
210
211 // Version 0 is only for startup / amnesia, don't do anything else there
212 if (TRACE_Version(trace) == 0)
213 return;
214
215 addCheckScheduled(trace, ins_head);
216
217 addRtnTracer(trace);
218
219 // Routine replacement
220 RTN rtn = TRACE_Rtn(trace);
221 if (RTN_Valid(rtn)
222 && RTN_Address(rtn) == TRACE_Address(trace))
223 {
224 lite::routineStartCallback(rtn, ins_head);
225 }
226
227 // Select per-basicblock instrumentation level.
228 // In fast-forward mode, this is the only instrumentation (except for some rare stuff like magic, syscall),
229 // so using a lighter version here significantly speeds up simulation
230
231 InstrumentLevel::Level instr_level;
232 if (Sim()->getConfig()->getBBVsEnabled()) {
233 // Someone (a Python script) has requested BBVs to be enabled.
234 instr_level = InstrumentLevel::INSTR_WITH_BBVS;
235 } else if (Sim()->getSamplingManager()->getSamplingProvider()) {
236 // Ask the sampling provider what mode they need
237 instr_level = Sim()->getSamplingManager()->getSamplingProvider()->requestedInstrumentation();
238 } else {
239 instr_level = InstrumentLevel::INSTR;
240 }
241
242 for (BBL bbl = bbl_head; BBL_Valid(bbl); bbl = BBL_Next(bbl))
243 {
244 switch (instr_level) {
245 case (InstrumentLevel::INSTR):
246 case (InstrumentLevel::INSTR_WITH_BBVS):
247 BBL_InsertCall(bbl, IPOINT_ANYWHERE, (AFUNPTR)InstructionModeling::countInstructions, IARG_THREAD_ID, IARG_ADDRINT, BBL_Address(bbl), IARG_UINT32, BBL_NumIns(bbl), IARG_END);
248 break;
249 case (InstrumentLevel::NONE):
250 break;
251 default:
252 LOG_PRINT_ERROR("Unexpected enum value '%d'", instr_level);
253 break;
254 }
255
256 // I-cache modeling during warmup
257 if (Sim()->getConfig()->getEnableICacheModeling()) {
258 INSTRUMENT(
259 INSTR_IF_CACHEONLY(inst_mode),
260 trace, BBL_InsHead(bbl), IPOINT_BEFORE,
261 AFUNPTR(InstructionModeling::accessInstructionCacheWarmup),
262 IARG_THREAD_ID,
263 IARG_ADDRINT, BBL_Address(bbl),
264 IARG_UINT32, BBL_Size(bbl),
265 IARG_END);
266 }
267
268 UINT32 icount = 0;
269
270 for(INS ins = BBL_InsHead(bbl); ; ins = INS_Next(ins))
271 {
272 // Instruction modelling
273 InstructionModeling::addInstructionModeling(trace, ins, inst_mode);
274 ++icount;
275
276 if (ins == BBL_InsTail(bbl) || icount >= 32 || INS_HasRealRep(ins))
277 {
278 if (INS_HasFallThrough(ins))
279 INSTRUMENT(INSTR_IF_DETAILED(inst_mode), trace, ins, IPOINT_AFTER, AFUNPTR(InstructionModeling::handleBasicBlock), IARG_THREAD_ID, IARG_END);
280 if (INS_IsBranch(ins))
281 INSTRUMENT(INSTR_IF_DETAILED(inst_mode), trace, ins, IPOINT_TAKEN_BRANCH, AFUNPTR(InstructionModeling::handleBasicBlock), IARG_THREAD_ID, IARG_END);
282 icount = 0;
283 }
284
285 if (ins == BBL_InsTail(bbl))
286 break;
287 }
288 }
289}

330VOID threadStartCallback(THREADID threadIndex, CONTEXT *ctxt, INT32 flags, VOID *v)
331{
332 if (! done_app_initialization)
333 {
334 // Spawn the main() thread
335 Sim()->getThreadManager()->spawnThread(INVALID_THREAD_ID, 0);
336
337 // Start the trace manager, if any
338 if (Sim()->getTraceManager())
339 Sim()->getTraceManager()->start();
340
341 // All the real initialization is done in
342 // replacement_start at the moment
343 done_app_initialization = true;
344 }
345
346 SubsecondTime time;
347 thread_id_t thread_id = Sim()->getThreadManager()->getThreadToSpawn(time);
348
349 // Set thread name for Sniper-in-Sniper simulations
350 String threadName = String("app-") + itostr(thread_id);
351 SimSetThreadName(threadName.c_str());
352
353 Sim()->getThreadManager()->onThreadStart(thread_id, time);
354
355 // Don't resize this vector at runtime as it's shared and we don't want it to be reallocated while someone uses it
356 LOG_ASSERT_ERROR(localStore.size() > threadIndex, "Ran out of thread_id slots, increase MAX_PIN_THREADS");
357
358 memset(&localStore[threadIndex], 0, sizeof(localStore[threadIndex]));
359 localStore[threadIndex].thread = Sim()->getThreadManager()->getThreadFromID(thread_id);
360 localStore[threadIndex].thread->m_os_info.tid = syscall(__NR_gettid);
361 if (Sim()->getConfig()->getEnableSpinLoopDetection())
362 localStore[threadIndex].sld.sld = new SpinLoopDetector(localStore[threadIndex].thread);
363 for (unsigned int idx = 0; idx < ThreadLocalStorage::NUM_SCRATCHPADS; ++idx)
364 {
365 int status = posix_memalign((void**) &localStore[threadIndex].scratch[idx], 4 * sizeof (void*), ThreadLocalStorage::SCRATCHPAD_SIZE);
366 assert(status == 0);
367 }
368}

395int main(int argc, char *argv[])
396{
397 // ---------------------------------------------------------------
398 // FIXME:
399 PIN_InitLock (&rtn_map_lock);
400 // ---------------------------------------------------------------
401
402 // Global initialization
403 PIN_InitSymbols();
404 PIN_Init(argc,argv);
405
406 initToolregs();
407
408 // To make sure output shows up immediately, make stdout and stderr line buffered
409 // (if we're writing into a pipe to run-graphite, or redirected to a file by the job runner, the default will be block buffered)
410 setvbuf(stdout, NULL, _IOLBF, 0);
411 setvbuf(stderr, NULL, _IOLBF, 0);
412
413 const char *ld_orig = getenv("SNIPER_SCRIPT_LD_LIBRARY_PATH");
414 if (ld_orig)
415 setenv("LD_LIBRARY_PATH", ld_orig, 1);
416
417 string_vec args;
418
419 // Set the default config path if it isn't
420 // overwritten on the command line.
421 String config_path = "carbon_sim.cfg";
422
423 parse_args(args, config_path, argc, argv);
424
425 cfg = new config::ConfigFile();
426 cfg->load(config_path);
427
428 handle_args(args, *cfg);
429
430 Simulator::setConfig(cfg, Config::PINTOOL);
431
432 std::cout << "PIN SIM TOOL" << std::endl;
433
434 Simulator::allocate();
435
436 Sim()->start();
437
438 // If -appdebug_enable is used, write out the port to connect GDB to
439 if(PIN_GetDebugStatus() != DEBUG_STATUS_DISABLED) {
440 DEBUG_CONNECTION_INFO info;
441 if (PIN_GetDebugConnectionInfo(&info) && info._type == DEBUG_CONNECTION_TYPE_TCP_SERVER) {
442 FILE* fp = fopen(Sim()->getConfig()->formatOutputFileName("appdebug_port.out").c_str(), "w");
443 fprintf(fp, "%d %d", PIN_GetPid(), info._tcpServer._tcpPort);
444 fclose(fp);
445 }
446 String scheme = Sim()->getCfg()->getString("clock_skew_minimization/scheme");
447 if (!(scheme == "none" || scheme == "barrier")) {
448 fprintf(stderr, "\n[WARNING] Application debugging is not compatible with %s synchronization.\n", scheme.c_str());
449 fprintf(stderr, " Consider adding -g --clock_skew_minimization/scheme={none|barrier}\n\n");
450 }
451 }
452
453 // Instrumentation
454 LOG_PRINT("Start of instrumentation.");
455
456 PIN_AddThreadStartFunction (threadStartCallback, 0);
457 PIN_AddThreadFiniFunction (threadFiniCallback, 0);
458
459 if (Sim()->getConfig()->getEnableSyscallEmulation())
460 {
461 PIN_AddSyscallEntryFunction(lite::syscallEnterRunModel, 0);
462 PIN_AddSyscallExitFunction(lite::syscallExitRunModel, 0);
463 }
464 bool enable_signals = cfg->getBool("general/enable_signals");
465 if (!enable_signals) {
466 PIN_InterceptSignal(SIGILL, lite::interceptSignal, NULL);
467 PIN_InterceptSignal(SIGFPE, lite::interceptSignal, NULL);
468 PIN_InterceptSignal(SIGSEGV, lite::interceptSignal, NULL);
469 }
470 PIN_InterceptSignal(SIGUSR1, handleSigUsr1, NULL);
471 PIN_AddInternalExceptionHandler(exceptionHandler, NULL);
472
473 RTN_AddInstrumentFunction(lite::routineCallback, 0);
474
475 TRACE_AddInstrumentFunction(traceCallback, 0);
476
477 if (!Sim()->getConfig()->getEnableSMCSupport())
478 CODECACHE_AddTraceInvalidatedFunction(traceInvalidate, 0);
479
480 PIN_AddDetachFunction(ApplicationDetach, 0);
481 PIN_AddFiniUnlockedFunction(ApplicationExit, 0);
482
483 PIN_AddForkFunction(FPOINT_AFTER_IN_CHILD, forkAfterChild, 0);
484
485 if (cfg->getBool("log/pin_codecache_trace"))
486 initCodeCacheTracing();
487
488 String syntax = cfg->getString("general/syntax");
489 if (syntax == "intel")
490 PIN_SetSyntaxIntel();
491 else if (syntax == "att")
492 PIN_SetSyntaxATT();
493 else if (syntax == "xed")
494 PIN_SetSyntaxXED();
495 else
496 LOG_PRINT_ERROR("Unknown assembly syntax %s, should be intel, att or xed.", syntax.c_str());
497
498 // Just in case ... might not be strictly necessary
499// PIN_SpawnInternalThread doesn't schedule its threads until after PIN_StartProgram
500// Transport::getSingleton()->barrier();
501
502#ifdef PINPLAY_SUPPORTED
503 if (Sim()->getConfig()->getEnablePinPlay())
504 {
505 pinplay_engine.Activate(argc, argv, false /*logger*/, true /*replayer*/);
506 LOG_ASSERT_ERROR(Sim()->getConfig()->getEnableSyscallEmulation() == false,
507 "System call emulation must be disabled when using PinPlay.");
508 }
509#else
510 LOG_ASSERT_ERROR(Sim()->getConfig()->getEnablePinPlay() == false,
511 "PinPlay support not compiled in. Please use a compatible pin kit when compiling.");
512#endif
513
514 // config::Config shouldn't be called outside of init/fini
515 // With Sim()->hideCfg(), we let Simulator know to complain when someone does call Sim()->getCfg()
516 Sim()->hideCfg();
517
518 // Never returns
519 LOG_PRINT("Running program...");
520 PIN_StartProgram();
521
522 return 0;
523}
