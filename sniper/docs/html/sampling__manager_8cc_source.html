75void
76SamplingManager::disableFastForward()
77{
78 m_fastforward = false;
79 m_warmup = false;
80 SubsecondTime barrier_next = SubsecondTime::Zero();
81 SubsecondTime core_last = SubsecondTime::MaxTime();
82 for(unsigned int core_id = 0; core_id < Sim()->getConfig()->getApplicationCores(); ++core_id)
83 {
84 Core *core = Sim()->getCoreManager()->getCoreFromID(core_id);
85 core->getPerformanceModel()->setFastForward(false, true);
86 core->disableInstructionsCallback();
87 if (Sim()->getThreadManager()->isThreadRunning(core_id)) {
88 barrier_next = std::max(barrier_next, core->getPerformanceModel()->getElapsedTime());
89 core_last = std::min(core_last, core->getPerformanceModel()->getElapsedTime());
90 }
91 }
92 if (m_uncoordinated) {
93 for(UInt32 core_id = 0; core_id < Sim()->getConfig()->getApplicationCores(); core_id++)
94 {
95 // In uncoordinated mode, some cores will be behind because they didn't reach their target instruction count.
96 // Reset all their times to the same value so they won't have to catch up in detailed mode.
97 PerformanceModel *perf = Sim()->getCoreManager()->getCoreFromID(core_id)->getPerformanceModel();
98 perf->getFastforwardPerformanceModel()->incrementElapsedTime(barrier_next - perf->getElapsedTime());
99 }
100 }
101 // Tell the barrier to go to detailed mode again, but make sure it skips over the fastforwarded section of time.
102 // FIXME:
103 // We restart the barrier at the maximum time, and have all threads make uncoordinated progress
104 // towards the fastest thread. There won't be any HOOK_PERIODIC calls (which may not make much sense anyway,
105 // as during this period at least one thread is way ahead and won't be running).
106 // Alternatively, we could reset the barrier to the min time of all cores, this way, all cores that are behind
107 // the fastest one, will start to make detailed progress again in lockstep. We'll also get HOOK_PERIODIC calls
108 // during this time.
109 if (Sim()->getClockSkewMinimizationServer())
110 Sim()->getClockSkewMinimizationServer()->setFastForward(false, barrier_next);
111 this->setInstrumentationMode(InstMode::DETAILED);
112}

116{

184SamplingManager::recalibrateInstructionsCallback(core_id_t core_id)
185{
186 Core *core = Sim()->getCoreManager()->getCoreFromID(core_id);
187 SubsecondTime now = core->getPerformanceModel()->getElapsedTime();
188 if (now > m_target_ffend) {
189 // Just a single instruction so we call into the barrier immediately
190 core->setInstructionsCallback(1);
191 } else {
192 SubsecondTime cpi = core->getPerformanceModel()->getFastforwardPerformanceModel()->getCurrentCPI();
193 // If CPI hasn't been set up, fall back to 1 IPC to avoid division by zero
194 if (cpi == SubsecondTime::Zero())
195 cpi = Sim()->getCoreManager()->getCoreFromID(core_id)->getDvfsDomain()->getPeriod();
196 UInt64 ninstrs = SubsecondTime::divideRounded(m_target_ffend - now, cpi);
197 core->setInstructionsCallback(ninstrs);
