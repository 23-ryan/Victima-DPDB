50BbvCount::count(UInt64 address, UInt64 count)
51{
52 sampleReset();
53
54 m_instrs_abs += count;
55
56 // Perform random projection of basic-block vectors onto NUM_BBV dimensions
57 // As there are too many BBVs, we cannot store the projection matrix, rather,
58 // we re-generate it on request using an RNG seeded with the BBV address.
59 // Since this is a hot loop in FAST_FORWARD mode, use an inlined RNG
60 // and four parallel code paths to exploit as much ILP as possible.
61 UInt64 s0 = rng_seed(address), s1 = rng_seed(address + 1), s2 = rng_seed(address + 2), s3 = rng_seed(address + 3);
62 for(int i = 0; i < NUM_BBV; i += 4)
63 {
64 UInt64 weight = rng_next(s0);
65 m_bbv_counts_abs[i] += (weight & 0xffff) * count;
66 weight = rng_next(s1);
67 m_bbv_counts_abs[i+1] += (weight & 0xffff) * count;
68 weight = rng_next(s2);
69 m_bbv_counts_abs[i+2] += (weight & 0xffff) * count;
70 weight = rng_next(s3);
71 m_bbv_counts_abs[i+3] += (weight & 0xffff) * count;
72 }
