99void ThreadManager::onThreadStart(thread_id_t thread_id, SubsecondTime time)
100{
101 ScopedLock sl(m_thread_lock);
102 LOG_PRINT("onThreadStart(%i)", thread_id);
103
104 Thread *thread = getThreadFromID(thread_id);
105
106 m_thread_tls->set(thread);
107 thread->updateCoreTLS();
108
109 // Set thread state to running for the duration of HOOK_THREAD_START, we'll move it to stalled later on if it didn't have a core
110 m_thread_state[thread_id].status = Core::RUNNING;
111
112 HooksManager::ThreadTime args = { thread_id: thread_id, time: time };
113 Sim()->getHooksManager()->callHooks(HookType::HOOK_THREAD_START, (UInt64)&args);
114 // Note: we may have been rescheduled during HOOK_THREAD_START
115 // (Happens if core was occupied during our createThread() but became free since then)
116 CLOG("thread", "Start %d", thread_id);
117
118 Core *core = thread->getCore();
119 if (core)
120 {
121 // Set the CoreState to 'RUNNING'
122 core->setState(Core::RUNNING);
123
124 PerformanceModel *pm = core->getPerformanceModel();
125 // If the core already has a later time, we have to wait
126 time = std::max(time, pm->getElapsedTime());
127 pm->queuePseudoInstruction(new SpawnInstruction(time));
128
129 LOG_PRINT("Setting status[%i] -> RUNNING", thread_id);
130 m_thread_state[thread_id].status = Core::RUNNING;
131
132 HooksManager::ThreadMigrate args = { thread_id: thread_id, core_id: core->getId(), time: time };
133 Sim()->getHooksManager()->callHooks(HookType::HOOK_THREAD_MIGRATE, (UInt64)&args);
134 }
135 else
136 {
137 m_thread_state[thread_id].status = Core::STALLED;
138 m_thread_state[thread_id].stalled_reason = STALL_UNSCHEDULED;
139 }
140
141 if (m_thread_state[thread_id].waiter != INVALID_THREAD_ID)
142 {
143 getThreadFromID(m_thread_state[thread_id].waiter)->signal(time);
144 m_thread_state[thread_id].waiter = INVALID_THREAD_ID;

149{
150 ScopedLock sl(m_thread_lock);
151
152 LOG_ASSERT_ERROR((UInt32)thread_id < m_thread_state.size(), "Thread id out of range: %d", thread_id);
153
154 Thread *thread = getThreadFromID(thread_id);
155 Core *core = thread->getCore();
156 LOG_ASSERT_ERROR(core != NULL, "Thread ended while not running on a core?");
157
158 SubsecondTime time = core->getPerformanceModel()->getElapsedTime();

178
179 m_thread_tls->set(NULL);
180 thread->setCore(NULL);
181 thread->updateCoreTLS();

291void ThreadManager::joinThread(thread_id_t thread_id, thread_id_t join_thread_id)
292{
293 Thread *thread = getThreadFromID(thread_id);
294 Core *core = thread->getCore();
295 SubsecondTime end_time;
296
297 LOG_PRINT("Joining on thread: %d", join_thread_id);
298
299 {
300 ScopedLock sl(getLock());
301
302 if (m_thread_state[join_thread_id].status == Core::IDLE)
303 {
304 LOG_PRINT("Not running.");
305 return;
306 }
307
308 SubsecondTime start_time = core->getPerformanceModel()->getElapsedTime();
309
310 LOG_ASSERT_ERROR(m_thread_state[join_thread_id].waiter == INVALID_THREAD_ID,
311 "Multiple threads joining on thread: %d", join_thread_id);
312
313 m_thread_state[join_thread_id].waiter = thread_id;
314 end_time = stallThread(thread_id, ThreadManager::STALL_JOIN, start_time);
315 }
316
317 if (thread->reschedule(end_time, core))
318 core = thread->getCore();
319
320 core->getPerformanceModel()->queuePseudoInstruction(new SyncInstruction(end_time, SyncInstruction::JOIN));
321
322 LOG_PRINT("Exiting join thread.");
323}

350SubsecondTime ThreadManager::stallThread(thread_id_t thread_id, stall_type_t reason, SubsecondTime time)
351{
352 stallThread_async(thread_id, reason, time);
353 // When all threads are stalled, we have a deadlock -- unless we let the barrier advance time
354 // which may wake up threads that are sleeping or waiting on a futex with a timeout value.
355 while(!anyThreadRunning())
356 {
357 Sim()->getClockSkewMinimizationServer()->advance();
358 }
359 // It's possible that a HOOK_PERIODIC, called by SkewMinServer::signal(), called by stallThread_async(), woke us up again.
360 // We will then have been signal()d, but this signal was lost since we weren't in wait()
361 // If this is the case, don't go to sleep but return our wakeup time immediately
362 if (m_thread_state[thread_id].status == Core::RUNNING)
363 return getThreadFromID(thread_id)->getWakeupTime();
364 else
365 return getThreadFromID(thread_id)->wait(m_thread_lock);
366}
