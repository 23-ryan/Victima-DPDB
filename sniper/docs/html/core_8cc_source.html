234MemoryResult
235Core::readInstructionMemory(IntPtr address, UInt32 instruction_size)
236{
237 LOG_PRINT("Instruction: Address(0x%x), Size(%u), Start READ",
238 address, instruction_size);
239
240 UInt64 blockmask = ~(getMemoryManager()->getCacheBlockSize() - 1);
241 bool single_cache_line = ((address & blockmask) == ((address + instruction_size - 1) & blockmask));
242
243 // Assume the core reads full instruction cache lines and caches them internally for subsequent instructions.
244 // This reduces L1-I accesses and power to more realistic levels.
245 // For Nehalem, it's in fact only 16 bytes, other architectures (Sandy Bridge) have a micro-op cache,
246 // so this is just an approximation.
247
248 // When accessing the same cache line as last time, don't access the L1-I
249 if ((address & blockmask) == m_icache_last_block)
250 {
251 if (single_cache_line)
252 {
253 return makeMemoryResult(HitWhere::L1I, getMemoryManager()->getL1HitLatency());
254 }
255 else
256 {
257 // Instruction spanning cache lines: drop the first line, do access the second one
258 address = (address & blockmask) + getMemoryManager()->getCacheBlockSize();
259 }
260 }
261
262 // Update the most recent cache line accessed
263 m_icache_last_block = address & blockmask;
264
265 // Cases with multiple cache lines or when we are not sure that it will be a hit call into the caches
266 return initiateMemoryAccess(MemComponent::L1_ICACHE,
267 Core::NONE, Core::READ, address & blockmask, NULL, getMemoryManager()->getCacheBlockSize(), MEM_MODELED_COUNT_TLBTIME, 0, SubsecondTime::MaxTime());
268}

495Core::nativeMemOp(lock_signal_t lock_signal, mem_op_t mem_op_type, IntPtr d_addr, char* data_buffer, UInt32 data_size)
496{
497 if (data_size <= 0)
498 {
499 return makeMemoryResult(HitWhere::UNKNOWN,SubsecondTime::Zero());
500 }
501
502 if (lock_signal == LOCK)
503 {
504 assert(mem_op_type == READ_EX);
505 m_global_core_lock.acquire();
506 }
507
508 if ( (mem_op_type == READ) || (mem_op_type == READ_EX) )
509 {
510 applicationMemCopy ((void*) data_buffer, (void*) d_addr, (size_t) data_size);
511 }
512 else if (mem_op_type == WRITE)
513 {
514 applicationMemCopy ((void*) d_addr, (void*) data_buffer, (size_t) data_size);
515 }
516
517 if (lock_signal == UNLOCK)
518 {
519 assert(mem_op_type == WRITE);
520 m_global_core_lock.release();
521 }
522
523 return makeMemoryResult(HitWhere::UNKNOWN,SubsecondTime::Zero());

533Core::emulateCpuid(UInt32 eax, UInt32 ecx, cpuid_result_t &res) const
534{
535 switch(eax)
536 {
537 case 0x0:
538 {
539 cpuid(0, 0, res);
540 res.eax = std::max(UInt32(0xb), res.eax); // Maximum input eax: make sure 0xb is included
541 break;
542 }
543 case 0x1:
544 {
545 // Return native results, except for CPU id
546 cpuid(eax, ecx, res);
547 res.ebx = (m_core_id << 24) | (Sim()->getConfig()->getApplicationCores() << 16) | (res.ebx &0xffff);
548 break;
549 }
550 case 0xb:
551 {
552 // Extended Topology Enumeration Leaf
553 switch(ecx)
554 {
555 case 0:
556 // Level 0: SMT
557 res.eax = TopologyInfo::SMT_SHIFT_BITS;
558 res.ebx = m_topology_info->smt_count; // SMT threads / core
559 res.ecx = ecx | (1 << 8); // Level type = SMT
560 break;
561 case 1:
562 // Level 1: cores
563 res.eax = TopologyInfo::PACKAGE_SHIFT_BITS;
564 res.ebx = m_topology_info->smt_count * m_topology_info->core_count; // HW contexts / package
565 res.ecx = ecx | (2 << 8); // Level type = Core
566 break;
567 default:
568 // Invalid level
569 res.eax = 0;
570 res.ebx = 0;
571 res.ecx = ecx;
572 break;
573 }
574 res.edx = m_topology_info->apic_id;
575 break;
576 }
577 default:
578 {
579 // Return native results (original cpuid instruction is deleted)
580 cpuid(eax, ecx, res);
581 break;
582 }
583 }
584
585 #if VERBOSE
586 printf("CPUID[%d]: %08x %08x => ", m_core_id, eax, ecx);
587 printf("%08x %08x %08x %08x\n", res.eax, res.ebx, res.ecx, res.edx);
588 #endif
589}
