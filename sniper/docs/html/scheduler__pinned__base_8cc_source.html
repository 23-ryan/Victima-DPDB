65void SchedulerPinnedBase::threadYield(thread_id_t thread_id)
66{
67 core_id_t core_id = m_thread_info[thread_id].getCoreRunning();
68
69 if (core_id != INVALID_CORE_ID)
70 {
71 Core *core = Sim()->getCoreManager()->getCoreFromID(core_id);
72 SubsecondTime time = core->getPerformanceModel()->getElapsedTime();
73
74 m_quantum_left[core_id] = SubsecondTime::Zero();
75 reschedule(time, core_id, false);
76
77 if (!m_thread_info[thread_id].hasAffinity(core_id))
78 {
79 core_id_t free_core_id = findFreeCoreForThread(thread_id);
80 if (free_core_id != INVALID_CORE_ID)
81 {
82 // We have just been moved to a different core(s), and one of them is free. Schedule us there now.
83 reschedule(time, free_core_id, false);
84 }
85 }

89bool SchedulerPinnedBase::threadSetAffinity(thread_id_t calling_thread_id, thread_id_t thread_id, size_t cpusetsize, const cpu_set_t *mask)
90{
91 if (m_thread_info.size() <= (size_t)thread_id)
92 m_thread_info.resize(thread_id + 16);
93
94 m_thread_info[thread_id].setExplicitAffinity();
95
96 if (!mask)
97 {
98 // No mask given: free to schedule anywhere.
99 for(core_id_t core_id = 0; core_id < (core_id_t)Sim()->getConfig()->getApplicationCores(); ++core_id)
100 {
101 m_thread_info[thread_id].addAffinity(core_id);
102 }
103 }
104 else
105 {
106 m_thread_info[thread_id].clearAffinity();
107
108 for(unsigned int cpu = 0; cpu < 8 * cpusetsize; ++cpu)
109 {
110 if (CPU_ISSET_S(cpu, cpusetsize, mask))
111 {
112 LOG_ASSERT_ERROR(cpu < Sim()->getConfig()->getApplicationCores(), "Invalid core %d found in sched_setaffinity() mask", cpu);
113
114 m_thread_info[thread_id].addAffinity(cpu);
115 }
116 }
117 }
118
119 // We're setting the affinity of a thread that isn't yet created. Do nothing else for now.
120 if (thread_id >= (thread_id_t)Sim()->getThreadManager()->getNumThreads())
121 return true;
122
123 if (thread_id == calling_thread_id)
124 {
125 threadYield(thread_id);
126 }
127 else if (m_thread_info[thread_id].isRunning() // Thread is running
128 && !m_thread_info[thread_id].hasAffinity(m_thread_info[thread_id].getCoreRunning())) // but not where we want it to
129 {
130 // Reschedule the thread as soon as possible
131 m_quantum_left[m_thread_info[thread_id].getCoreRunning()] = SubsecondTime::Zero();
132 }
133 else if (m_threads_runnable[thread_id] // Thread is runnable
134 && !m_thread_info[thread_id].isRunning()) // Thread is not running (we can't preempt it outside of the barrier)
135 {
136 core_id_t free_core_id = findFreeCoreForThread(thread_id);
137 if (free_core_id != INVALID_THREAD_ID) // Thread's new core is free
138 {
139 // We have just been moved to a different core, and that core is free. Schedule us there now.
140 Core *core = Sim()->getCoreManager()->getCoreFromID(free_core_id);
141 SubsecondTime time = std::max(core->getPerformanceModel()->getElapsedTime(), Sim()->getClockSkewMinimizationServer()->getGlobalTime());
142 reschedule(time, free_core_id, false);
143 }
144 }
145
146 return true;
147}

149bool SchedulerPinnedBase::threadGetAffinity(thread_id_t thread_id, size_t cpusetsize, cpu_set_t *mask)
150{
151 if (cpusetsize*8 < Sim()->getConfig()->getApplicationCores())
152 {
153 // Not enough space to return result
154 return false;
155 }
156
157 CPU_ZERO_S(cpusetsize, mask);
158 for(core_id_t core_id = 0; core_id < (core_id_t)Sim()->getConfig()->getApplicationCores(); ++core_id)
159 {
160 if (
161 m_thread_info[thread_id].hasAffinity(core_id)
162 // When application has not yet done any sched_setaffinity calls, lie and return a fully populated affinity bitset.
163 // This makes libiomp5 use all available cores.
164 || !m_thread_info[thread_id].hasExplicitAffinity()
165 )
166 CPU_SET_S(core_id, cpusetsize, mask);
167 }
168
169 return true;
170}

202void SchedulerPinnedBase::periodic(SubsecondTime time)
203{
204 SubsecondTime delta = time - m_last_periodic;
205
206 for(core_id_t core_id = 0; core_id < (core_id_t)Sim()->getConfig()->getApplicationCores(); ++core_id)
207 {
208 if (delta > m_quantum_left[core_id] || m_core_thread_running[core_id] == INVALID_THREAD_ID)
209 {
210 reschedule(time, core_id, true);
211 }
212 else
213 {
214 m_quantum_left[core_id] -= delta;
215 }
216 }
217
218 m_last_periodic = time;

221void SchedulerPinnedBase::reschedule(SubsecondTime time, core_id_t core_id, bool is_periodic)
222{
223 thread_id_t current_thread_id = m_core_thread_running[core_id];
224
225 if (current_thread_id != INVALID_THREAD_ID
226 && Sim()->getThreadManager()->getThreadState(current_thread_id) == Core::INITIALIZING)
227 {
228 // Thread on this core is starting up, don't reschedule it for now
229 return;
230 }
231
232 thread_id_t new_thread_id = INVALID_THREAD_ID;
233 SInt64 max_score = INT64_MIN;
234
235 for(thread_id_t thread_id = 0; thread_id < (thread_id_t)m_threads_runnable.size(); ++thread_id)
236 {
237 if (m_thread_info[thread_id].hasAffinity(core_id) // Thread is allowed to run on this core
238 && m_threads_runnable[thread_id] == true // Thread is not stalled
239 && (!m_thread_info[thread_id].isRunning() // Thread is not already running somewhere else
240 || m_thread_info[thread_id].getCoreRunning() == core_id)
241 )
242 {
243 SInt64 score;
244 if (m_thread_info[thread_id].isRunning())
245 // Thread is currently running: negative score depending on how long it's already running
246 score = SInt64(m_thread_info[thread_id].getLastScheduledIn().getPS()) - time.getPS();
247 else
248 // Thread is not currently running: positive score depending on how long we have been waiting
249 score = time.getPS() - SInt64(m_thread_info[thread_id].getLastScheduledOut().getPS());
250
251 // Find thread that was scheduled the longest time ago
252 if (score > max_score)
253 {
254 new_thread_id = thread_id;
255 max_score = score;
256 }
257 }
258 }
259
260 if (current_thread_id != new_thread_id)
261 {
262 // If a thread was running on this core, and we'll schedule another one, unschedule the current one
263 if (current_thread_id != INVALID_THREAD_ID)
264 {
265 m_thread_info[current_thread_id].setCoreRunning(INVALID_CORE_ID);
266 // Update last scheduled out time, with a small extra penalty to make sure we don't
267 // reconsider this thread in the same periodic() call but for a next core
268 m_thread_info[current_thread_id].setLastScheduledOut(time + SubsecondTime::PS(core_id));
269 moveThread(current_thread_id, INVALID_CORE_ID, time);
270 }
271
272 // Set core as running this thread *before* we call moveThread(), otherwise the HOOK_THREAD_RESUME callback for this
273 // thread might see an empty core, causing a recursive loop of reschedulings
274 m_core_thread_running[core_id] = new_thread_id;
275
276 // If we found a new thread to schedule, move it here
277 if (new_thread_id != INVALID_THREAD_ID)
278 {
279 // If thread was running somewhere else: let that core know
280 if (m_thread_info[new_thread_id].isRunning())
281 m_core_thread_running[m_thread_info[new_thread_id].getCoreRunning()] = INVALID_THREAD_ID;
282 // Move thread to this core
283 m_thread_info[new_thread_id].setCoreRunning(core_id);
284 m_thread_info[new_thread_id].setLastScheduledIn(time);
285 moveThread(new_thread_id, core_id, time);
286 }
287 }
288
289 m_quantum_left[core_id] = m_quantum;
290}
