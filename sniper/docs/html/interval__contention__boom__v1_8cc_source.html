39{
40 DynamicMicroOpBoomV1::uop_port_t reason = DynamicMicroOpBoomV1::UOP_PORT_SIZE;
41 uint64_t effective_cp_length = critical_path_length;
42
43 // For the standard ports, check if we have exceeded our execution limit
44 for (unsigned int i = 0 ; i < DynamicMicroOpBoomV1::UOP_PORT_SIZE ; i++)
45 {
46 // Skip shared ports
47 if (i != DynamicMicroOpBoomV1::UOP_PORT012 && effective_cp_length < m_count_byport[i]
48 )
49 {
50 effective_cp_length = m_count_byport[i];
51 reason = (DynamicMicroOpBoomV1::uop_port_t)i;
52 }
53 }
54 // Check shared port usage
55 uint64_t port012_use = m_count_byport[DynamicMicroOpBoomV1::UOP_PORT0] + m_count_byport[DynamicMicroOpBoomV1::UOP_PORT1]
56 + m_count_byport[DynamicMicroOpBoomV1::UOP_PORT2] + m_count_byport[DynamicMicroOpBoomV1::UOP_PORT012];
57 if (port012_use > (3*effective_cp_length))
58 {
59 effective_cp_length = (port012_use+2) / 3; // +2 to round up to the next cycle
60 reason = DynamicMicroOpBoomV1::UOP_PORT012;
61 }
62
63 if (update_reason && effective_cp_length > critical_path_length)
64 {
65 LOG_ASSERT_ERROR(reason != DynamicMicroOpBoomV1::UOP_PORT_SIZE, "There should be a reason for the cp extention, but there isn't");
66 m_cpContrByPort[reason] += 1000000 * (effective_cp_length - critical_path_length) / effective_cp_length;
