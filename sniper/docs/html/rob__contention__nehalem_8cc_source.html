29bool RobContentionNehalem::tryIssue(const DynamicMicroOp &uop)
30{
31 // Port contention
32 // TODO: Maybe the scheduler is more intelligent and doesn't just assing the first uop in the ROB
33 // that fits a particular free port. We could give precedence to uops that have dependants, etc.
34 // NOTE: mixes canIssue and doIssue, in the sense that ports* are incremented immediately.
35 // This works as long as, if we return true, this microop is indeed issued
36
37 const DynamicMicroOpNehalem *core_uop_info = uop.getCoreSpecificInfo<DynamicMicroOpNehalem>();
38 DynamicMicroOpNehalem::uop_port_t uop_port = core_uop_info->getPort();
39 if (uop_port == DynamicMicroOpNehalem::UOP_PORT015)
40 {
41 if (ports_generic >= 3)
42 return false;
43 else
44 ports_generic++;
45 }
46 else if (uop_port == DynamicMicroOpNehalem::UOP_PORT05)
47 {
48 if (ports_generic05 >= 2)
49 return false;
50 else
51 ports_generic05++;
52 }
53 else if (uop_port == DynamicMicroOpNehalem::UOP_PORT2 || uop_port == DynamicMicroOpNehalem::UOP_PORT34)
54 {
55 if (ports[uop_port])
56 return false;
57 else
58 ports[uop_port] = true;
59 }
60 else
61 { // PORT0, PORT1 or PORT5
62 if (ports[uop_port])
63 return false;
64 else if (ports_generic >= 3)
65 return false;
66 else if (uop_port != DynamicMicroOpNehalem::UOP_PORT1 && ports_generic05 >= 2)
67 return false;
68 else
69 {
70 ports[uop_port] = true;
71 ports_generic++;
72 if (uop_port != DynamicMicroOpNehalem::UOP_PORT1)
73 ports_generic05++;
74 }
75 }
76
77 // ALU contention
78 if (DynamicMicroOpNehalem::uop_alu_t alu = core_uop_info->getAlu())
79 {
80 if (alu_used_until[alu] > m_now)
81 return false;
82 }
83
84 return true;
85}

87void RobContentionNehalem::doIssue(DynamicMicroOp &uop)
88{
89 const DynamicMicroOpNehalem *core_uop_info = uop.getCoreSpecificInfo<DynamicMicroOpNehalem>();
90 DynamicMicroOpNehalem::uop_alu_t alu = core_uop_info->getAlu();

95bool RobContentionNehalem::noMore()
96{
97 // When we issued something to all ports in this cycle, stop walking the rest of the ROB
98 if (ports[DynamicMicroOpNehalem::UOP_PORT2] && ports[DynamicMicroOpNehalem::UOP_PORT34] && ports_generic >= 3)
99 return true;
100 else
101 return false;
