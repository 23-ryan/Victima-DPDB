112static PyObject *
113registerHook(PyObject *self, PyObject *args)
114{
115 int hook = -1;
116 PyObject *pFunc = NULL;
117
118 if (!PyArg_ParseTuple(args, "lO", &hook, &pFunc))
119 return NULL;
120
121 if (hook < 0 || hook >= HookType::HOOK_TYPES_MAX) {
122 PyErr_SetString(PyExc_ValueError, "Hook type out of range");
123 return NULL;
124 }
125 if (!PyCallable_Check(pFunc)) {
126 PyErr_SetString(PyExc_TypeError, "Second argument must be callable");
127 return NULL;
128 }
129
130 Py_INCREF(pFunc);
131
132 HookType::hook_type_t type = HookType::hook_type_t(hook);
133 switch(type) {
134 case HookType::HOOK_PERIODIC:
135 Sim()->getHooksManager()->registerHook(type, hookCallbackSubsecondTime, (UInt64)pFunc);
136 break;
137 case HookType::HOOK_SIM_START:
138 case HookType::HOOK_SIM_END:
139 case HookType::HOOK_ROI_BEGIN:
140 case HookType::HOOK_ROI_END:
141 case HookType::HOOK_APPLICATION_ROI_BEGIN:
142 case HookType::HOOK_APPLICATION_ROI_END:
143 case HookType::HOOK_SIGUSR1:
144 Sim()->getHooksManager()->registerHook(type, hookCallbackNone, (UInt64)pFunc);
145 break;
146 case HookType::HOOK_PERIODIC_INS:
147 case HookType::HOOK_CPUFREQ_CHANGE:
148 case HookType::HOOK_INSTR_COUNT:
149 case HookType::HOOK_INSTRUMENT_MODE:
150 case HookType::HOOK_APPLICATION_START:
151 case HookType::HOOK_APPLICATION_EXIT:
152 Sim()->getHooksManager()->registerHook(type, hookCallbackInt, (UInt64)pFunc);
153 break;
154 case HookType::HOOK_PRE_STAT_WRITE:
155 Sim()->getHooksManager()->registerHook(type, hookCallbackString, (UInt64)pFunc);
156 break;
157 case HookType::HOOK_MAGIC_MARKER:
