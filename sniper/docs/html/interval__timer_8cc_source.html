179// Simulate a collection of micro-ops and report the number of instructions executed and the latency
180boost::tuple<uint64_t,uint64_t> IntervalTimer::simulate(const std::vector<DynamicMicroOp*>& insts)
181{
182 uint64_t total_instructions_executed = 0, total_latency = 0;
183
184 for (std::vector<DynamicMicroOp*>::const_iterator i = insts.begin() ; i != insts.end(); ++i )
185 {
186 if ((*i)->isSquashed())
187 {
188 delete *i;
189 continue;
190 }
191
192 // Enforce memory dependency granularity
193 if ((*i)->getMicroOp()->isLoad() || (*i)->getMicroOp()->isStore())
194 {
195 Memory::Access addr = (*i)->getAddress();
196 addr.address &= m_mem_dep_mask;
197 (*i)->setAddress(addr);
198 }
199
200 m_windows->add(*i);
201 m_uop_type_count[(*i)->getMicroOp()->getSubtype()]++;
202 m_uops_total++;
203 if ((*i)->getMicroOp()->isX87()) m_uops_x87++;
204 if ((*i)->getMicroOp()->isPause()) m_uops_pause++;
205
206 if (m_uops_total > 10000 && m_uops_x87 > m_uops_total / 20)
207 {
208 LOG_PRINT_WARNING_ONCE("Significant fraction of x87 instructions encountered, accuracy will be low. Compile without -mno-sse2 -mno-sse3 to avoid these instructions.");
209 }
210
211 // Only dispatch instructions from the window when it is full
212 // It needs to be full so that we can walk the window to find independent misses
213 while (m_windows->wIsFull())
214 {
215 uint64_t instructions_executed, latency;
216 boost::tie(instructions_executed, latency) = dispatchWindow();
217 total_instructions_executed += instructions_executed;
218 total_latency += latency;
219 }
220 }
221
222 return boost::tuple<uint64_t,uint64_t>(total_instructions_executed, total_latency);
223}

225boost::tuple<uint64_t, uint64_t> IntervalTimer::dispatchWindow() {
226 uint64_t instructions_executed = 0;
227 uint64_t latency = 0;
228 uint64_t micro_ops_executed = 0;
229
230 uint32_t dispatch_rate = calculateCurrentDispatchRate();
231
232 StopDispatchReason continue_dispatching = STOP_DISPATCH_NO_REASON;
233
234 SubsecondTime micro_op_period = m_frequency_domain->getPeriod(); // Approximate the current cycle count if we don't find a MicroOp
235
236 // Instruction dispatch
237 // micro_ops_executed must be < dispatch_rate. micro_ops_executed <= dispatch_rate will cause significant speedups (see calculateCurrentDispatchRate())
238 while ( (!m_windows->wIsEmpty()) && (instructions_executed < m_dispatch_width) && (micro_ops_executed < dispatch_rate) && (continue_dispatching == STOP_DISPATCH_NO_REASON))
239 {
240 Windows::WindowEntry& micro_op = m_windows->getInstructionToDispatch();
241
242 uint64_t instruction_latency = dispatchInstruction(micro_op, continue_dispatching);
243 latency += instruction_latency;
244 m_windows->dispatchInstruction();
245
246 micro_ops_executed++;
247 if (micro_op.getDynMicroOp()->isLast())
248 {
249 instructions_executed++;
250 }
251
252 m_core->getPerformanceModel()->traceInstruction(micro_op.getDynMicroOp(), NULL);
253
254#if DEBUG_IT_INSN_PRINT
255 if (latency > 16)
256 {
257 uint64_t insn_count = m_perf_model->getInstructionCount();
258 uint64_t cycle_count = m_perf_model->getCycleCount();
259# ifdef ENABLE_MICROOP_STRINGS
260 const char *opcode_name = micro_op.getMicroOp()->getInstructionOpcodeName().c_str();
261# else
262 const char *opcode_name = "Unknown";
263# endif
264 fprintf(m_insn_log, "[%ld,%ld] %s latency=%d\n", cycle_count, insn_count, opcode_name, instruction_latency);
265 }
266#endif
267
268 micro_op_period = micro_op.getDynMicroOp()->getPeriod();
269 }
270
271 // The minimum latency for dispatching these micro-ops is 1 cycle
272 // Detect the microarchitectural limiting factors for this cycle for cpi accounting
273 if (latency == 0)
274 {
275 if (m_windows->wIsEmpty())
276 {
277 continue_dispatching = ADD_STOP_DISPATCH_REASON(STOP_DISPATCH_WINDOW_EMPTY, continue_dispatching);
278 }
279 if (instructions_executed >= m_dispatch_width)
280 {
281 continue_dispatching = ADD_STOP_DISPATCH_REASON(STOP_DISPATCH_DISPATCH_WIDTH, continue_dispatching);
282 }
283 if (micro_ops_executed >= dispatch_rate)
284 {
285 continue_dispatching = ADD_STOP_DISPATCH_REASON(STOP_DISPATCH_DISPATCH_RATE, continue_dispatching);
286 // Each time we have to stop because the critical path tells us to:
287 // Add the current critical path components to their corresponding total
288 // If the critical path got extended due to issue contention, add that separately
289 int critical_path_length = m_windows->getCriticalPathLength();
290 int effective_cp_length = m_windows->getEffectiveCriticalPathLength(critical_path_length, true);
291 for(unsigned int i = 0; i < (unsigned int)CPCONTR_TYPE_SIZE; ++i)
292 m_cpContrByType[i] += m_windows->getCpContrFraction((CpContrType)i, effective_cp_length);
293 #if 0
294 printf("cpContr: ");
295 for(unsigned int i = 0; i < (unsigned int)CPCONTR_TYPE_SIZE; ++i)
296 printf("%s %3lu ", CpContrTypeString((CpContrType)i).c_str(), m_windows->m_cpcontr_bytype[i]);
297 printf("/ total %4lu\n", m_windows->m_cpcontr_total);
298 #endif
299 }
300
301 latency = 1;
302 // Update CPI-stacks
303 m_cpiBase += 1 * micro_op_period;
304 m_cpiBaseStopDispatch[continue_dispatching] += 1;
305 }
306
307 return boost::tuple<uint64_t,uint64_t>(instructions_executed, latency);
308}

310uint32_t IntervalTimer::calculateCurrentDispatchRate() {
311
312 int critical_path_length = m_windows->getCriticalPathLength();
313
314 uint32_t dispatch_rate;
315
316 if (critical_path_length > 0)
317 {
318 FixedPoint ipc = FixedPoint(m_windows->getOldWindowLength()) / m_windows->getEffectiveCriticalPathLength(critical_path_length, false) + m_remaining_dispatch_bandwidth;
319 dispatch_rate = FixedPoint::floor(ipc);
320 m_remaining_dispatch_bandwidth = (dispatch_rate < m_dispatch_width) ? (ipc - dispatch_rate) : 0;
321 }
322 else
323 {
324 dispatch_rate = m_dispatch_width;
325 m_remaining_dispatch_bandwidth = 0;
326 }
327
328 return dispatch_rate;
329}

331void IntervalTimer::issueMemOp(Windows::WindowEntry& micro_op)
332{
333 // Issue memory operations to the memory hierarchy.
334 // This function is called:
335 // - not at all if all memory operations are issued at fetch (perf_model/core/interval_timer/issue_memops_at_dispatch == false)
336 // - from blockWindow for overlapping accesses
337 // - from dispatchInstruction otherwise
338 if ((micro_op.getMicroOp()->isLoad() || micro_op.getMicroOp()->isStore())
339 && micro_op.getDynMicroOp()->getDCacheHitWhere() == HitWhere::UNKNOWN)
340 {
341 MemoryResult res = m_core->accessMemory(
342 Core::NONE,
343 micro_op.getMicroOp()->isLoad() ? Core::READ : Core::WRITE,
344 micro_op.getDynMicroOp()->getAddress().address,
345 NULL,
346 micro_op.getMicroOp()->getMemoryAccessSize(),
347 Core::MEM_MODELED_RETURN,
348 micro_op.getMicroOp()->getInstruction() ? micro_op.getMicroOp()->getInstruction()->getAddress() : static_cast<uint64_t>(NULL)
349 );
350 uint64_t latency = SubsecondTime::divideRounded(res.latency, m_core->getDvfsDomain()->getPeriod());
351 micro_op.getDynMicroOp()->setExecLatency(micro_op.getDynMicroOp()->getExecLatency() + latency); // execlatency already contains bypass latency
352 micro_op.getDynMicroOp()->setDCacheHitWhere(res.hit_where);
353 }

356uint64_t IntervalTimer::dispatchInstruction(Windows::WindowEntry& micro_op, StopDispatchReason& continue_dispatching)
357{
358 // If it's not already done, issue the memory operation
359 issueMemOp(micro_op);
360
361 uint64_t latency = 0;
362
363 uint64_t max_producer_exec_time = getMaxProducerExecTime(micro_op);
364 micro_op.maxProducer = max_producer_exec_time;
365 micro_op.cphead = m_windows->getCriticalPathHead();
366 micro_op.cptail = m_windows->getCriticalPathTail();
367
368 bool icache_miss = (micro_op.getDynMicroOp()->getICacheHitWhere() != HitWhere::L1I) & (!micro_op.hasOverlapFlag(Windows::WindowEntry::ICACHE_OVERLAP));
369
370 if (icache_miss)
371 {
372 uint64_t icache_latency = micro_op.getDynMicroOp()->getICacheLatency();
373 latency += icache_latency;
374
375 m_windows->clearOldWindow(micro_op.cptail + icache_latency);
376
377 continue_dispatching = STOP_DISPATCH_ICACHE_MISS;
378 // Update icache CPI-stack counters
379 m_cpiInstructionCache[micro_op.getDynMicroOp()->getICacheHitWhere()] += icache_latency * micro_op.getDynMicroOp()->getPeriod();
380 }
381
382 if (micro_op.getMicroOp()->isBranch())
383 {
384 if (micro_op.getDynMicroOp()->isBranchMispredicted() && !micro_op.hasOverlapFlag(Windows::WindowEntry::BPRED_OVERLAP))
385 {
386 uint64_t bpred_latency = m_branch_misprediction_penalty + m_windows->calculateBranchResolutionLatency();
387 latency += bpred_latency;
388
389 continue_dispatching = STOP_DISPATCH_BRANCH_MISPREDICT;
390 m_windows->clearOldWindow(micro_op.cptail + bpred_latency);
391 // Update bpred CPI-stack counters
392 m_cpiBranchPredictor += bpred_latency * micro_op.getDynMicroOp()->getPeriod();
393 }
394 }
395
396 if (micro_op.getMicroOp()->isSerializing() || micro_op.getMicroOp()->isInterrupt())
397 {
398 uint64_t flushLatency = std::max(m_windows->getCriticalPathLength(), m_windows->getMinimalFlushLatency(m_dispatch_width));
399 uint64_t serialize_latency = flushLatency + micro_op.getDynMicroOp()->getExecLatency();
400 latency += serialize_latency;
401
402 m_numSerializationInsns++;
403 m_totalSerializationLatency += serialize_latency;
404 m_cpiSerialization += serialize_latency * micro_op.getDynMicroOp()->getPeriod();
405
406 micro_op.setExecTime(m_windows->getCriticalPathTail());
407 m_windows->clearOldWindow(micro_op.getExecTime());
408 }
409 else if (micro_op.getMicroOp()->isExecute() && micro_op.getMicroOp()->isMemBarrier())
410 {
411 // Handle the M/L/SFENCE operations (when supplied as a single execute operation) now as a strict MFENCE
412 //uint64_t current_cycle = m_windows->getCriticalPathTail();
413 uint64_t cycle_to_wait_until = std::max(max_producer_exec_time,std::max(m_max_store_completion_time,m_max_load_completion_time));
414 // If we have loads or stores that will complete in the future, wait for them all to complete
415 // Otherwise, our critical path is longer, and we become a nop
416 // This code below doesn't work. CriticalPath is just the time that we put the last instruction in.
417 // We need to use the maxProd, and we can't just add latencies, because we don't have a concept of latency
418 // in the old-window (only execution ciompletion time)
419 // We just need to make sure that the memory dependencies are properly taken care of.
420 //if (cycle_to_wait_until > current_cycle) {
421 // uint64_t mfence_latency = cycle_to_wait_until - current_cycle;
422 // // Keep track of the memory fence instructions and their contributed latencies
423 // m_totalMfenceLatency += mfence_latency;
424 //}
425 m_numMfenceInsns++;
426 micro_op.setExecTime(cycle_to_wait_until + micro_op.getDynMicroOp()->getExecLatency());
427 updateCriticalPath(micro_op, latency);
428 }
429 else
430 {
431 uint64_t exec_latency = micro_op.getDynMicroOp()->getExecLatency();
432
433 if (micro_op.getMicroOp()->isLoad())
434 {
435 if (micro_op.hasOverlapFlag(Windows::WindowEntry::DCACHE_OVERLAP))
436 {
437 // do nothing for overlapped loads, as they don't add to the critical path and happen in parallel to other loads (MLP > 1)
438 }
439 else if (micro_op.getDynMicroOp()->isLongLatencyLoad())
440 {
441 uint64_t dcache_latency = exec_latency;
442 // Long latency loads trump all other latencies
443 uint64_t dispatch_cycle = std::max(max_producer_exec_time, m_windows->getCriticalPathHead());
444 uint64_t sched_cycle = dispatch_cycle;
445 uint64_t contention_exec_cycle;
446 // This logic only works for LFENCEs, when marked on load uops
447 if (micro_op.getMicroOp()->isMemBarrier())
448 contention_exec_cycle = m_loadstore_contention.getBarrierCompletionTime(sched_cycle, dcache_latency);
449 else
450 contention_exec_cycle = m_loadstore_contention.getCompletionTime(sched_cycle, dcache_latency);
451
452 // Calculate our new latency from the load contention completion time
453 uint64_t reswin_extra_latency = sched_cycle - dispatch_cycle;
454 uint64_t contention_extra_latency = contention_exec_cycle - sched_cycle;
455 uint64_t long_latency_load_latency = reswin_extra_latency + contention_extra_latency; // dcache_latency is already taken into account in contention_extra_latency
456 latency += long_latency_load_latency;
457
458#if DEBUG_IT_INSN_PRINT
459 uint64_t insn_count = m_perf_model->getInstructionCount();
460 uint64_t cycle_count = m_perf_model->getCycleCount();
461# ifdef ENABLE_MICROOP_STRINGS
462 const char *opcode_name = micro_op.getInstructionOpcodeName().c_str();
463# else
464 const char *opcode_name = "Unknown";
465# endif
466 fprintf(m_insn_log, "[%ld,%ld] %s(", cycle_count, insn_count, opcode_name);
467 if (micro_op.isLoad())
468 fprintf(m_insn_log, "L");
469 if (micro_op.isExecute())
470 fprintf(m_insn_log, "X");
471 if (micro_op.isStore())
472 fprintf(m_insn_log, "S");
473 fprintf(m_insn_log, ") latency=%ld (%ld+%ld(%ld))\n", latency, reswin_extra_latency, contention_extra_latency, dcache_latency);
474#endif
475
476 micro_op.setExecTime(contention_exec_cycle);
477
478 micro_op.setExecTime(m_windows->getCriticalPathTail());
479 m_windows->clearOldWindow(micro_op.getExecTime() + long_latency_load_latency);
480
481 this->blockWindow();
482
483 m_numLongLatencyLoads++;
484 m_numTotalLongLatencyLoadLatency+=long_latency_load_latency;
485
486 // Update dcache CPI-stack counters
487 m_cpiDataCache[micro_op.getDynMicroOp()->getDCacheHitWhere()] += long_latency_load_latency * micro_op.getDynMicroOp()->getPeriod();
488 }
489 else
490 {
491 // Non long-latency-load load operations
492 uint64_t dcache_latency = exec_latency;
493 uint64_t dispatch_cycle = std::max(max_producer_exec_time, m_windows->getCriticalPathHead());
494 uint64_t sched_cycle = dispatch_cycle;
495 uint64_t contentionExecTime;
496 // This logic only works for LFENCEs, when marked on load uops
497 if (micro_op.getMicroOp()->isMemBarrier())
498 contentionExecTime = m_loadstore_contention.getBarrierCompletionTime(sched_cycle, dcache_latency);
499 else
500 contentionExecTime = m_loadstore_contention.getCompletionTime(sched_cycle, dcache_latency);
501 micro_op.setExecTime(contentionExecTime);
502 /* doesn't block window but adds to the critical path (unless long-latency). */
503 updateCriticalPath(micro_op, latency);
504 }
505
506 m_max_load_completion_time = std::max(m_max_load_completion_time, micro_op.getExecTime());
507
508 // Compute MLP
509 if (micro_op.getDynMicroOp()->isLongLatencyLoad())
510 {
511 uint64_t now = m_windows->getCriticalPathTail();
512 uint64_t done = now + exec_latency;
513
514 // Ins will be outstanding for until it is done. By accounting beforehand I don't need to
515 // worry about fast-forwarding simulations
516 m_outstandingLongLatencyInsns += exec_latency;
517
518 // Only account for the cycles that have not yet been accounted for by other long
519 // latency misses (don't account cycles twice).
520 if (m_lastAccountedMemoryCycle < now)
521 {
522 m_lastAccountedMemoryCycle = now;
523 }
524 if (done > m_lastAccountedMemoryCycle)
525 {
526 m_outstandingLongLatencyCycles += done - m_lastAccountedMemoryCycle;
527 m_lastAccountedMemoryCycle = done;
528 }
529 }
530
531 } else if (micro_op.getMicroOp()->isStore()) {
532
533 uint64_t store_latency = exec_latency;
534 uint64_t dispatch_cycle = std::max(max_producer_exec_time, m_windows->getCriticalPathHead());
535 uint64_t sched_cycle = dispatch_cycle;
536 uint64_t bypass_latency = m_core_model->getBypassLatency(micro_op.getDynMicroOp());
537 uint64_t data_ready_cycle = sched_cycle + bypass_latency + 1; // This store result will be ready to use one cycle later
538 micro_op.setExecTime(data_ready_cycle); // Time the critical path calculation will use
539 updateCriticalPath(micro_op, latency);
540
541 // FIXME See Redmine issue #89: CPU forwarding logic vs. RFO store completion delaying the CPU
542 uint64_t exec_time_cycle = sched_cycle + store_latency; // For future instructions that depend on this instruction's result
543 m_max_store_completion_time = std::max(m_max_store_completion_time, exec_time_cycle);
544
545 // FIXME: Don't update micro_op.setExecTime (else it will eventually affect the critical path head, once this instruction
546 // goes through Windows::dispatchInstruction). If we need this time, figure out another place to put it.
547 //micro_op.setExecTime(exec_time_cycle);
548
549 } else {
550 uint64_t dispatch_cycle = std::max(max_producer_exec_time, m_windows->getCriticalPathHead());
551 micro_op.setExecTime(dispatch_cycle + exec_latency);
552
553 updateCriticalPath(micro_op, latency);
554 }
555 }
556
557 return latency;
558}

564void IntervalTimer::updateCriticalPath(Windows::WindowEntry& micro_op, uint64_t& latency)
565{
566 uint64_t lll = m_windows->longLatencyOperationLatency(micro_op);
567 if (lll == 0)
568 {
569 m_windows->updateCriticalPathTail(micro_op);
570 }
571 else

574 m_cpiLongLatency += lll * micro_op.getDynMicroOp()->getPeriod();
575 m_windows->clearOldWindow(micro_op.getExecTime());
576 }

583void IntervalTimer::blockWindow()
584{
585 // Calculate the overlaps for all instructions in the window.
586 bool mem_barrier_pending = false;
587
588 Windows::Iterator window_iterator = m_windows->getWindowIterator();
589
590 Windows::WindowEntry& head = window_iterator.next(); // Returns the current head: disregard it.
591 head.setIndependentMiss();
592
593 IntPtr head_address = head.getMicroOp()->isLoad() ? head.getDynMicroOp()->getLoadAccess().phys : 0;
594 head_address &= m_lll_dep_mask;
595
596 while(window_iterator.hasNext()) {
597 Windows::WindowEntry& micro_op = window_iterator.next();
598
599 micro_op.clearDependent();
600 micro_op.addOverlapFlag(Windows::WindowEntry::ICACHE_OVERLAP);
601 m_numICacheOverlapped++;
602
603 if (micro_op.getMicroOp()->isSerializing())
604 break;
605
606 if (micro_op.getMicroOp()->isMemBarrier())
607 mem_barrier_pending = true;
608
609 // Generate dependencies using dependencies
610 for(uint32_t i = 0; i < micro_op.getDynMicroOp()->getDependenciesLength(); i++)
611 {
612 if (m_windows->windowContains(micro_op.getDynMicroOp()->getDependency(i)))
613 {
614 Windows::WindowEntry& dependee = m_windows->getInstruction(micro_op.getDynMicroOp()->getDependency(i));
615 if (dependee.isDependent())
616 {
617 // Dependee depends on the long-latency load blocking the window: do not issue this uop now
618 micro_op.setDataDependent();
619 break;
620 }
621 else if (dependee.isIndependent() && dependee.getDynMicroOp()->isLongLatencyLoad())
622 {
623 // Our dependee is independent of the long-latency load blocking the window,
624 // but it is a long-latency event by itself: do not issue this uop now
625 // Since the window head is independent and long-latency,
626 // this code path will also start the chain of dependent loads.
627 micro_op.setDataDependent();
628 break;
629 }
630 // else: our dependee is independent of the long-latency load blocking the window,
631 // and is not a long-latency load in itself, which means it will complete under the original LLL.
632 // Therefore, we can also be hidden under the long-latency load which makes us not dependent.
633 }
634 }
635
636 if (micro_op.getMicroOp()->isLoad())
637 {
638 IntPtr address = micro_op.getDynMicroOp()->getLoadAccess().phys;
639 if ((address & m_lll_dep_mask) == head_address)
640 {
641 // This load accesses the same cache line as the long-latency load that's blocking the ROB
642 // We should be an overlapped miss, but due to the way the cache model works (instant completion),
643 // we will see an L1 hit. Still, we can't complete until the miss is resolved.
644 // Model this by making us dependent on the long-latency load.
645 micro_op.setDataDependent();
646 }
647 }
648
649 if (micro_op.getMicroOp()->isBranch() && !micro_op.isDependent())
650 {
651 micro_op.addOverlapFlag(Windows::WindowEntry::BPRED_OVERLAP);
652 m_numBPredOverlapped++;
653 }
654 else if (micro_op.getMicroOp()->isLoad() && !micro_op.isDependent())
655 {
656 /* no previous membars & not data dependend, mark as overlapped
657 * if long latency miss -> mark as independent miss
658 * if short latency -> check if delayed hit and also mark as independent (secondary) miss
659 */
660 if (!mem_barrier_pending)
661 {
662 if (!micro_op.hasOverlapFlag(Windows::WindowEntry::DCACHE_OVERLAP))
663 {
664 uint64_t overlappedLatency = micro_op.getDynMicroOp()->getExecLatency();
665 uint64_t longLatency = head.getDynMicroOp()->getExecLatency();
666 m_totalHiddenDCacheLatency += overlappedLatency;
667
668 if (overlappedLatency > longLatency)
669 {
670 m_totalHiddenLongerDCacheLatency += overlappedLatency - longLatency;
671 m_numHiddenLongerDCacheLatency += 1;
672 }
673
674 micro_op.addOverlapFlag(Windows::WindowEntry::DCACHE_OVERLAP);
675 m_numDCacheOverlapped++;
676
677 // Issue the memory operation now
678 issueMemOp(micro_op);
679
680 micro_op.setIndependentMiss();
681 micro_op.setExecTime(m_windows->getCriticalPathHead());
682 }
683 }
684 }
685 }
686}

688// Allow the use of negative times for comparisons
689uint64_t IntervalTimer::getMaxProducerExecTime(Windows::WindowEntry& micro_op) {
690 int64_t oldestStartTime = m_windows->getOldestInstruction().getExecTime() - m_windows->getOldestInstruction().getDynMicroOp()->getExecLatency();
691 int64_t oldestFetchTime = m_windows->getOldestInstruction().getFetchTime();
692 int64_t currentFetchTime = micro_op.getFetchTime();
693 int64_t relativeFetchTime = currentFetchTime - oldestFetchTime;
694
695 int64_t max_producer_exec_time = 0;
696
697 for(uint32_t i = 0; i < micro_op.getDynMicroOp()->getDependenciesLength(); i++) {
698 if (m_windows->oldWindowContains(micro_op.getDynMicroOp()->getDependency(i))) {
699 Windows::WindowEntry& producer = m_windows->getInstruction(micro_op.getDynMicroOp()->getDependency(i));
700 int64_t producerExecTime = producer.getExecTime();
701 int64_t producerStartTime = producerExecTime - producer.getDynMicroOp()->getExecLatency();
702 int64_t relativeStartTime = producerStartTime - oldestStartTime;
703
704 if (relativeFetchTime > relativeStartTime) {
705 if (relativeStartTime < 0)
706 relativeStartTime = 0;
707 producerExecTime += (relativeFetchTime - relativeStartTime);
708 }
709
710 max_producer_exec_time = std::max(max_producer_exec_time, producerExecTime);
711 }
712 }
713
714 return max_producer_exec_time;
715}
