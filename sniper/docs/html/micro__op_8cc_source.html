123MicroOp::uop_subtype_t MicroOp::getSubtype_Exec(const MicroOp& uop)
124{
125 dl::Decoder *dec = Sim()->getDecoder();
126
127 // Get the uop subtype for the EXEC part of this instruction
128 // (ignoring the fact that this particular microop may be a load/store,
129 // used in determining the data type for load/store when calculating bypass delays)
130 if (dec->is_branch_opcode(uop.getInstructionOpcode()))
131 return UOP_SUBTYPE_BRANCH;
132
133 else if (dec->is_fpvector_addsub_opcode(uop.getInstructionOpcode(), uop.getDecodedInstruction()))
134 return UOP_SUBTYPE_FP_ADDSUB;
135
136 else if (dec->is_fpvector_muldiv_opcode(uop.getInstructionOpcode(), uop.getDecodedInstruction()))
137 return UOP_SUBTYPE_FP_MULDIV;
138
139 else
140 return UOP_SUBTYPE_GENERIC;
141}

144MicroOp::uop_subtype_t MicroOp::getSubtype(const MicroOp& uop)
145{
146 // Count all of the ADD/SUB/DIV/LD/ST/BR, and if we have too many, break
147 // Count all of the GENERIC insns, and if we have too many (3x-per-cycle), break
148 if (uop.isLoad())
149 return UOP_SUBTYPE_LOAD;
150 else if (uop.isStore())
151 return UOP_SUBTYPE_STORE;
152 else if (uop.isBranch()) // conditional branches
153 return UOP_SUBTYPE_BRANCH;
154 else if (uop.isExecute())
155 return getSubtype_Exec(uop);
156 else
157 return UOP_SUBTYPE_GENERIC;
158}

181bool MicroOp::isFpLoadStore() const
182{
183 if (isLoad() || isStore())
184 {

187 case UOP_SUBTYPE_FP_ADDSUB:
188 case UOP_SUBTYPE_FP_MULDIV:
