86void DynamicMicroOp::removeDependency(uint64_t sequenceNumber)
87{
88 if (sequenceNumber >= this->sequenceNumber - this->microOpTypeOffset - this->intraInstructionDependencies) {
89 // Intra-instruction dependency
90 while(intraInstructionDependencies && !(sequenceNumber == this->sequenceNumber - this->microOpTypeOffset - this->intraInstructionDependencies)) {
91 // Remove the first intra-instruction dependency, but since this is not the one to be removed, add it to the regular dependencies list
92 dependencies[dependenciesLength] = this->sequenceNumber - this->microOpTypeOffset - this->intraInstructionDependencies;
93 dependenciesLength++;
94 LOG_ASSERT_ERROR(dependenciesLength < MAXIMUM_NUMBER_OF_DEPENDENCIES, "dependenciesLength(%u) > MAX(%u)", dependenciesLength, MAXIMUM_NUMBER_OF_DEPENDENCIES);
95 intraInstructionDependencies--;
96 }
97 // Make sure the exit condition was that the dependency to be removed is now the first one, not that we have exhausted the list
98 LOG_ASSERT_ERROR(intraInstructionDependencies > 0, "Something went wrong while removing an intra-instruction dependency");
99 // Remove the first intra-instruction dependency by decrementing intraInstructionDependencies
100 intraInstructionDependencies--;
101 } else {
102 // Inter-instruction dependency
103 LOG_ASSERT_ERROR(dependenciesLength > 0, "Cannot remove dependency when there are none");
104 if (dependencies[dependenciesLength-1] == sequenceNumber)
105 ; // sequenceNumber to remove is already at the end, we can just decrement dependenciesLength
106 else {
107 // Move sequenceNumber to the end of the list
108 uint64_t idx = Tools::index(dependencies, dependenciesLength, sequenceNumber);
109 LOG_ASSERT_ERROR(idx != UINT64_MAX, "MicroOp dependency list does not contain %ld", sequenceNumber);
110 Tools::swap(dependencies, idx, dependenciesLength-1);
111 }
112 dependenciesLength--;
113 }
