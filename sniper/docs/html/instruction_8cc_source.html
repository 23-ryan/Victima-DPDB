39// An instruction cost is the number of cycles it takes to execute the instruction, times the
40// period of the processor that we are currently running on
41SubsecondTime Instruction::getCost(Core *core) const
42{
43 LOG_ASSERT_ERROR(m_type < MAX_INSTRUCTION_COUNT, "Unknown instruction type: %d", m_type);
44 const ComponentPeriod *period = core->getDvfsDomain();
45 return static_cast<SubsecondTime>(*period) * Instruction::m_instruction_costs[m_type];
46}
