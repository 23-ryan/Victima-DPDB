1/*************************************************************************************************/
2/* SCHEDULER SEQUENTIAL */
3/* ==================== */
4/* This scheduler can allocate sequentially sets of threads in one or more than one cores. */
5/* considering every thread of a set, sequential chunks of the same process. */
6/* Every process is allocated on a different core */
7/* */
8/* It has been developed thinking in pinballs and the way that Sniper consider every pinball as */
9/* a different application. */
10/* */
11/* To set-up the pinballs sets that conforms an entire process and for gather the outputs it */
12/* was added new parameters. */
13/* - scheduler/sequential/sequence : To indicate the sets of pinballs/threads that belongs */
14/* to the same process. */
15/* - scheduler/sequential/sequence_file : If it is not setted, the gathered data will be */
16/* showed on the standard output. In the indicated file aversely. */
17/* */
18/* Example: We have 3 processes and split every one in 2 pinballs. */
19/* ./run-sniper */
20/* --pinballs=pb1,pb2,pb3,pb4,pb5,pb6 */
21/* -g --scheduler/type=sequential */
22/* -g --scheduler/sequential/sequence="2|2|2" */
23/* -g --scheduler/sequential/sequence_file="FILE.OUT" */
24/* --sim-end=last */
25/* -n 3 */
26/*************************************************************************************************/

46SchedulerSequential::SchedulerSequential(ThreadManager *thread_manager)

59 boost::split(aux_seqs, sequence, boost::is_any_of("|"));

66 "Not enought cores. Every pinballs set must be executed on a different core.");
67
68 core_waiting_threads.resize(Sim()->getConfig()->getApplicationCores());
69
70 // Setting the first Thread of every core

72 next_thread_to_execute.resize(Sim()->getConfig()->getApplicationCores());
73
74 for(long unsigned int i=0; i < next_thread_to_execute.size(); ++i)

76 next_thread_to_execute.at(i) = next_tte;

82

99void SchedulerSequential::threadStart(thread_id_t thread_id, SubsecondTime time)
100{
101 total_pinballs++;
102 core_id_t core_id = (core_id_t)atoi(
103 m_thread_info[thread_id].getAffinityString().c_str() );
104
105 if (/*m_core_thread_running[core_id] == INVALID_THREAD_ID && */
106 /*thread_id == next_thread_to_execute.at(core_id)*/
107 m_core_thread_running[core_id] == thread_id)
108 {
109 // Only the first thread of every core execute this
110 next_thread_to_execute.at(core_id)++;
111 // Seems that the first thread of every core has been allocated,
112 // then rescheduler is not needed.
113 //reschedule(time, core_id, false);
114 }
115 else if (m_core_thread_running[core_id] != thread_id)
116 {
117 aux_mm << "Pushed to core queue " << core_id;
118 print_message(thread_id, aux_mm.str().c_str());
119 aux_mm.str("");
120
121 m_threads_runnable[thread_id] = false;
122 core_waiting_threads[core_id].push(thread_id);
123 }

126void SchedulerSequential::threadExit(thread_id_t thread_id, SubsecondTime time)
127{
128 print_message(thread_id, "Finnish it's job.");
129 m_threads_runnable[thread_id] = false;
130
131 core_id_t current_core_id = (core_id_t)atoi( m_thread_info[thread_id].getAffinityString().c_str() );
132
133 // Updating info, ended thread
134 Sim()->getThreadStatsManager()->update(thread_id, time);
135 m_thread_info[thread_id].setLastScheduledOut(time);
136
137 // Alocating next thread on queue
138 if ( !core_waiting_threads[current_core_id].empty() )
139 {
140 thread_id_t next_thread = core_waiting_threads[current_core_id].top();
141
142 // I am not sure if the first thread can end its job while the initialization
143 // of the other threads is executing and if the initialization of threads could
144 // be in a disordered way respect the --pinballs option as well
145 if ( next_thread != next_thread_to_execute.at(current_core_id) )
146 return;
147
148 core_waiting_threads[current_core_id].pop();
149 m_threads_runnable[next_thread] = true;
150 next_thread_to_execute.at(current_core_id)++;
151
152 //Sim()->getThreadStatsManager()->update(next_thread, time);
153 reschedule(time, current_core_id, false);
154 print_message(next_thread, "Scheduled.");
155 }
