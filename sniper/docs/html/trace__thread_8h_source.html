35 // In multi-process mode, we want each process to have its own private memory space
36 // Therefore, perform a virtual to physical address mapping by including the core_id
37 // Virtual addresses are converted to physical addresses by pasting the core_id at
38 // bit positions pa_core_shift..+pa_core_size
39 // The highest virtual address used is normally 00007fffffffffff, with pa_core_shift==48
40 // the core_id is 2 above the highest used bit.
41 static const UInt64 pa_core_shift = 48;
42 static const UInt64 pa_core_size = 16;
43 static const UInt64 pa_va_mask = ~(((UInt64(1) << pa_core_size) - 1) << pa_core_shift);
44 // Optionally we can also do address randomization on a per-page basis.
