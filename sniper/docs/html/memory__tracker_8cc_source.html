136UInt64 MemoryTracker::ce_get_owner(core_id_t core_id, UInt64 address)
137{
138 ScopedLock sl(m_lock);
139 AllocationSite *owner = NULL;
140
141 // upper_bound returns the first entry greater than address
142 // Because the key in m_allocations is the first cache line that no longer falls into the range,
143 // we will find the correct alloction *if* address falls within it
144 auto upper = m_allocations.upper_bound(address);
145 if (upper != m_allocations.end() && address >= upper->first - upper->second.size)
146 owner = upper->second.site;
147
148 #ifdef ASSERT_FIND_OWNER
149 AllocationSite *owner_slow = (m_allocations_slow.count(address & ~63) == 0) ? NULL : m_allocations_slow[address & ~63];
150 LOG_ASSERT_WARNING(owner == owner_slow, "ASSERT_FIND_OWNER: owners for %lx don't match (fast %p != slow %p)", address, owner, owner_slow);
151 #endif
152
153 return (UInt64)owner;
154}
