99std::pair<bool, SubsecondTime>
100DramCache::doAccess(Cache::access_t access, IntPtr address, core_id_t requester, Byte* data_buf, SubsecondTime now, ShmemPerf *perf)
101{
102 PrL1CacheBlockInfo* block_info = (PrL1CacheBlockInfo*)m_cache->peekSingleLine(address);
103 SubsecondTime latency = m_tags_access_time;
104 perf->updateTime(now);
105 perf->updateTime(now + latency, ShmemPerf::DRAM_CACHE_TAGS);
106 bool cache_hit = false, prefetch_hit = false;
107
108 if (block_info)
109 {
110 cache_hit = true;
111
112 if (block_info->hasOption(CacheBlockInfo::PREFETCH))
113 {
114 // This line was fetched by the prefetcher and has proven useful
115 m_hits_prefetch++;
116 prefetch_hit = true;
117 block_info->clearOption(CacheBlockInfo::PREFETCH);
118
119 // If prefetch is still in progress: delay
120 SubsecondTime t_completed = m_prefetch_mshr.getTagCompletionTime(address);
121 if (t_completed != SubsecondTime::MaxTime() && t_completed > now + latency)
122 {
123 m_prefetch_mshr_delay += t_completed - (now + latency);
124 latency = t_completed - now;
125 }
126 }
127
128 m_cache->accessSingleLine(address, access, data_buf, m_cache_block_size, now + latency, true);
129
130 latency += accessDataArray(access, requester, now + latency, perf);
131 if (access == Cache::STORE)
132 block_info->setCState(CacheState::MODIFIED);
133 }
134 else
135 {
136 if (access == Cache::LOAD)
137 {
138 // For LOADs, get data from DRAM
139 SubsecondTime dram_latency;
140 HitWhere::where_t hit_where;
141 boost::tie(dram_latency, hit_where) = m_dram_cntlr->getDataFromDram(address, requester, data_buf, now + latency, perf,false);
142 latency += dram_latency;
143 }
144 // For STOREs, we only do complete cache lines so we don't need to read from DRAM
145
146 insertLine(access, address, requester, data_buf, now + latency);
147 }
148
149 if (m_prefetcher)
150 callPrefetcher(address, cache_hit, prefetch_hit, now + latency);
151
152 return std::pair<bool, SubsecondTime>(block_info ? true : false, latency);
153}

178SubsecondTime
179DramCache::accessDataArray(Cache::access_t access, core_id_t requester, SubsecondTime t_start, ShmemPerf *perf)
180{
181 SubsecondTime processing_time = m_data_array_bandwidth.getRoundedLatency(8 * m_cache_block_size); // bytes to bits
182
183 // Compute Queue Delay
184 SubsecondTime queue_delay;
185 if (m_queue_model)
186 {
187 queue_delay = m_queue_model->computeQueueDelay(t_start, processing_time, requester);
188 }
189 else
190 {
191 queue_delay = SubsecondTime::Zero();
192 }
193
194 perf->updateTime(t_start);
195 perf->updateTime(t_start + queue_delay, ShmemPerf::DRAM_CACHE_QUEUE);
196 perf->updateTime(t_start + queue_delay + processing_time, ShmemPerf::DRAM_CACHE_BUS);
197 perf->updateTime(t_start + queue_delay + processing_time + m_data_access_time, ShmemPerf::DRAM_CACHE_DATA);
198
199 return queue_delay + processing_time + m_data_access_time;
200}

202void
203DramCache::callPrefetcher(IntPtr train_address, bool cache_hit, bool prefetch_hit, SubsecondTime t_issue)
204{
205 // Always train the prefetcher
206 std::vector<IntPtr> prefetchList = m_prefetcher->getNextAddress(train_address, INVALID_CORE_ID,Core::INVALID_MEM_OP, cache_hit, prefetch_hit, 0xdeadbeef);
207
208 // Only do prefetches on misses, or on hits to lines previously brought in by the prefetcher (if enabled)
209 if (!cache_hit || (m_prefetch_on_prefetch_hit && prefetch_hit))
210 {
211 for(std::vector<IntPtr>::iterator it = prefetchList.begin(); it != prefetchList.end(); ++it)
212 {
213 IntPtr prefetch_address = *it;
214 if (!m_cache->peekSingleLine(prefetch_address))
215 {
216 // Get data from DRAM
217 SubsecondTime dram_latency;
218 HitWhere::where_t hit_where;
219 Byte data_buf[m_cache_block_size];
220 boost::tie(dram_latency, hit_where) = m_dram_cntlr->getDataFromDram(prefetch_address, m_core_id, data_buf, t_issue, NULL,false);
221 // Insert into data array
222 insertLine(Cache::LOAD, prefetch_address, m_core_id, data_buf, t_issue + dram_latency);
223 // Set prefetched bit
224 PrL1CacheBlockInfo* block_info = (PrL1CacheBlockInfo*)m_cache->peekSingleLine(prefetch_address);
225 block_info->setOption(CacheBlockInfo::PREFETCH);
226 // Update completion time
227 m_prefetch_mshr.getCompletionTime(t_issue, dram_latency, prefetch_address);
228
229 ++m_prefetches;
230 }
231 }
232 }
