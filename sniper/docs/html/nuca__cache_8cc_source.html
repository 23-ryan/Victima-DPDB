170NucaCache::accessDataArray(Cache::access_t access, SubsecondTime t_start, ShmemPerf *perf)
171{
172 perf->updateTime(t_start);
173
174 // Compute Queue Delay
175 SubsecondTime queue_delay;
176 if (m_queue_model)
177 {
178 SubsecondTime processing_time = m_data_array_bandwidth.getRoundedLatency(8 * m_cache_block_size); // bytes to bits
179
180 queue_delay = processing_time + m_queue_model->computeQueueDelay(t_start, processing_time, m_core_id);
181
182 perf->updateTime(t_start + processing_time, ShmemPerf::NUCA_BUS);
183 perf->updateTime(t_start + queue_delay, ShmemPerf::NUCA_QUEUE);
184 }
185 else
186 {
187 queue_delay = SubsecondTime::Zero();
188 }
189
190 perf->updateTime(t_start + queue_delay + m_data_access_time.getLatency(), ShmemPerf::NUCA_DATA);
191
