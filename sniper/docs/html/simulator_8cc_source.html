56void Simulator::createDecoder()
57{
58 // Set up the decoder object if it does not exist yet
59 if (! m_decoder)
60 {
61 // Get architecture
62 dl::dl_arch dla;
63 String architecture = Sim()->getCfg()->getString("general/arch");
64 if (architecture == "intel")
65 dla = dl::DL_ARCH_INTEL;
66 else if (architecture == "riscv")
67 dla = dl::DL_ARCH_RISCV;
68 else
69 LOG_PRINT_ERROR("Unknown architecture %s, should be intel or arm.", architecture.c_str());
70 // Get mode
71 dl::dl_mode dlm;
72 String mode = Sim()->getCfg()->getString("general/mode");
73 if (mode == "64")
74 dlm = dl::DL_MODE_64;
75 else if (mode == "32")
76 dlm = dl::DL_MODE_32;
77 else
78 LOG_PRINT_ERROR("Unknown mode %s, should be 32 or 64 (bits).", mode.c_str());
79 // Get syntax
80 dl::dl_syntax dls;
81 String syntax = Sim()->getCfg()->getString("general/syntax");
82 if (dla == dl::DL_ARCH_RISCV)
83 dls = dl::DL_SYNTAX_DEFAULT;
84 else if (syntax == "intel")
85 dls = dl::DL_SYNTAX_INTEL;
86 else if (syntax == "att")
87 dls = dl::DL_SYNTAX_ATT;
88 else if (syntax == "xed")
89 dls = dl::DL_SYNTAX_XED;
90 else
91 LOG_PRINT_ERROR("Unknown assembly syntax %s, should be intel, att or xed.", syntax.c_str());
92
93 m_factory = new dl::DecoderFactory;
94 m_decoder = m_factory->CreateDecoder(dla, dlm, dls); // create decoder for [arch, mode, syntax]
95 }

298void Simulator::setInstrumentationMode(InstMode::inst_mode_t new_mode, bool update_barrier)
299{
300 if (new_mode != InstMode::inst_mode)
301 {
302 if (m_inst_mode_output && InstMode::inst_mode != InstMode::INVALID)
303 {
304 printf("[SNIPER] Setting instrumentation mode to %s\n", inst_mode_names[new_mode]); fflush(stdout);
305 }
306 InstMode::inst_mode = new_mode;
307
308 if (Sim()->getConfig()->getSimulationMode() == Config::PINTOOL)
309 InstMode::updateInstrumentationMode();
310
311 // If there is a fast-forward performance model, it needs to take care of barrier synchronization.
312 // If we're called with update_barrier == false, the caller (SamplingManager) manages the barrier.
313 // Else, disable the barrier in fast-forward/cache-only
314 if (update_barrier && !Sim()->getFastForwardPerformanceManager())
315 getClockSkewMinimizationServer()->setDisable(new_mode != InstMode::DETAILED);
316
317 Sim()->getHooksManager()->callHooks(HookType::HOOK_INSTRUMENT_MODE, (UInt64)new_mode);
318 }
