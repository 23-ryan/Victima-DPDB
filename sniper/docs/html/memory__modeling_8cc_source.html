233ADDRINT handleMemoryReadFaultinjection(THREADID thread_id, BOOL executing, ADDRINT eip, bool is_atomic_update, ADDRINT read_address, UInt32 read_data_size, UInt32 op_num, ADDRINT *save_ea)
234{
235 *save_ea = read_address;
236
237 if (is_atomic_update)
238 g_atomic_lock.acquire();
239
240 Core* core = localStore[thread_id].thread->getCore();
241
242 if (executing)
243 {
244 char buf_fault[1024];
245 char *buf_data = localStore[thread_id].scratch[op_num];
246
247 // Load fault mask from simulated memory
248 MemoryResult memres = core->accessMemory(
249 /*(is_atomic_update) ? Core::LOCK :*/ Core::NONE,
250 (is_atomic_update) ? Core::READ_EX : Core::READ,
251 read_address,
252 buf_fault,
253 read_data_size,
254 eip ? Core::MEM_MODELED_RETURN : Core::MEM_MODELED_COUNT,
255 eip,
256 SubsecondTime::MaxTime(),
257 true);
258
259 if (eip)
260 {
261 assert(localStore[thread_id].dynins);
262 localStore[thread_id].dynins->addMemory(executing, memres.latency, read_address, read_data_size, Operand::READ, 0, memres.hit_where);
263 }
264
265 // Load correct data from real memory
266 PIN_SafeCopy(buf_data, (const void*)read_address, read_data_size);
267
268 // Apply the fault mask to the data
269 Sim()->getFaultinjectionManager()->applyFault(core, read_address, read_data_size, memres, (Byte*)buf_data, (const Byte*)buf_fault);
270
271 // Address to do the actual read from: read from our data buffer
272 return (ADDRINT)buf_data;
273 }
274 else if (eip)
275 {
276 // CMOV with false predicate, no actual write, but performance model expects a MemoryInfo
277 assert(localStore[thread_id].dynins);
278 localStore[thread_id].dynins->addMemory(false, SubsecondTime::Zero(), read_address, read_data_size, Operand::READ, 0, HitWhere::PREDICATE_FALSE);
279 }
280
281 return read_address;
282}

292void handleMemoryWrite(THREADID thread_id, BOOL executing, ADDRINT eip, bool is_atomic_update, IntPtr write_address, UInt32 write_data_size)
293{
294 /* Optimization for atomic instructions: we know the second (write) access will be a hit. Also, since this is Lite mode,
295 we don't need to write back the data. Therefore, only tell the cache to add a hit to its counters and push the
296 appropriate DynamicInstructionInfo, but don't keep the cache hierarchy locked for potentially a very long time. */
297 Core* core = localStore[thread_id].thread->getCore();
298 if (is_atomic_update && executing)
299 core->logMemoryHit(false, Core::WRITE, write_address, Core::MEM_MODELED_COUNT, eip);
300 else if (executing)
301 core->accessMemoryFast(false, Core::WRITE, write_address);
302}
