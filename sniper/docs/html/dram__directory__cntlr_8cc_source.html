319DramDirectoryCntlr::processNullifyReq(ShmemReq* shmem_req)
320{
321 IntPtr address = shmem_req->getShmemMsg()->getAddress();
322 core_id_t requester = shmem_req->getShmemMsg()->getRequester();
323
324 // NOTE: no ShmemPerf accounting for nullify requests as they may happen *after* core requests are completed
325
326 MYLOG("Start @ %lx", address);
327
328 DirectoryEntry* directory_entry = m_dram_directory_cache->getDirectoryEntry(address);
329 assert(directory_entry);
330
331 DirectoryBlockInfo* directory_block_info = directory_entry->getDirectoryBlockInfo();
332 DirectoryState::dstate_t curr_dstate = directory_block_info->getDState();
333
334 switch (curr_dstate)
335 {
336 case DirectoryState::EXCLUSIVE:
337 case DirectoryState::MODIFIED:
338 //std::cout << "Nullify request: " << __LINE__ << " with address = " << address << std::endl;
339 getMemoryManager()->sendMsg(ShmemMsg::FLUSH_REQ,
340 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
341 requester /* requester */,
342 directory_entry->getOwner() /* receiver */,
343 address,
344 NULL, 0,
345 HitWhere::UNKNOWN,
346 &m_dummy_shmem_perf,
347 ShmemPerfModel::_SIM_THREAD, shmem_req->getBlockType());
348 break;
349
350 case DirectoryState::SHARED:
351
352 {
353 std::pair<bool, std::vector<SInt32> > sharers_list_pair = directory_entry->getSharersList();
354 if (sharers_list_pair.first == true)
355 {
356 // Broadcast Invalidation Request to all cores
357 // (irrespective of whether they are sharers or not)
358
359 getMemoryManager()->broadcastMsg(ShmemMsg::INV_REQ,
360 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
361 requester /* requester */,
362 address,
363 NULL, 0,
364 NULL,
365 ShmemPerfModel::_SIM_THREAD);
366 }
367 else
368 {
369 // Send Invalidation Request to only a specific set of sharers
370 for (UInt32 i = 0; i < sharers_list_pair.second.size(); i++)
371 {
372 getMemoryManager()->sendMsg(ShmemMsg::INV_REQ,
373 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
374 requester /* requester */,
375 sharers_list_pair.second[i] /* receiver */,
376 address,
377 NULL, 0,
378 HitWhere::UNKNOWN,
379 &m_dummy_shmem_perf,
380 ShmemPerfModel::_SIM_THREAD,shmem_req->getBlockType());
381 }
382 }
383 }
384 break;
385
386 case DirectoryState::UNCACHED:
387
388 {
389 m_dram_directory_cache->invalidateDirectoryEntry(address);
390
391 // Process Next Request
392 processNextReqFromL2Cache(address);
393 }
394 break;
395
396 default:
397 LOG_PRINT_ERROR("Unsupported Directory State: %u", curr_dstate);
398 break;
399 }
400
401 MYLOG("End @ %lx", address);
402}

404void
405DramDirectoryCntlr::processExReqFromL2Cache(ShmemReq* shmem_req, Byte* cached_data_buf)
406{
407 IntPtr address = shmem_req->getShmemMsg()->getAddress();
408 core_id_t requester = shmem_req->getShmemMsg()->getRequester();
409
410 MYLOG("Start @ %lx", address);
411 updateShmemPerf(shmem_req);
412
413 DirectoryEntry* directory_entry = m_dram_directory_cache->getDirectoryEntry(address);
414 if (directory_entry == NULL)
415 {
416 //std::cout<<"Process Directory Entry Allocation Req : "<<__LINE__<<std::endl;
417 directory_entry = processDirectoryEntryAllocationReq(shmem_req);
418 }
419
420 DirectoryBlockInfo* directory_block_info = directory_entry->getDirectoryBlockInfo();
421 DirectoryState::dstate_t curr_dstate = directory_block_info->getDState();
422
423 updateShmemPerf(shmem_req, ShmemPerf::TD_ACCESS);
424
425 switch (curr_dstate)
426 {
427 case DirectoryState::EXCLUSIVE: // Cache may have done a silent upgrade to MODIFIED, so send a FLUSH (dirty data) rather than an INV (data clean)
428 case DirectoryState::MODIFIED:
429 {
430 assert(cached_data_buf == NULL);
431 //std::cout << "Dram request: " << __LINE__ <<std::endl;
432 MYLOG("Send FLUSH_REQ>%d for %lx", directory_entry->getOwner(), address )
433 getMemoryManager()->sendMsg(ShmemMsg::FLUSH_REQ,
434 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
435 requester /* requester */,
436 directory_entry->getOwner() /* receiver */,
437 address,
438 NULL, 0,
439 HitWhere::UNKNOWN, shmem_req->getShmemMsg()->getPerf(), ShmemPerfModel::_SIM_THREAD,shmem_req->getBlockType());
440 break;
441 }
442
443 case DirectoryState::SHARED:
444 {
445 assert(cached_data_buf == NULL);
446 std::pair<bool, std::vector<SInt32> > sharers_list_pair = directory_entry->getSharersList();
447 if (sharers_list_pair.first == true)
448 {
449 // Broadcast Invalidation Request to all cores
450 // (irrespective of whether they are sharers or not)
451 getMemoryManager()->broadcastMsg(ShmemMsg::INV_REQ,
452 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
453 requester /* requester */,
454 address,
455 NULL, 0,
456 NULL, // No ShmemPerf on broadcast
457 ShmemPerfModel::_SIM_THREAD);
458 }
459 else
460 {
461 // Send Invalidation Request to only a specific set of sharers
462 for (UInt32 i = 0; i < sharers_list_pair.second.size(); i++)
463 {
464 MYLOG("Send INV_REQ>%d for %lx", sharers_list_pair.second[i], address )
465 getMemoryManager()->sendMsg(ShmemMsg::INV_REQ,
466 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
467 requester /* requester */,
468 sharers_list_pair.second[i] /* receiver */,
469 address,
470 NULL, 0,
471 HitWhere::UNKNOWN,
472 i == 0 ? shmem_req->getShmemMsg()->getPerf() : &m_dummy_shmem_perf,
473 ShmemPerfModel::_SIM_THREAD,shmem_req->getBlockType());
474 }
475 }
476 break;
477 }
478
479 case DirectoryState::UNCACHED:
480 {
481 // Modifiy the directory entry contents
482 bool add_result = directory_entry->addSharer(requester, m_dram_directory_cache->getMaxHwSharers());
483 assert(add_result == true);
484 directory_entry->setOwner(requester);
485 directory_block_info->setDState(DirectoryState::MODIFIED);
486
487 retrieveDataAndSendToL2Cache(ShmemMsg::EX_REP, requester, address, cached_data_buf, shmem_req->getShmemMsg());
488 break;
489 }
490
491 default:
492 LOG_PRINT_ERROR("Unsupported Directory State: %u", curr_dstate);
493 break;
494 }
495 MYLOG("End @ %lx", address);
496}

498void
499DramDirectoryCntlr::processShReqFromL2Cache(ShmemReq* shmem_req, Byte* cached_data_buf)
500{
501 IntPtr address = shmem_req->getShmemMsg()->getAddress();
502 core_id_t requester = shmem_req->getShmemMsg()->getRequester();
503
504 MYLOG("Start @ %lx", address);
505 updateShmemPerf(shmem_req);
506
507 DirectoryEntry* directory_entry = m_dram_directory_cache->getDirectoryEntry(address);
508 if (directory_entry == NULL)
509 {
510 //std::cout<<"Process Directory Entry Allocation Req : "<<__LINE__<<std::endl;
511 directory_entry = processDirectoryEntryAllocationReq(shmem_req);
512 }
513
514 DirectoryBlockInfo* directory_block_info = directory_entry->getDirectoryBlockInfo();
515 DirectoryState::dstate_t curr_dstate = directory_block_info->getDState();
516
517 updateShmemPerf(shmem_req, ShmemPerf::TD_ACCESS);
518
519 switch (curr_dstate)
520 {
521 case DirectoryState::EXCLUSIVE:
522 {
523 assert (requester != directory_entry->getOwner());
524 MYLOG("WB_REQ>%d for %lx", directory_entry->getOwner(), address )
525 assert(cached_data_buf == NULL);
526 getMemoryManager()->sendMsg(ShmemMsg::WB_REQ,
527 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
528 requester /* requester */,
529 directory_entry->getOwner() /* receiver */,
530 address,
531 NULL, 0,
532 HitWhere::UNKNOWN, shmem_req->getShmemMsg()->getPerf(), ShmemPerfModel::_SIM_THREAD,shmem_req->getBlockType());
533 break;
534 }
535
536 case DirectoryState::MODIFIED:
537 {
538 MYLOG("WB_REQ>%d for %lx", directory_entry->getOwner(), address )
539 assert(cached_data_buf == NULL);
540 getMemoryManager()->sendMsg(ShmemMsg::WB_REQ,
541 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
542 requester /* requester */,
543 directory_entry->getOwner() /* receiver */,
544 address,
545 NULL, 0,
546 HitWhere::UNKNOWN, shmem_req->getShmemMsg()->getPerf(), ShmemPerfModel::_SIM_THREAD,shmem_req->getBlockType());
547 break;
548 }
549
550 case DirectoryState::SHARED:
551 {
552 if (directory_entry->hasSharer(requester))
553 {
554 MYLOG("got a WB/INV REP from the forwarder and now handling the original SH REQ");
555 ++forward;
556 if (cached_data_buf == NULL)
557 {
558 // Forwarder evicted the data while we requested it. Will have to get it from DRAM anyway.
559 ++forward_failed;
560 }
561 retrieveDataAndSendToL2Cache(ShmemMsg::SH_REP, requester, address, cached_data_buf, shmem_req->getShmemMsg());
562 }
563 else
564 {
565 bool add_result = directory_entry->addSharer(requester, m_dram_directory_cache->getMaxHwSharers());
566 if (add_result == false)
567 {
568 core_id_t sharer_id = directory_entry->getOneSharer();
569 // Send a message to another sharer to invalidate that
570 MYLOG("INV_REQ>%d for %lx because I could not add sharer", directory_entry->getOwner(), address )
571 getMemoryManager()->sendMsg(ShmemMsg::INV_REQ,
572 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
573 requester /* requester */,
574 sharer_id /* receiver */,
575 address,
576 NULL, 0,
577 HitWhere::UNKNOWN, shmem_req->getShmemMsg()->getPerf(), ShmemPerfModel::_SIM_THREAD,shmem_req->getBlockType());
578 }
579 else
580 {
581 MYLOG("SHARED state, retrieve data and send")
582 retrieveDataAndSendToL2Cache(ShmemMsg::SH_REP, requester, address, cached_data_buf, shmem_req->getShmemMsg());
583 }
584 }
585 break;
586 }
587
588 case DirectoryState::UNCACHED:
589 {
590 MYLOG("was UNCACHED, is now EXCLUSIVE")
591 // Modifiy the directory entry contents
592 bool add_result = directory_entry->addSharer(requester, m_dram_directory_cache->getMaxHwSharers());
593 assert(add_result == true);
594
595 directory_entry->setOwner(requester);
596
597 if (m_protocol == CoherencyProtocol::MSI)
598 {
599 directory_block_info->setDState(DirectoryState::SHARED);
600 retrieveDataAndSendToL2Cache(ShmemMsg::SH_REP, requester, address, cached_data_buf, shmem_req->getShmemMsg());
601 }
602 else
603 {
604 directory_block_info->setDState(DirectoryState::EXCLUSIVE);
605 retrieveDataAndSendToL2Cache(ShmemMsg::EX_REP, requester, address, cached_data_buf, shmem_req->getShmemMsg());
606 }
607
608 break;
609 }
610
611 default:
612 LOG_PRINT_ERROR("Unsupported Directory State: %u", curr_dstate);
613 break;
614 }
615 MYLOG("End @ %lx", address);
616}

621{
622 DirectoryEntry* directory_entry = m_dram_directory_cache->getDirectoryEntry(address);
623 assert(directory_entry != NULL);
624
625 MYLOG("Start @ %lx", address);
626 if (cached_data_buf != NULL)
627 {
628 // Forwarder state moves to the last cache receiving a copy,

634 //std::cout << "Dram request: " << __LINE__ <<std::endl;
635 getMemoryManager()->sendMsg(reply_msg_type,
636 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
637 receiver /* requester */,
638 receiver /* receiver */,
639 address,
640 cached_data_buf, getCacheBlockSize(),
641 HitWhere::CACHE_REMOTE /* cached_data_buf was filled by a WB_REQ or FLUSH_REQ */,
642 orig_shmem_msg->getPerf(),
643 ShmemPerfModel::_SIM_THREAD,orig_shmem_msg->getBlockType());
644
645 // Process Next Request
646 processNextReqFromL2Cache(address);
647 }
648 else
649 {
650 if (m_nuca_cache)
651 {
652 SubsecondTime nuca_latency;
653 HitWhere::where_t hit_where;
654 Byte nuca_data_buf[getCacheBlockSize()];
655 boost::tie(nuca_latency, hit_where) = m_nuca_cache->read(address, nuca_data_buf, getShmemPerfModel()->getElapsedTime(ShmemPerfModel::_SIM_THREAD), orig_shmem_msg->getPerf(), true,orig_shmem_msg->getBlockType());
656
657 getShmemPerfModel()->incrElapsedTime(nuca_latency, ShmemPerfModel::_SIM_THREAD);
658
659 if (hit_where != HitWhere::MISS)
660 {
661 getMemoryManager()->sendMsg(reply_msg_type,
662 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
663 receiver /* requester */,
664 receiver /* receiver */,
665 address,
666 nuca_data_buf, getCacheBlockSize(),
667 HitWhere::NUCA_CACHE,
668 orig_shmem_msg->getPerf(),
669 ShmemPerfModel::_SIM_THREAD,orig_shmem_msg->getBlockType());
670
671 // Process Next Request
672 processNextReqFromL2Cache(address);
673
674 return;
675 }
676 }
677
678 assert(m_dram_directory_req_queue_list->size(address) > 0);
679 ShmemReq* shmem_req = m_dram_directory_req_queue_list->front(address);
680
681 // MESIF protocol: get data from a sharing cache
682
683 if (m_protocol == CoherencyProtocol::MESIF
684 && shmem_req->getForwardingFrom() == INVALID_CORE_ID // If forwarding already failed once, don't try again
685 && directory_entry->getDirectoryBlockInfo()->getDState() == DirectoryState::SHARED
686 && directory_entry->getForwarder() != INVALID_CORE_ID)
687 {
688 core_id_t forwarder = directory_entry->getForwarder();
689 shmem_req->setForwardingFrom(forwarder);
690
691 // Send WB_REQ to forwarder to have it send us the data
692 getMemoryManager()->sendMsg(ShmemMsg::WB_REQ,
693 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
694 receiver /* requester */,
695 forwarder /* receiver */,
696 address,
697 NULL, 0,
698 HitWhere::UNKNOWN, shmem_req->getShmemMsg()->getPerf(), ShmemPerfModel::_SIM_THREAD,shmem_req->getBlockType());
699 return;
700 }
701
702 // Get the data from DRAM
703 // This could be directly forwarded to the cache or passed
704 // through the Dram Directory Controller
705
706 if (shmem_req->getShmemMsg()->getMsgType() == ShmemMsg::UPGRADE_REQ)
707 {
708 MYLOG("Have to get data from DRAM while doing an UPGRADE_REQ: lost data somehow\n");
709 }
710
711 // Remember that this request is waiting for data, and should not be woken up by voluntary invalidates
712 shmem_req->setWaitForData(true);
713
714 core_id_t dram_node = m_dram_controller_home_lookup->getHome(address);
715
716 MYLOG("Sending request to DRAM for the data");
717 getMemoryManager()->sendMsg(PrL1PrL2DramDirectoryMSI::ShmemMsg::DRAM_READ_REQ,
718 MemComponent::TAG_DIR, MemComponent::DRAM,
719 receiver /* requester */,
720 dram_node /* receiver */,
721 address,
722 NULL, 0,
723 HitWhere::UNKNOWN,
724 orig_shmem_msg->getPerf(),
725 ShmemPerfModel::_SIM_THREAD,orig_shmem_msg->getBlockType());
726 }
727 MYLOG("End @ %lx", address);

731DramDirectoryCntlr::processDRAMReply(core_id_t sender, ShmemMsg* shmem_msg)
732{
733 IntPtr address = shmem_msg->getAddress();
734
735 MYLOG("Start @ %lx", address);
736 // Data received from DRAM
737
738 // Which node to reply to?
739
740 assert(m_dram_directory_req_queue_list->size(address) >= 1);
741 ShmemReq* shmem_req = m_dram_directory_req_queue_list->front(address);
742 updateShmemPerf(shmem_req);
743
744 // Which reply type to use?
745
746 ShmemMsg::msg_t reply_msg_type;
747 DirectoryEntry* directory_entry = m_dram_directory_cache->getDirectoryEntry(address);
748 assert(directory_entry);
749 DirectoryBlockInfo* directory_block_info = directory_entry->getDirectoryBlockInfo();
750 DirectoryState::dstate_t curr_dstate = directory_block_info->getDState();
751
752 updateShmemPerf(shmem_req, ShmemPerf::TD_ACCESS);
753
754 switch(shmem_req->getShmemMsg()->getMsgType())
755 {
756 case ShmemMsg::SH_REQ:
757 assert(curr_dstate == DirectoryState::SHARED || curr_dstate == DirectoryState::EXCLUSIVE);
758 if (curr_dstate == DirectoryState::EXCLUSIVE)
759 {
760 reply_msg_type = ShmemMsg::EX_REP;
761 }
762 else
763 {
764 reply_msg_type = ShmemMsg::SH_REP;
765 }
766 break;
767 case ShmemMsg::EX_REQ:
768 reply_msg_type = ShmemMsg::EX_REP;
769 assert(curr_dstate == DirectoryState::MODIFIED);
770 break;
771 case ShmemMsg::UPGRADE_REQ:
772 {
773 // if we had to get the data from DRAM, nobody has it anymore: send EX_REP
774 reply_msg_type = ShmemMsg::EX_REP;
775 break;
776 }
777 default:
778 LOG_PRINT_ERROR("Unsupported request type: %u", shmem_req->getShmemMsg()->getMsgType());
779 }
780
781 // Which HitWhere to report?
782
783 HitWhere::where_t hit_where = shmem_msg->getWhere();
784 if (hit_where == HitWhere::DRAM)
785 hit_where = (sender == shmem_msg->getRequester()) ? HitWhere::DRAM_LOCAL : HitWhere::DRAM_REMOTE;
786
787 // Send reply
788 MYLOG("MSG DRAM>%d for %lx", shmem_req->getShmemMsg()->getRequester(), address )
789 getMemoryManager()->sendMsg(reply_msg_type,
790 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
791 shmem_req->getShmemMsg()->getRequester() /* requester */,
792 shmem_req->getShmemMsg()->getRequester() /* receiver */,
793 address,
794 shmem_msg->getDataBuf(), getCacheBlockSize(),
795 hit_where,
796 shmem_req->getShmemMsg()->getPerf(),
797 ShmemPerfModel::_SIM_THREAD,shmem_req->getBlockType());
798
799 // Keep a copy in NUCA
800 sendDataToNUCA(address, shmem_req->getShmemMsg()->getRequester(), shmem_msg->getDataBuf(), getShmemPerfModel()->getElapsedTime(ShmemPerfModel::_SIM_THREAD), false,shmem_msg->getBlockType());
801
802 // Process Next Request
803 processNextReqFromL2Cache(address);
804 MYLOG("End @ %lx", address);

808void
809DramDirectoryCntlr::processInvRepFromL2Cache(core_id_t sender, ShmemMsg* shmem_msg)
810{
811 IntPtr address = shmem_msg->getAddress();
812
813 MYLOG("Start @ %lx", address);
814
815 DirectoryEntry* directory_entry = m_dram_directory_cache->getDirectoryEntry(address);
816 assert(directory_entry);
817
818 DirectoryBlockInfo* directory_block_info = directory_entry->getDirectoryBlockInfo();
819 LOG_ASSERT_ERROR(directory_block_info->getDState() == DirectoryState::SHARED || directory_block_info->getDState() == DirectoryState::EXCLUSIVE, "Ooops (%lx)", address);
820 assert(directory_block_info->getDState() == DirectoryState::SHARED || directory_block_info->getDState() == DirectoryState::EXCLUSIVE);
821
822 directory_entry->removeSharer(sender);
823 if (directory_entry->getForwarder() == sender)
824 {
825 directory_entry->setForwarder(INVALID_CORE_ID);
826 }
827 if (directory_entry->getNumSharers() == 0)
828 {
829 directory_block_info->setDState(DirectoryState::UNCACHED);
830 }
831
832 if (m_dram_directory_req_queue_list->size(address) > 0)
833 {
834 MYLOG("More requests outstanding for address %lx", address);
835 ShmemReq* shmem_req = m_dram_directory_req_queue_list->front(address);
836
837 // Update Times in the Shmem Perf Model and the Shmem Req
838 shmem_req->updateTime(getShmemPerfModel()->getElapsedTime(ShmemPerfModel::_SIM_THREAD));
839 getShmemPerfModel()->updateElapsedTime(shmem_req->getTime(), ShmemPerfModel::_SIM_THREAD);
840
841 if (shmem_req->getShmemMsg()->getMsgType() == ShmemMsg::EX_REQ)
842 {
843 // An ShmemMsg::EX_REQ caused the invalidation
844 if (directory_block_info->getDState() == DirectoryState::UNCACHED)
845 {
846 updateShmemPerf(shmem_req, ShmemPerf::INV_IMBALANCE);
847 //std::cout<<"ExReq : "<<__LINE__<<"\n";
848 processExReqFromL2Cache(shmem_req);
849 }
850 }
851 else if (shmem_req->getShmemMsg()->getMsgType() == ShmemMsg::UPGRADE_REQ)
852 {
853 MYLOG("as part of UPGR: got INV_REP from %d, %d sharers left", sender,directory_entry->getNumSharers() )
854
855 // An ShmemMsg::UPGRADE_REQ caused the invalidation: additional action is only required
856 // when all SHARED copies have been invalided
857 if (directory_entry->getNumSharers() == 1)
858 {
859 core_id_t requester = shmem_req->getShmemMsg()->getRequester();
860 if (directory_entry->hasSharer(requester))
861 {
862 // all Shared copies have been invalidated, UPGRADE_REQ can be completed now.
863 MYLOG("UPGRADE_REQ: all OTHER shared copies have been invalided: complete upgrade req.\n");
864 updateShmemPerf(shmem_req, ShmemPerf::INV_IMBALANCE);
865 processUpgradeReqFromL2Cache(shmem_req);
866 }
867 }
868 else if (directory_entry->getNumSharers() == 0)
869 {
870 MYLOG("UPGRADE_REQ: all shared copies have been invalided, requester has no copy anymore.\n");
871
872 directory_block_info->setDState(DirectoryState::UNCACHED);
873
874 updateShmemPerf(shmem_req, ShmemPerf::INV_IMBALANCE);
875 processUpgradeReqFromL2Cache(shmem_req);
876 }
877 else
878 {
879 MYLOG("could not do complete UPGRADE_REQ because there are still multiple sharers");
880 }
881 }
882 else if (shmem_req->getShmemMsg()->getMsgType() == ShmemMsg::SH_REQ)
883 {
884 if (shmem_req->getWaitForData())
885 {
886 // This is a voluntary invalidate (probably part of an upgrade or eviction),
887 // the next request should only be woken up once its data arrives from DRAM.
888 }
889 else if (shmem_req->isForwarding() && sender != shmem_req->getForwardingFrom())
890 {
891 // This is a voluntary invalidate (probably part of an upgrade or eviction),
892 // the next request should only be woken up once its data arrives from the forwarder.
893 }
894 else
895 {
896 // A ShmemMsg::SH_REQ caused the invalidation
897 updateShmemPerf(shmem_req, ShmemPerf::INV_IMBALANCE);
898 //std::cout<<"ShReq : "<<__LINE__<<"\n";
899 processShReqFromL2Cache(shmem_req);
900 }
901 }
902 else // shmem_req->getShmemMsg()->getMsgType() == ShmemMsg::NULLIFY_REQ
903 {
904 if (directory_block_info->getDState() == DirectoryState::UNCACHED)
905 {
906 updateShmemPerf(shmem_req, ShmemPerf::INV_IMBALANCE);
907 //std::cout<<"Nullify Request: "<<__LINE__<<std::endl;
908 processNullifyReq(shmem_req);
909 }
910 }
911 }
912 MYLOG("End @ %lx", address);
913}

915void
916DramDirectoryCntlr::processUpgradeReqFromL2Cache(ShmemReq* shmem_req, Byte* cached_data_buf)
917{
918 ShmemMsg* shmem_msg = shmem_req->getShmemMsg();
919
920 IntPtr address = shmem_msg->getAddress();
921 core_id_t requester = shmem_msg->getRequester();
922 updateShmemPerf(shmem_req);
923
924 MYLOG("processUpgradeReqFromL2Cache for address: %lx, requester= %d", address, requester);
925 DirectoryEntry* directory_entry = m_dram_directory_cache->getDirectoryEntry(address);
926 if (directory_entry == NULL)
927 {
928 //std::cout<<"Process Directory Entry Allocation Req : "<<__LINE__<<std::endl;
929 directory_entry = processDirectoryEntryAllocationReq(shmem_req);
930 }
931 assert(directory_entry);
932
933 DirectoryBlockInfo* directory_block_info = directory_entry->getDirectoryBlockInfo();
934
935 std::pair<bool, std::vector<SInt32> > sharers_list_pair = directory_entry->getSharersList();
936
937 DirectoryState::dstate_t curr_dstate = directory_block_info->getDState();
938
939 MYLOG("state=%d :: ", curr_dstate);
940 MYLOG("owner=%d" , directory_entry->getOwner());
941 if (!sharers_list_pair.first)
942 {
943 for (UInt32 i = 0; i < sharers_list_pair.second.size(); i++)
944 {
945 MYLOG("sharer: %d",sharers_list_pair.second[i]);
946 }
947 }
948
949 updateShmemPerf(shmem_msg, ShmemPerf::TD_ACCESS);
950
951 switch (curr_dstate)
952 {
953 case DirectoryState::EXCLUSIVE:
954 case DirectoryState::MODIFIED:
955 {
956 if (sharers_list_pair.second[0] == requester)
957 {
958 MYLOG("upgrade request immediately finished, sending UPGRADE_REP to %d", requester);
959 assert (directory_entry->getOwner() == requester);
960
961 getMemoryManager()->sendMsg(ShmemMsg::UPGRADE_REP,
962 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
963 requester /* requester */,
964 requester /* receiver */,
965 address,
966 NULL, 0,
967 HitWhere::UNKNOWN, shmem_msg->getPerf(), ShmemPerfModel::_SIM_THREAD,shmem_msg->getBlockType());
968
969 directory_block_info->setDState(DirectoryState::MODIFIED);
970
971 processNextReqFromL2Cache(address);
972 }
973 else
974 {
975 // Send FLUSH_REQ to the current owner
976 //std::cout << "Dram request: " << __LINE__ <<std::endl;
977 MYLOG("FLUSH REQ (UPGR)>%u @ %lx",sharers_list_pair.second[0] , address);
978 getMemoryManager()->sendMsg(ShmemMsg::FLUSH_REQ,
979 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
980 requester /* requester */,
981 directory_entry->getOwner() /* receiver */,
982 address,
983 NULL, 0,
984 HitWhere::UNKNOWN, shmem_msg->getPerf(), ShmemPerfModel::_SIM_THREAD,shmem_msg->getBlockType());
985 }
986
987 break;
988 }
989 case DirectoryState::SHARED:
990 {
991 if ((sharers_list_pair.second.size() == 1) && (sharers_list_pair.second[0] == requester))
992 {
993 // Let the requester know it can take ownership
994 MYLOG("sending UPGRADE_REP>%d", requester )
995 directory_entry->setOwner(requester);
996 directory_block_info->setDState(DirectoryState::MODIFIED);
997
998 getMemoryManager()->sendMsg(ShmemMsg::UPGRADE_REP,
999 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
1000 requester /* requester */,
1001 requester /* receiver */,
1002 address,
1003 NULL, 0,
1004 HitWhere::UNKNOWN, shmem_msg->getPerf(), ShmemPerfModel::_SIM_THREAD,shmem_msg->getBlockType());
1005
1006 processNextReqFromL2Cache(address);
1007 }
1008 else
1009 {
1010 bool requesterHasCopy = directory_entry->hasSharer(requester);
1011 if (!requesterHasCopy)
1012 {
1013 MYLOG("UPGRADE_REQ: %lu sharer(s), but the requester isn't holding a copy!?", sharers_list_pair.second.size());
1014 }
1015
1016 // send inv_req to all sharers
1017 if (sharers_list_pair.first == true)
1018 {
1019 // Broadcast Invalidation Request to all cores
1020 // (irrespective of whether they are sharers or not)
1021 getMemoryManager()->broadcastMsg(ShmemMsg::INV_REQ,
1022 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
1023 requester /* requester */,
1024 address,
1025 NULL, 0,
1026 NULL, // No ShmemPerf on broadcast
1027 ShmemPerfModel::_SIM_THREAD);
1028 }
1029 else
1030 {
1031 bool shmem_perf_sent = false;
1032 // Send Invalidation Request to only a specific set of sharers
1033 for (UInt32 i = 0; i < sharers_list_pair.second.size(); i++)
1034 {
1035 if (sharers_list_pair.second[i] != requester)
1036 {
1037 MYLOG("INV REQ (UPGR)>%u @ %lx",sharers_list_pair.second[i] , shmem_msg->getAddress());
1038 // avoid having to fetch the data from DRAM, so ask at least one core to FLUSH instead of INV
1039 //std::cout << "Dram request: " << __LINE__ <<std::endl;
1040 ShmemMsg::msg_t msg_type = (!requesterHasCopy && i==0) ? ShmemMsg::FLUSH_REQ : ShmemMsg::INV_REQ;
1041 if(msg_type==ShmemMsg::INV_REQ){
1042 //std::cout << "Dram request: " << __LINE__ <<std::endl;
1043 }
1044
1045 //ShmemMsg::msg_t msg_type = ShmemMsg::INV_REQ;
1046 getMemoryManager()->sendMsg( msg_type, //ShmemMsg::INV_REQ,
1047 MemComponent::TAG_DIR, MemComponent::L2_CACHE,
1048 requester /* requester */,
1049 sharers_list_pair.second[i] /* receiver */,
1050 address,
1051 NULL, 0,
1052 HitWhere::UNKNOWN,
1053 shmem_perf_sent == false ? shmem_msg->getPerf() : &m_dummy_shmem_perf,
1054 ShmemPerfModel::_SIM_THREAD,shmem_msg->getBlockType());
1055 shmem_perf_sent = true;
1056 }
1057 }
1058 }
1059 }
1060 break;
1061 }
1062 case DirectoryState::UNCACHED:
1063 {
1064 MYLOG("%lx is UNCACHED", address);
1065 assert (sharers_list_pair.second.size() == 0);
1066
1067 // Modifiy the directory entry contents
1068 bool add_result = directory_entry->addSharer(requester, m_dram_directory_cache->getMaxHwSharers());
1069 assert(add_result == true);
1070 directory_entry->setOwner(requester);
1071 directory_block_info->setDState(DirectoryState::MODIFIED);
1072
1073 if (cached_data_buf == NULL)
1074 {
1075 // maybe the data is stored in the msg already?
1076 cached_data_buf = shmem_msg->getDataBuf();
1077 }
1078 retrieveDataAndSendToL2Cache(ShmemMsg::EX_REP, requester, address, cached_data_buf, shmem_msg);
1079
1080 break;
1081 }
1082 default:
1083 {
1084 LOG_PRINT_ERROR("Unsupported Directory State: %u", curr_dstate);
1085 break;
1086 }
1087 }
1088 MYLOG("End @ %lx", address);
1089}

1106 directory_entry->setForwarder(INVALID_CORE_ID);
1107 directory_entry->setOwner(INVALID_CORE_ID);
1108
1109 // could be that this is a FLUSH to force a core with S-state to to write back clean data

1113 directory_block_info->setDState(DirectoryState::UNCACHED);

1116 {

1152 processUpgradeReqFromL2Cache(shmem_req, shmem_msg->getDataBuf());

1174 // This was just an eviction
1175 // Write Data to Dram
1176 sendDataToDram(address, shmem_msg->getRequester(), shmem_msg->getDataBuf(), now,shmem_msg->getBlockType());
1177 }

1179 MYLOG("End @ %lx", address);
