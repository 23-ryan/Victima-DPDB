174UInt64 TraceThread::remapAddress(UInt64 va_page)
175{
176 // va is the virtual address shifted right by the page size
177 // By randomly remapping the lower 24 bits of va_page, addresses will be distributed
178 // over a 1<<(16+3*8) = 64 GB range which should avoid artificial set contention in all cache levels.
179 // Of course we want the remapping to be invertible so we never map different incoming addresses
180 // onto the same outgoing address. This is guaranteed since m_address_randomization_table
181 // contains each 0..255 number only once.
182 UInt64 result = va_page;
183 uint8_t *array = (uint8_t *)&result;
184 array[0] = m_address_randomization_table[array[0]];
185 array[1] = m_address_randomization_table[array[1]];
186 array[2] = m_address_randomization_table[array[2]];
187 return result;
188}

496 // Received first instruction, let TraceManager know our SIFT connection is up and running
497 Sim()->getTraceManager()->signalStarted();

605void TraceThread::handleInstructionWarmup(Sift::Instruction &inst, Sift::Instruction &next_inst, Core *core, bool do_icache_warmup, UInt64 icache_warmup_addr, UInt64 icache_warmup_size)
606{
607 if (m_decoder_cache.count(inst.sinst->addr) == 0)
608 m_decoder_cache[inst.sinst->addr] = staticDecode(inst);
609
610 const dl::DecodedInst &dec_inst = *(m_decoder_cache[inst.sinst->addr]);
611
612 // Warmup instruction caches
613
614 if (do_icache_warmup && Sim()->getConfig()->getEnableICacheModeling())
615 {
616 core->readInstructionMemory(va2pa(icache_warmup_addr), icache_warmup_size);
617 }
618
619 // Warmup branch predictor
620
621 if (inst.is_branch)
622 {
623 bool mispredict = core->accessBranchPredictor(va2pa(inst.sinst->addr), inst.taken, va2pa(next_inst.sinst->addr));
624 if (mispredict)
625 core->getPerformanceModel()->handleBranchMispredict();
626 }
627
628 // Warmup data caches
629
630 if (inst.executed)
631 {
632 const bool is_atomic_update = dec_inst.is_atomic();
633 const bool is_prefetch = dec_inst.is_prefetch();
634
635 // Ignore memory-referencing operands in NOP instructions
636 if (!dec_inst.is_nop())
637 {
638 for(uint32_t mem_idx = 0; mem_idx < Sim()->getDecoder()->num_memory_operands(&dec_inst); ++mem_idx)
639 {
640 if (Sim()->getDecoder()->op_read_mem(&dec_inst, mem_idx))
641 {
642 UInt64 mem_address;
643 // LDP ARM instructions, second element to be loaded, using the address of the first element
644 if (dec_inst.is_mem_pair() && ((int)mem_idx == (inst.num_addresses + 1)))
645 {
646 LOG_ASSERT_ERROR((int)mem_idx < (inst.num_addresses + 1), "Did not receive enough data addresses");
647
648 mem_address = inst.addresses[mem_idx - 1] + Sim()->getDecoder()->size_mem_op(&dec_inst, mem_idx);
649 }
650 else
651 {
652 LOG_ASSERT_ERROR(mem_idx < inst.num_addresses, "Did not receive enough data addresses");
653
654 mem_address = inst.addresses[mem_idx];
655 }
656
657 /*
658 bool no_mapping = false;
659 UInt64 pa = va2pa(mem_address, is_prefetch ? &no_mapping : NULL);
660 if (no_mapping)
661 continue;
662 */
663 UInt64 va = mem_address;
664
665 core->accessMemory(
666 /*(is_atomic_update) ? Core::LOCK :*/ Core::NONE,
667 (is_atomic_update) ? Core::READ_EX : Core::READ,
668 va,
669 NULL,
670 Sim()->getDecoder()->size_mem_op(&dec_inst, mem_idx),
671 Core::MEM_MODELED_COUNT,
672 va2pa(inst.sinst->addr));
673 }
674 }
675
676 for(uint32_t mem_idx = 0; mem_idx < Sim()->getDecoder()->num_memory_operands(&dec_inst); ++mem_idx)
677 {
678 if (Sim()->getDecoder()->op_write_mem(&dec_inst, mem_idx))
679 {
680 UInt64 mem_address;
681 // STP ARM instructions, second element to be stored, using the address of the first element
682 if (dec_inst.is_mem_pair() && ((int)mem_idx == (inst.num_addresses + 1)))
683 {
684 LOG_ASSERT_ERROR((int)mem_idx < (inst.num_addresses + 1), "Did not receive enough data addresses");
685 mem_address = inst.addresses[mem_idx - 1] + Sim()->getDecoder()->size_mem_op(&dec_inst, mem_idx);
686 }
687 else
688 {
689 LOG_ASSERT_ERROR(mem_idx < inst.num_addresses, "Did not receive enough data addresses");
690
691 mem_address = inst.addresses[mem_idx];
692 }
693 /*
694 bool no_mapping = false;
695 UInt64 pa = va2pa(mem_address, is_prefetch ? &no_mapping : NULL);
696 if (no_mapping)
697 continue;
698 */
699
700 UInt64 va = mem_address; /* pa == va @konkanello */
701
702 if (is_atomic_update)
703 core->logMemoryHit(false, Core::WRITE, va, Core::MEM_MODELED_COUNT, va2pa(inst.sinst->addr));
704 else
705 core->accessMemory(
706 /*(is_atomic_update) ? Core::UNLOCK :*/ Core::NONE,
707 Core::WRITE,
708 va,
709 NULL,
710 Sim()->getDecoder()->size_mem_op(&dec_inst, mem_idx),
711 Core::MEM_MODELED_COUNT,
712 va2pa(inst.sinst->addr));
713 }
714 }
715 }

719void TraceThread::handleInstructionDetailed(Sift::Instruction &inst, Sift::Instruction &next_inst, PerformanceModel *prfmdl)
720{
721
722 // Set up instruction
723
724 if (m_icache.count(inst.sinst->addr) == 0)
725 m_icache[inst.sinst->addr] = decode(inst);
726 // Here get the decoder instruction without checking, because we must have it for sure
727 const dl::DecodedInst &dec_inst = *(m_decoder_cache[inst.sinst->addr]);
728
729 Instruction *ins = m_icache[inst.sinst->addr];
730 DynamicInstruction *dynins = prfmdl->createDynamicInstruction(ins, va2pa(inst.sinst->addr));
731
732 // Add dynamic instruction info
733
734 if (inst.is_branch)
735 {
736 dynins->addBranch(inst.taken, va2pa(next_inst.sinst->addr));
737 }
738
739 // Ignore memory-referencing operands in NOP instructions
740 if (!dec_inst.is_nop())
741 {
742 const bool is_prefetch = dec_inst.is_prefetch();
743
744 for(uint32_t mem_idx = 0; mem_idx < Sim()->getDecoder()->num_memory_operands(&dec_inst); ++mem_idx)
745 {
746 if (Sim()->getDecoder()->op_read_mem(&dec_inst, mem_idx))
747 {
748 addDetailedMemoryInfo(dynins, inst, dec_inst, mem_idx, Operand::READ, is_prefetch, prfmdl);
749 }
750 }
751
752 for(uint32_t mem_idx = 0; mem_idx < Sim()->getDecoder()->num_memory_operands(&dec_inst); ++mem_idx)
753 {
754 if (Sim()->getDecoder()->op_write_mem(&dec_inst, mem_idx))
755 {
756 addDetailedMemoryInfo(dynins, inst, dec_inst, mem_idx, Operand::WRITE, is_prefetch, prfmdl);
757 }
758 }
759 }
760
761 // Push instruction
762
763 prfmdl->queueInstruction(dynins);
764
765 // simulate
766
767 prfmdl->iterate();
768}

770void TraceThread::addDetailedMemoryInfo(DynamicInstruction *dynins, Sift::Instruction &inst, const dl::DecodedInst &decoded_inst, uint32_t mem_idx, Operand::Direction op_type, bool is_prefetch, PerformanceModel *prfmdl)
771{
772 UInt64 mem_address;
773 // LDP/STP ARM instructions, second element to be ld/st, using the address of the first element
774 if (decoded_inst.is_mem_pair() && ((int)mem_idx == inst.num_addresses))
775 {
776 assert((int)mem_idx < (inst.num_addresses + 1));
777 mem_address = inst.addresses[mem_idx - 1] + Sim()->getDecoder()->size_mem_op(&decoded_inst, mem_idx);
778 }
779 else
780 {
781 assert(mem_idx < inst.num_addresses);
782 mem_address = inst.addresses[mem_idx];
783 }
784
785
786 bool no_mapping = false;
787 UInt64 va = mem_address;
788 UInt64 pa = va2pa(mem_address, is_prefetch ? &no_mapping : NULL);
789
790 if (no_mapping)
791 {
792 dynins->addMemory(
793 inst.executed,
794 SubsecondTime::Zero(),
795 0,
796 Sim()->getDecoder()->size_mem_op(&decoded_inst, mem_idx),
797 op_type,
798 0,
799 HitWhere::PREFETCH_NO_MAPPING);
800 }
801 else
802 {
803 dynins->addMemory(
804 inst.executed,
805 SubsecondTime::Zero(),
806 va,
807 Sim()->getDecoder()->size_mem_op(&decoded_inst, mem_idx),
808 op_type,
809 0,
810 HitWhere::UNKNOWN);
811 }

816void TraceThread::unblock()
817{
818 LOG_ASSERT_ERROR(m_blocked == true, "Must call only when m_blocked == true");
819
820 SubsecondTime end_time = Sim()->getClockSkewMinimizationServer()->getGlobalTime(true /*upper_bound*/);
821 m_thread->getSyscallMdl()->runExit(0);
822 {
823 ScopedLock sl(Sim()->getThreadManager()->getLock());
824 // We were blocked on a system call, but started executing instructions again.
825 // This means we woke up. Since there is no explicit wakeup nor an associated time,
826 // use global time.
827 Sim()->getThreadManager()->resumeThread_async(m_thread->getId(), INVALID_THREAD_ID, end_time, NULL);
828 }
829
830 if (m_thread->getCore())
831 {
832 m_thread->getCore()->getPerformanceModel()->queuePseudoInstruction(new SyncInstruction(end_time, SyncInstruction::SYSCALL));
833 }
834 else
835 {
836 // We were scheduled out during the system call
837 m_thread->reschedule(end_time, NULL);
838 m_thread->getCore()->getPerformanceModel()->queuePseudoInstruction(new SyncInstruction(end_time, SyncInstruction::UNSCHEDULED));
839 }
840
841 m_blocked = false;
842}

844void TraceThread::run()
845{
846 // Set thread name for Sniper-in-Sniper simulations
847 String threadName = String("trace-") + itostr(m_thread->getId());
848 SimSetThreadName(threadName.c_str());
849
850 Sim()->getThreadManager()->onThreadStart(m_thread->getId(), m_time_start);
851
852 // Open the trace (be sure to do this before potentially blocking on reschedule() as this causes deadlock)
853 m_trace.initStream();
854 m_trace_has_pa = m_trace.getTraceHasPhysicalAddresses();
855
856 if (m_thread->getCore() == NULL)
857 {
858 // We didn't get scheduled on startup, wait here
859 SubsecondTime time = SubsecondTime::Zero();
860 m_thread->reschedule(time, NULL);
861 }
862
863 Core *core = m_thread->getCore();
864 PerformanceModel *prfmdl = core->getPerformanceModel();
865
866 Sift::Instruction inst, next_inst;
867
868 bool have_first = m_trace.Read(inst);
869 // Received first instruction, let TraceManager know our SIFT connection is up and running
870 Sim()->getTraceManager()->signalStarted();
871 m_started = true;
872
873 while(have_first && m_trace.Read(next_inst))
874 {
875 if (m_blocked)
876 {
877 unblock();
878 }
879
880 core = m_thread->getCore();
881 prfmdl = core->getPerformanceModel();
882
883 bool do_icache_warmup = false;
884 UInt64 icache_warmup_addr = 0, icache_warmup_size = 0;
885
886 // Reconstruct and count basic blocks
887
888 if (m_bbv_end || m_bbv_last != inst.sinst->addr)
889 {
890 // We're the start of a new basic block
891 core->countInstructions(m_bbv_base, m_bbv_count);
892 // In cache-only mode, we'll want to do I-cache warmup
893 if (m_bbv_base)
894 {
895 do_icache_warmup = true;
896 icache_warmup_addr = m_bbv_base;
897 icache_warmup_size = m_bbv_last - m_bbv_base;
898 }
899 // Set up new basic block info
900 m_bbv_base = inst.sinst->addr;
901 m_bbv_count = 0;
902 }
903 m_bbv_count++;
904 m_bbv_last = inst.sinst->addr + inst.sinst->size;
905 // Force BBV end on non-taken branches
906 m_bbv_end = inst.is_branch;
907
908
909 switch(Sim()->getInstrumentationMode())
910 {
911 case InstMode::FAST_FORWARD:
912 break;
913
914 case InstMode::CACHE_ONLY:
915 handleInstructionWarmup(inst, next_inst, core, do_icache_warmup, icache_warmup_addr, icache_warmup_size);
916 break;
917
918 case InstMode::DETAILED:
919 handleInstructionDetailed(inst, next_inst, prfmdl);
920 break;
921
922 default:
923 LOG_PRINT_ERROR("Unknown instrumentation mode");
924 }
925
926
927 // We may have been rescheduled to a different core
928 // by prfmdl->iterate (in handleInstructionDetailed),
929 // or core->countInstructions (when using a fast-forward performance model)
930 SubsecondTime time = prfmdl->getElapsedTime();
931 if (m_thread->reschedule(time, core))
932 {
933 core = m_thread->getCore();
934 prfmdl = core->getPerformanceModel();
935 }
936
937
938 if (m_stop)
939 break;
940
941 inst = next_inst;
942 }
943
944 printf("[TRACE:%u] -- %s --\n", m_thread->getId(), m_stop ? "STOP" : "DONE");
945
946 SubsecondTime time_end = prfmdl->getElapsedTime();
947
948 Sim()->getThreadManager()->onThreadExit(m_thread->getId());
949 Sim()->getTraceManager()->signalDone(this, time_end, m_stop /*aborted*/);
950}
