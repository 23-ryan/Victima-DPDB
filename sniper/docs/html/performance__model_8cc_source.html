17PerformanceModel* PerformanceModel::create(Core* core)
18{
19 String type;
20
21 try {
22 type = Sim()->getCfg()->getStringArray("perf_model/core/type", core->getId());
23 } catch (...) {
24 LOG_PRINT_ERROR("No perf model type provided.");
25 }
26
27 if (type == "oneipc")
28 return new OneIPCPerformanceModel(core);
29 else if (type == "interval")
30 {
31 // The interval model needs the branch misprediction penalty
32 int mispredict_penalty = Sim()->getCfg()->getIntArray("perf_model/branch_predictor/mispredict_penalty", core->getId());
33 return new IntervalPerformanceModel(core, mispredict_penalty);
34 }
35 else if (type == "rob")
36 {
37 uint32_t smt_threads = Sim()->getCfg()->getIntArray("perf_model/core/logical_cpus", core->getId());
38 if (smt_threads == 1)
39 return new RobPerformanceModel(core);
40 else
41 return new RobSmtPerformanceModel(core);
42 }
43 else
44 {
45 LOG_PRINT_ERROR("Invalid perf model type: %s", type.c_str());
46 return NULL;
47 }
48}

145void PerformanceModel::queuePseudoInstruction(PseudoInstruction *i)
146{
147 if (i->getType() == INST_SPAWN)
148 {
149 SpawnInstruction const* spawn_insn = dynamic_cast<SpawnInstruction const*>(i);
150 LOG_ASSERT_ERROR(spawn_insn != NULL, "Expected a SpawnInstruction, but did not get one.");
151 setElapsedTime(spawn_insn->getTime());
152 delete i;
153 return;
154 }
155
156 if (!m_enabled)
157 {
158 // queueInstruction and pushDynamicInstructionInfo are not being called in fast-forward by using instrumentation modes
159 // For queueDynamicInstruction, which are used all over the place, ignore them manually
160 delete i;
161 return;
162 }
163
164 if (i->isIdle())

166 handleIdleInstruction(i);

168 }
169 else

171 if (m_fastforward)

173 m_fastforward_model->queuePseudoInstruction(i);
174 }
175 else
176 {
177 #ifdef ENABLE_PERF_MODEL_OWN_THREAD
178 m_instruction_queue.push_wait(createDynamicInstruction(i, 0));
179 #else
180 m_instruction_queue.push(createDynamicInstruction(i, 0));
181 #endif
182 }
183 }

203void PerformanceModel::handleIdleInstruction(PseudoInstruction *instruction)
204{
205 // If fast-forwarding without detailed synchronization, our fast-forwarding IPC
206 // already contains idle periods so we can ignore these now
207 if (m_fastforward && !m_detailed_sync)
208 return;
209
210 if (instruction->getType() == INST_SYNC)
211 {
212 // Keep track of the type of Sync instruction and it's latency to calculate CPI numbers
213 SyncInstruction const* sync_insn = dynamic_cast<SyncInstruction const*>(instruction);
214 LOG_ASSERT_ERROR(sync_insn != NULL, "Expected a SyncInstruction, but did not get one.");
215
216 // Thread may wake up on a different core than where it went to sleep, and/or a different thread
217 // may have run on this core while the thread was asleep
218 // So, compute actual delay between our last local time and the time we're supposed to wake up
219 SubsecondTime time_begin = getElapsedTime();
220 SubsecondTime time_end = sync_insn->getTime();
221 SubsecondTime insn_cost;
222
223 if (time_end > time_begin)
224 insn_cost = time_end - time_begin;
225 else
226 // Core may have executed other instructions already
227 insn_cost = SubsecondTime::Zero();
228
229 switch(sync_insn->getSyncType()) {
230 case(SyncInstruction::FUTEX):
231 m_cpiSyncFutex += insn_cost;
232 break;
233 case(SyncInstruction::PTHREAD_MUTEX):
234 m_cpiSyncPthreadMutex += insn_cost;
235 break;
236 case(SyncInstruction::PTHREAD_COND):
237 m_cpiSyncPthreadCond += insn_cost;
238 break;
239 case(SyncInstruction::PTHREAD_BARRIER):
240 m_cpiSyncPthreadBarrier += insn_cost;
241 break;
242 case(SyncInstruction::JOIN):
243 m_cpiSyncJoin += insn_cost;
244 break;
245 case(SyncInstruction::PAUSE):
246 m_cpiSyncPause += insn_cost;
247 break;
248 case(SyncInstruction::SLEEP):
249 m_cpiSyncSleep += insn_cost;
250 break;
251 case(SyncInstruction::SYSCALL):
252 m_cpiSyncSyscall += insn_cost;
253 break;
254 case(SyncInstruction::UNSCHEDULED):
255 m_cpiSyncUnscheduled += insn_cost;
256 break;
257 default:
258 LOG_ASSERT_ERROR(false, "Unexpected SyncInstruction::type_t enum type. (%d)", sync_insn->getSyncType());
259 }
260 incrementIdleElapsedTime(insn_cost);
261 }
262 else if (instruction->getType() == INST_DELAY)
263 {
264 SubsecondTime insn_cost = instruction->getCost(getCore());
265 DelayInstruction const* delay_insn = dynamic_cast<DelayInstruction const*>(instruction);
266 LOG_ASSERT_ERROR(delay_insn != NULL, "Expected a DelayInstruction, but did not get one.");
267 switch(delay_insn->getDelayType()) {
268 case(DelayInstruction::DVFS_TRANSITION):
269 m_cpiSyncDvfsTransition += insn_cost;
270 break;
271 default:
272 LOG_ASSERT_ERROR(false, "Unexpected DelayInstruction::type_t enum type. (%d)", delay_insn->getDelayType());
273 }
274 incrementIdleElapsedTime(insn_cost);
275 }
276 else if (instruction->getType() == INST_RECV)
277 {
278 SubsecondTime insn_cost = instruction->getCost(getCore());
279 __attribute__((unused)) RecvInstruction const* recv_insn = dynamic_cast<RecvInstruction const*>(instruction);
280 LOG_ASSERT_ERROR(recv_insn != NULL, "Expected a RecvInstruction, but did not get one.");
281 m_cpiRecv += insn_cost;
282 incrementIdleElapsedTime(insn_cost);
283 }
284 else
285 {
286 LOG_PRINT_ERROR("Unexpectedly received something other than a Sync or Recv Instruction");
287 }
288
289 // Notify the fast-forward performance model
290 if (m_fastforward)
291 m_fastforward_model->notifyElapsedTimeUpdate();
292}

294void PerformanceModel::iterate()
295{
296 while (m_instruction_queue.size() > 0)
297 {
298 // While the functional thread is waiting because of clock skew minimization, wait here as well
299 #ifdef ENABLE_PERF_MODEL_OWN_THREAD
300 while(m_hold)
301 sched_yield();
302 #endif
303
304 DynamicInstruction *ins = m_instruction_queue.front();
305
306 LOG_ASSERT_ERROR(!ins->instruction->isIdle(), "Idle instructions should not make it here!");
307
308 if (!m_fastforward && m_enabled)
309 handleInstruction(ins);
310
311 delete ins;
312
313 m_instruction_queue.pop();
314 }
315
316 synchronize();
317}

340{
