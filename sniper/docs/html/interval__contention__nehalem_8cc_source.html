38uint64_t IntervalContentionNehalem::getEffectiveCriticalPathLength(uint64_t critical_path_length, bool update_reason)
39{
40 DynamicMicroOpNehalem::uop_port_t reason = DynamicMicroOpNehalem::UOP_PORT_SIZE;
41 uint64_t effective_cp_length = critical_path_length;
42
43 // For the standard ports, check if we have exceeded our execution limit
44 for (unsigned int i = 0 ; i < DynamicMicroOpNehalem::UOP_PORT_SIZE ; i++)
45 {
46 // Skip shared ports
47 if (i != DynamicMicroOpNehalem::UOP_PORT015
48 && i != DynamicMicroOpNehalem::UOP_PORT05
49 && effective_cp_length < m_count_byport[i]
50 )
51 {
52 effective_cp_length = m_count_byport[i];
53 reason = (DynamicMicroOpNehalem::uop_port_t)i;
54 }
55 }
56 // Check shared port usage
57 uint64_t port05_use = m_count_byport[DynamicMicroOpNehalem::UOP_PORT0] + m_count_byport[DynamicMicroOpNehalem::UOP_PORT5]
58 + m_count_byport[DynamicMicroOpNehalem::UOP_PORT05];
59 if (port05_use > (2*effective_cp_length))
60 {
61 effective_cp_length = (port05_use+1) / 2; // +1 to round up to the next cycle
62 reason = DynamicMicroOpNehalem::UOP_PORT05;
63 }
64 uint64_t port015_use = m_count_byport[DynamicMicroOpNehalem::UOP_PORT0] + m_count_byport[DynamicMicroOpNehalem::UOP_PORT1]
65 + m_count_byport[DynamicMicroOpNehalem::UOP_PORT5] + m_count_byport[DynamicMicroOpNehalem::UOP_PORT015];
66 if (port015_use > (3*effective_cp_length))
67 {
68 effective_cp_length = (port015_use+2) / 3; // +2 to round up to the next cycle
69 reason = DynamicMicroOpNehalem::UOP_PORT015;
70 }
71
72 if (update_reason && effective_cp_length > critical_path_length)
73 {
74 LOG_ASSERT_ERROR(reason != DynamicMicroOpNehalem::UOP_PORT_SIZE, "There should be a reason for the cp extention, but there isn't");
75 m_cpContrByPort[reason] += 1000000 * (effective_cp_length - critical_path_length) / effective_cp_length;
76 }
77
78 return effective_cp_length;
79}
