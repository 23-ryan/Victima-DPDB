63bool SyscallMdl::runEnter(IntPtr syscall_number, syscall_args_t &args)
64{
65 Core *core = m_thread->getCore();
66 LOG_ASSERT_ERROR(core != NULL, "Syscall by thread %d: core should not be null", m_thread->getId());
67
68 LOG_PRINT("Got Syscall: %i", syscall_number);
69 CLOG("syscall", "Enter thread %d core %d syscall %" PRIdPTR, m_thread->getId(), core->getId(), syscall_number);
70
71 m_syscall_number = syscall_number;
72 m_in_syscall = true;
73 m_syscall_args = args;
74
75 HookSyscallEnter hook_args;
76 hook_args.thread_id = m_thread->getId();
77 hook_args.core_id = core->getId();
78 hook_args.time = core->getPerformanceModel()->getElapsedTime();
79 hook_args.syscall_number = syscall_number;
80 hook_args.args = args;
81 {
82 ScopedLock sl(Sim()->getThreadManager()->getLock());
83 Sim()->getHooksManager()->callHooks(HookType::HOOK_SYSCALL_ENTER, (UInt64)&hook_args);
84 }
85
86 switch (syscall_number)
87 {
88 case SYS_futex:
89 m_ret_val = handleFutexCall(args);
90 m_emulated = true;
91 break;
92
93 case SYS_clock_gettime:
94 {
95 if (Sim()->getConfig()->getOSEmuClockReplace())
96 {
97 clockid_t clock = (clock_t)args.arg0;
98 struct timespec *ts = (struct timespec *)args.arg1;
99 UInt64 time_ns = Sim()->getConfig()->getOSEmuTimeStart() * 1000000000
100 + m_thread->getCore()->getPerformanceModel()->getElapsedTime().getNS();
101
102 switch(clock) {
103 case CLOCK_REALTIME:
104 case CLOCK_MONOTONIC:
105 case CLOCK_MONOTONIC_COARSE:
106 case CLOCK_MONOTONIC_RAW:
107 ts->tv_sec = time_ns / 1000000000;
108 ts->tv_nsec = time_ns % 1000000000;
109 m_ret_val = 0;
110 break;
111 default:
112 LOG_ASSERT_ERROR(false, "SYS_clock_gettime does not currently support clock(%u)", clock);
113 }
114 m_emulated = true;
115 }
116 // else: don't set m_emulated and the system call will be executed natively
117 break;
118 }
119
120 case SYS_nanosleep:
121 {
122 const struct timespec *req = (struct timespec *)args.arg0;
123 struct timespec *rem = (struct timespec *)args.arg1;
124 Core *core = m_thread->getCore();
125 SubsecondTime start_time = core->getPerformanceModel()->getElapsedTime();
126
127 struct timespec local_req;
128 core->accessMemory(Core::NONE, Core::READ, (IntPtr) req, (char*) &local_req, sizeof(local_req));
129
130 SubsecondTime time_wake = start_time + SubsecondTime::SEC(local_req.tv_sec) + SubsecondTime::NS(local_req.tv_nsec);
131 SubsecondTime end_time;
132 Sim()->getSyscallServer()->handleSleepCall(m_thread->getId(), time_wake, start_time, end_time);
133
134 if (m_thread->reschedule(end_time, core))
135 core = m_thread->getCore();
136
137 core->getPerformanceModel()->queuePseudoInstruction(new SyncInstruction(end_time, SyncInstruction::SLEEP));
138
139 if (rem)
140 {
141 // Interruption not supported, always return 0 remaining time
142 struct timespec local_rem;
143 local_rem.tv_sec = 0;
144 local_rem.tv_nsec = 0;
145 core->accessMemory(Core::NONE, Core::WRITE, (IntPtr) rem, (char*) &local_rem, sizeof(local_rem));
146 }
147
148 // Always succeeds
149 m_ret_val = 0;
150 m_emulated = true;
151
152 break;
153 }
154
155 case SYS_read:
156 case SYS_pause:
157 case SYS_select:
158 case SYS_poll:
159 case SYS_wait4:
160 {
161 // System call is blocking, mark thread as asleep
162 ScopedLock sl(Sim()->getThreadManager()->getLock());
163 Sim()->getThreadManager()->stallThread_async(m_thread->getId(),
164 syscall_number == SYS_pause ? ThreadManager::STALL_PAUSE : ThreadManager::STALL_SYSCALL,
165 m_thread->getCore()->getPerformanceModel()->getElapsedTime());
166 m_stalled = true;
167 break;
168 }
169
170 case SYS_sched_yield:
171 {
172 {
173 ScopedLock sl(Sim()->getThreadManager()->getLock());
174 Sim()->getThreadManager()->getScheduler()->threadYield(m_thread->getId());
175 }
176
177 // We may have been rescheduled
178 SubsecondTime time = core->getPerformanceModel()->getElapsedTime();
179 if (m_thread->reschedule(time, core))
180 core = m_thread->getCore();
181 core->getPerformanceModel()->queuePseudoInstruction(new SyncInstruction(time, SyncInstruction::UNSCHEDULED));
182
183 // Always succeeds
184 m_ret_val = 0;
185 m_emulated = true;
186
187 break;
188 }
189
190 case SYS_sched_setaffinity:
191 {
192 pid_t pid = (pid_t)args.arg0;
193 size_t cpusetsize = (size_t)args.arg1;
194 const cpu_set_t *cpuset = (const cpu_set_t *)args.arg2;
195 Thread *thread;
196 bool success = false;
197
198 if (cpuset == NULL)
199 {
200 m_ret_val = -EFAULT;
201 m_emulated = true;
202 break;
203 }
204
205 if (pid == 0)
206 thread = m_thread;
207 else
208 thread = Sim()->getThreadManager()->findThreadByTid(pid);
209
210 if (thread)
211 {
212 char *local_cpuset = new char[cpusetsize];
213 core->accessMemory(Core::NONE, Core::READ, (IntPtr) cpuset, local_cpuset, cpusetsize);
214
215 ScopedLock sl(Sim()->getThreadManager()->getLock());
216 success = Sim()->getThreadManager()->getScheduler()->threadSetAffinity(m_thread->getId(), thread->getId(), cpusetsize, (cpu_set_t *)local_cpuset);
217
218 delete [] local_cpuset;
219 }
220 else
221 {
222 m_ret_val = -ESRCH;
223 m_emulated = true;
224 break;
225 }
226
227 // We may have been rescheduled
228 SubsecondTime time = core->getPerformanceModel()->getElapsedTime();
229 if (m_thread->reschedule(time, core))
230 core = m_thread->getCore();
231 core->getPerformanceModel()->queuePseudoInstruction(new SyncInstruction(time, SyncInstruction::UNSCHEDULED));
232
233 m_ret_val = success ? 0 : -EINVAL;
234 m_emulated = true;
235
236 break;
237 }
238
239 case SYS_sched_getaffinity:
240 {
241 pid_t pid = (pid_t)args.arg0;
242 size_t cpusetsize = (size_t)args.arg1;
243 cpu_set_t *cpuset = (cpu_set_t *)args.arg2;
244 Thread *thread;
245 bool success = false;
246
247 if (pid == 0)
248 thread = m_thread;
249 else
250 thread = Sim()->getThreadManager()->findThreadByTid(pid);
251
252 if (thread)
253 {
254 char *local_cpuset = cpuset ? new char[cpusetsize] : 0;
255
256 ScopedLock sl(Sim()->getThreadManager()->getLock());
257 success = Sim()->getThreadManager()->getScheduler()->threadGetAffinity(m_thread->getId(), cpusetsize, (cpu_set_t *)local_cpuset);
258
259 if (success && cpuset)
260 core->accessMemory(Core::NONE, Core::WRITE, (IntPtr) cpuset, local_cpuset, cpusetsize);
261 if (local_cpuset)
262 delete [] local_cpuset;
263 }
264 // else: success is already false, return EINVAL for invalid pid
265
266 // On success: return size of affinity mask (in bytes) needed to represent however many cores we're modeling
267 // (returning a value that is too large can cause a segfault in the application's libc)
268 m_ret_val = success ? (Sim()->getConfig()->getApplicationCores()+7)/8 : -EINVAL;
269 m_emulated = true;
270
271 break;
272 }
273
274 case SYS_write:
275 {
276 int fd = (int)args.arg0;
277 size_t count = (size_t)args.arg2;
278
279 if ((fd == STDOUT_FILENO && Sim()->getConfig()->suppressStdout()) || (fd == STDERR_FILENO && Sim()->getConfig()->suppressStderr()))
280 {
281 m_ret_val = count;
282 m_emulated = true;
283 }
284
285 if (fd == STDOUT_FILENO)
286 m_stdout_bytes += count;
287 if (fd == STDERR_FILENO)
288 m_stderr_bytes += count;
289
290 break;
291 }
292
293 case -1:
294 default:
295 break;
296 }
297
298 LOG_PRINT("Syscall finished");
299 CLOG("syscall", "Finished thread %d", m_thread->getId());
300
301 return m_stalled;
302}

304IntPtr SyscallMdl::runExit(IntPtr old_return)
305{
306 CLOG("syscall", "Exit thread %d", m_thread->getId());
307
308 if (m_stalled)
309 {
310 SubsecondTime time_wake = Sim()->getClockSkewMinimizationServer()->getGlobalTime(true /*upper_bound*/);
311
312 {
313 // System call is blocking, mark thread as awake
314 ScopedLock sl(Sim()->getThreadManager()->getLock());
315 Sim()->getThreadManager()->resumeThread_async(m_thread->getId(), INVALID_THREAD_ID, time_wake, NULL);
316 }
317
318 Core *core = Sim()->getCoreManager()->getCurrentCore();
319 m_thread->reschedule(time_wake, core);
320 core = m_thread->getCore();
321
322 core->getPerformanceModel()->queuePseudoInstruction(new SyncInstruction(time_wake,
323 m_syscall_number == SYS_pause ? SyncInstruction::PAUSE : SyncInstruction::SYSCALL));
324
325 m_stalled = false;
326 }
327
328 if (!m_emulated)
329 {
330 m_ret_val = old_return;
331 }
332
333 Core *core = m_thread->getCore();
334 HookSyscallExit hook_args;
335 hook_args.thread_id = m_thread->getId();
336 hook_args.core_id = core->getId();
337 hook_args.time = core->getPerformanceModel()->getElapsedTime();
338 hook_args.ret_val = m_ret_val;
339 hook_args.emulated = m_emulated;
340 {
341 ScopedLock sl(Sim()->getThreadManager()->getLock());
342 Sim()->getHooksManager()->callHooks(HookType::HOOK_SYSCALL_EXIT, (UInt64)&hook_args);
343 }
344
345 m_emulated = false;
346 m_in_syscall = false;
347
348 return m_ret_val;
349}

357IntPtr SyscallMdl::handleFutexCall(syscall_args_t &args)
358{
359 SyscallServer::futex_args_t fargs;
360 fargs.uaddr = (int*) args.arg0;
361 fargs.op = (int) args.arg1;
362 fargs.val = (int) args.arg2;
363 fargs.timeout = (const struct timespec*) args.arg3;
364 fargs.val2 = (int) args.arg3;
365 fargs.uaddr2 = (int*) args.arg4;
366 fargs.val3 = (int) args.arg5;
367
368 int cmd = (fargs.op & FUTEX_CMD_MASK) & ~FUTEX_PRIVATE_FLAG;
369
370 Core *core = m_thread->getCore();
371 LOG_ASSERT_ERROR(core != NULL, "Syscall by thread %d: core should not be null", m_thread->getId());
372
373 SubsecondTime start_time;
374 SubsecondTime end_time;
375 start_time = core->getPerformanceModel()->getElapsedTime();
376
377 updateState(core, PthreadEmu::STATE_WAITING);
378
379 IntPtr ret_val = Sim()->getSyscallServer()->handleFutexCall(m_thread->getId(), fargs, start_time, end_time);
380
381 if (m_thread->reschedule(end_time, core))
382 core = m_thread->getCore();
383
384 core->getPerformanceModel()->queuePseudoInstruction(new SyncInstruction(end_time, SyncInstruction::FUTEX));
385
386 SubsecondTime delay = end_time - start_time;
387
388 updateState(core, PthreadEmu::STATE_RUNNING, delay);
389
390 // Update the futex statistics
391 futexCount(cmd, delay);
392
393 return ret_val;
394}
