193UInt64
194DramPerfModelDisagg::parseAddressBits(UInt64 address, UInt32 &data, UInt32 offset, UInt32 size, UInt64 base_address = 0)
195{
196 //parse data from the address based on the offset and size, return the address without the bits used to parse the data.
197 UInt32 log2_size = floorLog2(size);
198 if (base_address != 0) {
199 data = (base_address >> offset) % size;
200 } else {
201 data = (address >> offset) % size;
202 }
203 return ((address >> (offset + log2_size)) << offset) | (address & ((1 << offset) - 1));
204}

206void
207DramPerfModelDisagg::parseDeviceAddress(IntPtr address, UInt32 &channel, UInt32 &rank, UInt32 &bank_group, UInt32 &bank, UInt32 &column, UInt64 &page)
208{
209 // Construct DDR address which has bits used for interleaving removed
210 UInt64 linearAddress = m_address_home_lookup->getLinearAddress(address);
211 UInt64 address_bits = linearAddress >> 6;
212 /*address_bits = */parseAddressBits(address_bits, channel, m_channel_offset, m_num_channels, m_channel_offset < m_home_lookup_bit ? address : linearAddress);
213 address_bits = parseAddressBits(address_bits, rank, m_rank_offset, m_num_ranks, m_rank_offset < m_home_lookup_bit ? address : linearAddress);
214
215
216 if (m_open_page_mapping)
217 {
218 // Open-page mapping: column address is bottom bits, then bank, then page
219 if(m_column_offset)
220 {
221 // Column address is split into 2 halves ColHi and ColLo and
222 // the address looks like: | Page | ColHi | Bank | ColLo |
223 // m_column_offset specifies the number of ColHi bits
224 column = (((address_bits >> m_column_hi_offset) << m_bank_offset)
225 | (address_bits & ((1 << m_bank_offset) - 1))) % m_dram_page_size;
226 address_bits = address_bits >> m_bank_offset;
227 bank_group = address_bits % m_num_bank_groups;
228 bank = address_bits % m_num_banks;
229 address_bits = address_bits >> (m_num_banks_log2 + m_column_offset);
230 }
231 else
232 {
233 column = address_bits % m_dram_page_size; address_bits /= m_dram_page_size;
234 bank_group = address_bits % m_num_bank_groups;
235 bank = address_bits % m_num_banks; address_bits /= m_num_banks;
236 }
237 page = address_bits;
238
239#if 0
240 // Test address parsing done in this function for open page mapping
241 std::bitset<10> bs_col (column);
242 std::string str_col = bs_col.to_string<char,std::string::traits_type,std::string::allocator_type>();
243 std::stringstream ss_original, ss_recomputed;
244 ss_original << std::bitset<64>(linearAddress >> m_block_size_log2) << std::endl;
245 ss_recomputed << std::bitset<50>(page) << str_col.substr(0,m_column_offset) << std::bitset<4>(bank)
246 << str_col.substr(m_column_offset, str_col.length()-m_column_offset) << std::endl;
247 LOG_ASSERT_ERROR(ss_original.str() == ss_recomputed.str(), "Error in device address parsing!");
248#endif
249 }
250 else
251 {
252 bank_group = address_bits % m_num_bank_groups;
253 bank = address_bits % m_num_banks;
254 address_bits /= m_num_banks;
255
256 // Closed-page mapping: column address is bits X+banksize:X, row address is everything else
257 // (from whatever is left after cutting channel/rank/bank from the bottom)
258 column = (address_bits >> m_column_bits_shift) % m_dram_page_size;
259 page = (((address_bits >> m_column_bits_shift) / m_dram_page_size) << m_column_bits_shift)
260 | (address_bits & ((1 << m_column_bits_shift) - 1));
261 }
262
263 if(m_randomize_address)
264 {
265 std::bitset<3> row_bits(page >> m_randomize_offset); // Row[offset+2:offset]
266 UInt32 row_bits3 = row_bits.to_ulong();
267 row_bits[2] = 0;
268 UInt32 row_bits2 = row_bits.to_ulong();
269 bank_group ^= ((m_num_bank_groups == 8) ? row_bits3 : row_bits2); // BankGroup XOR Row
270 bank /= m_num_bank_groups;
271 bank ^= row_bits2; // Bank XOR Row
272 bank = m_banks_per_bank_group * bank_group + bank;
273 rank = (m_num_ranks > 1) ? rank ^ row_bits[0] : rank; // Rank XOR Row
274 }
275
276 //printf("[%2d] address %12lx linearAddress %12lx channel %2x rank %2x bank_group %2x bank %2x page %8lx crb %4u\n", m_core_id, address, linearAddress, channel, rank, bank_group, bank, page, (((channel * m_num_ranks) + rank) * m_num_banks) + bank);
277}

527 std::map<UInt64, SubsecondTime>::iterator i;
528 SubsecondTime max=SubsecondTime::Zero();
529 if(Sim()->getClockSkewMinimizationServer()->getGlobalTime()>pkt_time){

541 parseDeviceAddress(address, channel, rank, bank_group, bank, column, page);

544
545 SubsecondTime t_now = pkt_time;
546 perf->updateTime(t_now);
547
548 // DDR controller pipeline delay
549 t_now += m_controller_delay;
550 perf->updateTime(t_now, ShmemPerf::DRAM_CNTLR);
551
552 // DDR refresh
553 if (m_refresh_interval != SubsecondTime::Zero())
554 {
555 SubsecondTime refresh_base = (t_now.getPS() / m_refresh_interval.getPS()) * m_refresh_interval;
556 if (t_now - refresh_base < m_refresh_length)
557 {
558 t_now = refresh_base + m_refresh_length;
559 perf->updateTime(t_now, ShmemPerf::DRAM_REFRESH);
560 }
561 }
562
563 // Page hit/miss
564 UInt64 crb = (channel * m_num_ranks * m_num_banks) + (rank * m_num_banks) + bank; // Combine channel, rank, bank to index m_banks
565 LOG_ASSERT_ERROR(crb < m_total_banks, "Bank index out of bounds");
566 BankInfo &bank_info = m_banks[crb];
567
568 //printf("[%2d] %s (%12lx, %4lu, %4lu), t_open = %lu, t_now = %lu, bank_info.t_avail = %lu\n", m_core_id, bank_info.open_page == page && bank_info.t_avail + m_bank_keep_open >= t_now ? 569 // Page hit/miss
570 if ((bank_info.open_page == page) // Last access was to this row
571 && bank_info.t_avail + m_bank_keep_open >= t_now // Bank hasn't been closed in the meantime
572 )
573 {
574 if (bank_info.t_avail > t_now)
575 {
576 t_now = bank_info.t_avail;
577 perf->updateTime(t_now, ShmemPerf::DRAM_BANK_PENDING);
578 }
579 if((bank_info.core_id != requester) && selective_constant_time_policy){
580 t_now += m_bank_close_delay;

583

602 {

605 //std::cout<<"Page conflict md\n";

609 //std::cout<<"Page conflict dm\n";

613 //std::cout<<"Page conflict md\n";

617 //std::cout<<"Page conflict dm\n";

621 }

623 {
624 // Bank was being closed, we have to wait for that to complete
625 t_now = bank_info.t_avail + m_bank_keep_open +m_bank_close_delay;
626
627 ++m_page_closing;
628 }
629 else
630 {
631 // Bank was already closed, no delay.
632 ++m_page_empty;
633 if(constant_time_policy)
634 t_now += m_bank_close_delay;
635 }
636
637 // Open page
638 t_now += m_bank_open_delay;
639 perf->updateTime(t_now, ShmemPerf::DRAM_BANK_CONFLICT);
640
641 if(is_metadata){
642 bank_info.open_page_type=page_type::METADATA;
643 }
644 else{
645 bank_info.open_page_type=page_type::NOT_METADATA;
646 }
647 if(open_row_policy)
648 bank_info.open_page = page;
649 else
650 bank_info.open_page = 0;
651
652 }
653 bank_info.core_id = requester;
654 // Rank access time and availability
655 UInt64 cr = (channel * m_num_ranks) + rank;
656 LOG_ASSERT_ERROR(cr < m_total_ranks, "Rank index out of bounds");
657 SubsecondTime rank_avail_request = (m_num_bank_groups > 1) ? m_intercommand_delay_short : m_intercommand_delay;
658 SubsecondTime rank_avail_delay = m_rank_avail.size() ? m_rank_avail[cr]->computeQueueDelay(t_now, rank_avail_request, requester) : SubsecondTime::Zero();
659
660 // Bank group access time and availability
661 UInt64 crbg = (channel * m_num_ranks * m_num_bank_groups) + (rank * m_num_bank_groups) + bank_group;
662 LOG_ASSERT_ERROR(crbg < m_total_bank_groups, "Bank-group index out of bounds");
663 SubsecondTime group_avail_delay = m_bank_group_avail.size() ? m_bank_group_avail[crbg]->computeQueueDelay(t_now, m_intercommand_delay_long, requester) : SubsecondTime::Zero();
664
665 // Device access time (tCAS)
666 t_now += m_dram_access_cost;
667 perf->updateTime(t_now, ShmemPerf::DRAM_DEVICE);
668
669 // Mark bank as busy until it can receive its next command
670 // Done before waiting for the bus to be free: sort of assumes best-case bus scheduling
671 bank_info.t_avail = t_now;
672
673 // Add the wait time for the larger of bank group and rank availability delay
674 t_now += (rank_avail_delay > group_avail_delay) ? rank_avail_delay : group_avail_delay;
675 perf->updateTime(t_now, ShmemPerf::DRAM_DEVICE);
676
677 // DDR bus latency and queuing delay
678 SubsecondTime ddr_processing_time = m_bus_bandwidth.getRoundedLatency(8 * pkt_size); // bytes to bits
679 //std::cout << m_bus_bandwidth.getRoundedLatency(8*pkt_size) << std::endl;
680 SubsecondTime ddr_queue_delay = m_queue_model.size() ? m_queue_model[channel]->computeQueueDelay(t_now, ddr_processing_time, requester) : SubsecondTime::Zero();
681 t_now += ddr_queue_delay;
682 perf->updateTime(t_now, ShmemPerf::DRAM_QUEUE);
683 t_now += ddr_processing_time;
684 perf->updateTime(t_now, ShmemPerf::DRAM_BUS);
685
686 // Update Memory Counters -- Nandita:Removing this for now... not sure it's used anywhere
687 //queue_delay = ddr_queue_delay;
688
689 //std::cout << "Local Latency: " << t_now - pkt_time << std::endl;
690
691 //std::cout << "Result: Local" << std::endl;
692 return t_now - pkt_time;
