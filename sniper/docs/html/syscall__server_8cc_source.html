176int SyscallServer::futexDoOp(Core *core, int encoded_op, int *uaddr)
177{
178 int op = (encoded_op >> 28) & 7;
179 int cmp = (encoded_op >> 24) & 15;
180 int oparg = (encoded_op << 8) >> 20;
181 int cmparg = (encoded_op << 20) >> 20;
182 int oldval, newval;
183 int ret = 0;
184
185 if (encoded_op & (FUTEX_OP_OPARG_SHIFT << 28))
186 oparg = 1 << oparg;
187
188 core->accessMemory(Core::LOCK, Core::READ_EX, (IntPtr) uaddr, (char*) &oldval, sizeof(oldval));
189
190 switch (op) {
191 case FUTEX_OP_SET:
192 newval = oparg;
193 break;
194 case FUTEX_OP_ADD:
195 newval = oldval + oparg;
196 break;
197 case FUTEX_OP_OR:
198 newval = oldval | oparg;
199 break;
200 case FUTEX_OP_ANDN:
201 newval = oldval & ~oparg;
202 break;
203 case FUTEX_OP_XOR:
204 newval = oldval ^ oparg;
205 break;
206 default:
207 LOG_ASSERT_ERROR(false, "futexWakeOp: unknown op = %d", op);
208 }
209
210 if (op != FUTEX_OP_SET)
211 {
212 // TODO: Implement these using an atomic read-modify-write version of core->accessMemory
213 LOG_PRINT_WARNING_ONCE("FUTEX_WAKE_OP is implemented non-atomically, race condition may have occured");
214 }
215
216 core->accessMemory(Core::UNLOCK, Core::WRITE, (IntPtr) uaddr, (char*) &newval, sizeof(newval));
217
218 switch (cmp) {
219 case FUTEX_OP_CMP_EQ:
220 ret = (oldval == cmparg);
221 break;
222 case FUTEX_OP_CMP_NE:
223 ret = (oldval != cmparg);
224 break;
225 case FUTEX_OP_CMP_LT:
226 ret = (oldval < cmparg);
227 break;
228 case FUTEX_OP_CMP_GE:
229 ret = (oldval >= cmparg);
230 break;
231 case FUTEX_OP_CMP_LE:
232 ret = (oldval <= cmparg);
233 break;
234 case FUTEX_OP_CMP_GT:
235 ret = (oldval > cmparg);
236 break;
237 default:
238 LOG_ASSERT_ERROR(false, "futexWakeOp: unknown cmp = %d", cmp);
239 }
240
241 return ret;
242}

281{

283 SimFutex *sim_futex = findFutexByUaddr(uaddr, thread_id);

285

320void SyscallServer::futexPeriodic(SubsecondTime time)
321{
322 // Wake sleeping threads
323 for(SimFutex::ThreadQueue::iterator it = m_sleeping.begin(); it != m_sleeping.end(); ++it)
324 {
325 if (it->timeout <= time)
326 {
327 thread_id_t waiter = it->thread_id;
328 m_sleeping.erase(it);
329
330 Sim()->getThreadManager()->resumeThread(waiter, waiter, time, (void*)false);
331
332 // Iterator will be invalid, wake up potential others in the next barrier synchronization which should only be 100ns away
333 break;
334 }
335 }
336 // Wake timeout futexes
337 for(FutexMap::iterator it = m_futexes.begin(); it != m_futexes.end(); ++it)
338 {
339 it->second.wakeTimedOut(time);
340 }
