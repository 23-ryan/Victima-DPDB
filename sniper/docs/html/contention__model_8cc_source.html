161ContentionModel::getCompletionTime(SubsecondTime t_start, SubsecondTime t_delay, UInt64 tag)
162{
163 if (m_num_outstanding == 0)
164 return t_start + t_delay;
165
166 SubsecondTime t_end;
167
168 if (t_start == SubsecondTime::Zero())
169 t_end = t_delay;
170
171 else if (t_start < m_t_last)
172 {
173 // if (m_name == "ptw.mshr"){
174 // std::cout << "Out of order PTW so: reset " << std::endl;
175 // }
176 /* Out of order packet. Assume no congestion, only transfer latency. */
177 t_end = t_start + t_delay;
178
179 ++m_n_outoforder;
180
181 #if 1
182 /* Update time of last seen item */
183 m_t_last = t_start;
184
185 /* Reset all counters to start again from now */
186 m_time[0].first = t_end;
187 m_time[0].second = tag;
188 for(UInt32 i = 1; i < m_num_outstanding; ++i)
189 {
190 m_time[i].first = SubsecondTime::Zero();
191 m_time[i].second = 0;
192 }
193 #endif
194
195 }
196 else
197 {
198
199 // if (m_name == "ptw.mshr"){
200 // std::cout << "In order PTW so: reset " << std::endl;
201 // }
202
203 if (t_start == m_t_last)
204 m_n_simultaneous ++;
205
206 UInt64 unit = 0;
207 /* Find first free entry */
208 for(UInt32 i = 0; i < m_num_outstanding; ++i)
209 {
210 if (m_time[i].first <= t_start)
211 {
212 /* This one is free now */
213 unit = i;
214 break;
215 }
216 else if (m_time[i].first < m_time[unit].first)
217 {
218 /* Unit i is the first one free */
219 unit = i;
220 }
221 }
222
223 SubsecondTime t_begin;
224 if (t_start < m_time[unit].first)
225 /* Delay until the time the first unit becomes free */
226 t_begin = m_time[unit].first;
227 else
228 /* We only arrive after this unit became free */
229 t_begin = t_start;
230 /* Compute end of packet sending time */
231 t_end = t_begin + t_delay;
232
233 m_time[unit].first = t_end;
234 m_time[unit].second = tag;
235
236 /* Update statistics */
237 m_total_delay += t_begin - t_start;
238
239 /* Update time of last seen item */
240 m_t_last = t_start;
241 }
242
243 ++m_n_requests;
244
245 return t_end;
246}
