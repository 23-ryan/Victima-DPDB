345{
346 Core *core = localStore[thread_id].thread->getCore();

348 PthreadEmu::state_t new_state;

393 switch(clk_id)

431void emuKmpReapMonitor(THREADID threadIndex, CONTEXT *ctxt)
432{
433 // Hack to make ICC's OpenMP runtime library work.
434 // This runtime creates a monitor thread which blocks in a condition variable with a timeout.
435 // On exit, thread 0 executes __kmp_reap_monitor() which join()s on this monitor thread.
436 // In due time, the timeout occurs and the monitor thread returns
437 // from pthread_cond_timedwait(), sees that it should be exiting, and returns.
438 // However, in simulation all of this happens post-ROI, where time is not advancing so the timeout never occurs.
439 // Having time advance using a one-IPC model during pre- and post-ROI would be nice, but for now,
440 // just forcefully terminate the application when the master thread reaches __kmp_reap_monitor().
441 void *res;
442#if PIN_REV >= 71313
443 PIN_CallApplicationFunction(ctxt, threadIndex, CALLINGSTD_DEFAULT, ptr_exit, NULL, PIN_PARG(void*), &res, PIN_PARG(int), 0, PIN_PARG_END());
444#else
445 PIN_CallApplicationFunction(ctxt, threadIndex, CALLINGSTD_DEFAULT, ptr_exit, PIN_PARG(void*), &res, PIN_PARG(int), 0, PIN_PARG_END());
446#endif
447}
