131// _LOG_PRINT(Log::Error) does not return, but the compiler doesn't know this which can result in
132// ''control reaches end of non-void function'' warnings in places where the LOG_PRINT_ASSERT(false) idiom is used.
133// We cannot easily add __attribute__ ((__noreturn__)) to the LOG_PRINT_ERROR macro
134// since that only works on functions, and we have the added complexity of passing the ... argument around.
135// Adding exit() to the end of LOG_PRINT_ERROR effectively makes it inherit exit's __noreturn__
136// which suppresses the warning (even though the exit() itself isn't executed as it's never reached).
137
138#define LOG_PRINT_ERROR(...) \
139 { \
140 _LOG_PRINT(Log::Error, __VA_ARGS__); \
141 exit(-1); \
142 }
