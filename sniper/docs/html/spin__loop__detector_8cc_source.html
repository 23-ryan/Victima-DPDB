10void SpinLoopDetector::commitBCT(uint64_t eip)
11{
12 // We just executed a backward control transfer (BCT)
13 // ``A BCT can be a backward conditional taken branch, unconditional branch, or jump instruction.''
14
15 // Note: since we keep the SDT as a queue, a BCT's position is not constant so we use explicit ids
16 // We also have an unbounded RUB so there is no overflow bit
17
18 Core *core = m_thread->getCore();
19
20 // Find BCT in the SDT
21 Sdt::iterator entry = m_sdt.begin();
22 for( ; entry != m_sdt.end(); ++entry)
23 if (entry->eip == eip)
24 break;
25
26 if (entry == m_sdt.end())
27 {
28 // Unknown BCT: add to SDT, increment next SDT entry id, and update mask of valid ids
29 m_sdt.push_back(SdtEntry(eip, m_sdt_nextid, core->getInstructionCount(), core->getPerformanceModel()->getElapsedTime()));
30 m_sdt_nextid = (m_sdt_nextid + 1) % SDT_MAX_SIZE;
31
32 // Keep only top SDT_MAX_SIZE entries
33 if (m_sdt.size() > SDT_MAX_SIZE)
34 m_sdt.pop_front();
35
36 m_sdt_bitmask = 0;
37 for(Sdt::iterator it = m_sdt.begin(); it != m_sdt.end(); ++it)
38 m_sdt_bitmask |= (1 << it->id);
39 }
40 else
41 {
42 // Found! Check all RUB entries for our id to check if observable register state
43 // has changed since the end of last iteration (Spin Condition 1)
44 uint8_t id = entry->id;
45
46 bool spinning = true;
47 for(Rub::iterator it = m_rub.begin(); it != m_rub.end(); ++it)
48 {
49 if (it->second.second & (1 << id))
50 {
51 // ``There exists at least one entry in the RUB that corresponds to this SDT entry.
52 // This means that the observable register state of the thread differs and, thus,
53 // the thread was not spinning.''
54 spinning = false;
55 // Reset RUB entry for our loop id
56 it->second.second &= ~(1 << id);
57 }
58 }
59
60 if (spinning)
61 {
62 // Spin loop detected !!
63 core->updateSpinCount(
64 core->getInstructionCount() - entry->icount,
65 // This time difference will have been caused by instructions earlier in the queue, not by the instructions
66 // in this exact spin loop. But, when spinning for a long time, the earlier instructions will be spin loops as well
67 // so if spin count is large (which is when we care) it should be reasonably accurate.
68 core->getPerformanceModel()->getElapsedTime() - entry->tcount
69 );
70 }
71
72 entry->icount = core->getInstructionCount();
73 entry->tcount = core->getPerformanceModel()->getElapsedTime();
74 }
75}

84void SpinLoopDetector::commitRegisterWrite(reg_t reg, uint64_t old_value, uint64_t value)
85{
86 // Implements Register Update Buffer (RUB), see Section 3.2, 3rd paragraph
87
88 // ``For each instruction it then commits, the processor checks if the instruction's architectural
89 // (i.e., logical) destination register is already in the RUB.''
90 if (m_rub.count(reg) == 0)
91 {
92 // ``If not, the processor has discovered a new register written by the thread
93 // It then compares the new value of this register (i.e., the value being committed)
94 // to its old value (i.e., the value before being overwritten by the commit).''
95 if (value != old_value)
96 {
97 // ``If not equal, the processor adds the register number and its old value to the RUB.''
98 m_rub[reg] = std::make_pair(old_value, m_sdt_bitmask);
99 }
100 }
101 else
102 {
103 // ``If the register is already in the RUB, then the processor compares its new value to its current value in the RUB.''
104 if (value == m_rub[reg].first)
105 {
106 // ``If equal, it deletes this register from the RUB;''
107 m_rub.erase(reg);
108 }
109 // ``otherwise, no action is necessary.''
110 else
111 {
112 // Actually, this happens when the register is already in the RUB for an older loop.
113 // Unless it's a silent write, update the bitmask to represent all current loop candidates.
114 if (value != old_value)
115 m_rub[reg].second |= m_sdt_bitmask;
116 }
117 }
118}
