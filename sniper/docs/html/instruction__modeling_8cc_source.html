84static void handleRdtsc(THREADID thread_id, PIN_REGISTER * gax, PIN_REGISTER * gdx)
85{
86 Core *core = localStore[thread_id].thread->getCore();
87 assert (core);
88 SubsecondTime cycles_fs = core->getPerformanceModel()->getElapsedTime();
89 // Convert SubsecondTime to cycles in global clock domain
90 const ComponentPeriod *dom_global = Sim()->getDvfsManager()->getGlobalDomain();
91 UInt64 cycles = SubsecondTime::divideRounded(cycles_fs, *dom_global);
92 // Return in eax and edx
93 gdx->dword[0] = cycles >> 32;
94 gax->dword[0] = cycles & 0xffffffff;
95}

118static void fillOperandListMemOps(OperandList *list, INS ins)
119{
120 if (INS_IsMemoryRead (ins) || INS_IsMemoryWrite (ins))
121 {
122 // first all reads (dyninstrinfo pushed from redirectMemOp)
123 for (unsigned int i = 0; i < INS_MemoryOperandCount(ins); i++)
124 {

128 // then all writes (dyninstrinfo pushed from completeMemWrite)

137static void fillOperandList(OperandList *list, INS ins)
138{
139 // memory
140 fillOperandListMemOps(list, ins);
141
142 // for handling register operands
143 unsigned int max_read_regs = INS_MaxNumRRegs(ins);
144 unsigned int max_write_regs = INS_MaxNumWRegs(ins);
145
146 for (unsigned int i = 0; i < max_read_regs; i++)
147 {
148 REG reg_i = INS_RegR(ins, i);
149 if (REG_valid(reg_i))
150 list->push_back(Operand(Operand::REG, reg_i, Operand::READ, REG_StringShort(reg_i).c_str(), INS_RegRContain(ins, reg_i)));
151 }
152
153 for (unsigned int i = 0; i < max_write_regs; i++)
154 {
155 REG reg_i = INS_RegW(ins, i);
156 if (REG_valid(reg_i))
157 list->push_back(Operand(Operand::REG, reg_i, Operand::WRITE, REG_StringShort(reg_i).c_str(), INS_RegWContain(ins, reg_i)));
158 }
159
160 // immediate
161 for (unsigned int i = 0; i < INS_OperandCount(ins); i++)
162 {
163 if (INS_OperandIsImmediate(ins, i))
164 {
165 list->push_back(Operand(Operand::IMMEDIATE, INS_OperandImmediate(ins, i), Operand::READ));
166 }
167 }
168}

172VOID InstructionModeling::addInstructionModeling(TRACE trace, INS ins, InstMode::inst_mode_t inst_mode)
173{
174 // Functional modeling
175
176 // Simics-style magic instruction: xchg bx, bx
177 if (INS_IsXchg(ins) && INS_OperandReg(ins, 0) == REG_BX && INS_OperandReg(ins, 1) == REG_BX)
178 {
179 if (Sim()->getConfig()->getEnableSyscallEmulation())
180 {
181 INS_InsertPredicatedCall(ins, IPOINT_BEFORE, (AFUNPTR)handleMagic, IARG_THREAD_ID, IARG_REG_VALUE, REG_GAX,
182 #ifdef TARGET_IA32
183 IARG_REG_VALUE, REG_GDX,
184 #else
185 IARG_REG_VALUE, REG_GBX,
186 #endif
187 IARG_REG_VALUE, REG_GCX, IARG_RETURN_REGS, REG_GAX, IARG_END);
188 // Stop the trace after MAGIC (Redmine #118), which has potentially changed the instrumentation mode,
189 // so execution can resume in the correct instrumentation version
190 INS_InsertDirectJump(ins, IPOINT_AFTER, INS_Address(ins) + INS_Size(ins));
191 }
192 else
193 {
194 INS_InsertPredicatedCall(ins, IPOINT_BEFORE, (AFUNPTR)handleMagic, IARG_THREAD_ID, IARG_REG_VALUE, REG_GAX,
195 #ifdef TARGET_IA32
196 IARG_REG_VALUE, REG_GDX,
197 #else
198 IARG_REG_VALUE, REG_GBX,
199 #endif
200 IARG_REG_VALUE, REG_GCX, IARG_END);
201 }
202 }
203
204 if (Sim()->getConfig()->getEnableSyscallEmulation())
205 {
206 if (INS_IsRDTSC(ins))
207 INS_InsertPredicatedCall(ins, IPOINT_AFTER, (AFUNPTR)handleRdtsc, IARG_THREAD_ID, IARG_REG_REFERENCE, REG_GAX, IARG_REG_REFERENCE, REG_GDX, IARG_END);
208
209 if (INS_Opcode(ins) == XED_ICLASS_CPUID)
210 {
211 INS_InsertPredicatedCall(ins, IPOINT_BEFORE, (AFUNPTR)handleCpuid, IARG_THREAD_ID, IARG_REG_REFERENCE, REG_GAX, IARG_REG_REFERENCE, REG_GBX, IARG_REG_REFERENCE, REG_GCX, IARG_REG_REFERENCE, REG_GDX, IARG_END);
212 INS_Delete(ins);
213 }
214 }
215
216 if (INS_Opcode(ins) == XED_ICLASS_PAUSE)
217 INS_InsertPredicatedCall(ins, IPOINT_AFTER, (AFUNPTR)handlePause, IARG_END);
218
219 if (INS_IsSyscall(ins) && Sim()->getConfig()->getEnableSyscallEmulation())
220 {
221 INS_InsertPredicatedCall(ins, IPOINT_BEFORE,
222 AFUNPTR(lite::handleSyscall),
223 IARG_THREAD_ID,
224 IARG_CONTEXT,
225 IARG_END);
226 }
227
228
229 // Set up localStore[thread_id].dynins
230
231 if (INSTR_IF_DETAILED(inst_mode) && !INS_IsSyscall(ins))
232 {
233 Instruction *inst = InstructionModeling::decodeInstruction(ins);
234 INSTRUMENT(INSTR_IF_DETAILED(inst_mode), trace, ins, IPOINT_BEFORE, AFUNPTR(InstructionModeling::handleInstruction), IARG_THREAD_ID, IARG_PTR, inst, IARG_END);
235 }
236
237
238 // Timing models, will add dynamic information to localStore[thread_id].dynins
239
240 if (INS_IsBranch(ins) && INS_HasFallThrough(ins))
241 {
242 // In warming mode, warm up the branch predictors
243 INSTRUMENT_PREDICATED(
244 INSTR_IF_CACHEONLY(inst_mode),
245 trace, ins, IPOINT_TAKEN_BRANCH, (AFUNPTR)handleBranchWarming,
246 IARG_THREAD_ID,
247 IARG_ADDRINT, INS_Address(ins),
248 IARG_BOOL, TRUE,
249 IARG_BRANCH_TARGET_ADDR,
250 IARG_END);
251
252 INSTRUMENT_PREDICATED(
253 INSTR_IF_CACHEONLY(inst_mode),
254 trace, ins, IPOINT_AFTER, (AFUNPTR)handleBranchWarming,
255 IARG_THREAD_ID,
256 IARG_ADDRINT, INS_Address(ins),
257 IARG_BOOL, FALSE,
258 IARG_BRANCH_TARGET_ADDR,
259 IARG_END);
260
261 // In detailed mode, push a DynamicInstructionInfo
262 INSTRUMENT_PREDICATED(
263 INSTR_IF_DETAILED(inst_mode),
264 trace, ins, IPOINT_TAKEN_BRANCH, (AFUNPTR)handleBranch,
265 IARG_THREAD_ID,
266 IARG_ADDRINT, INS_Address(ins),
267 IARG_BOOL, TRUE,
268 IARG_BRANCH_TARGET_ADDR,
269 IARG_END);
270
271 INSTRUMENT_PREDICATED(
272 INSTR_IF_DETAILED(inst_mode),
273 trace, ins, IPOINT_AFTER, (AFUNPTR)handleBranch,
274 IARG_THREAD_ID,
275 IARG_ADDRINT, INS_Address(ins),
276 IARG_BOOL, FALSE,
277 IARG_BRANCH_TARGET_ADDR,
278 IARG_END);
279 }
280
281 if (!INS_IsSyscall(ins))
282 {
283 // Instrument Memory Operations
284 lite::addMemoryModeling(trace, ins, inst_mode);
285 }
286
287 // Spin loop detection
288 if (Sim()->getConfig()->getEnableSpinLoopDetection())
289 addSpinLoopDetection(trace, ins, inst_mode);
290}

292Instruction* InstructionModeling::decodeInstruction(INS ins)
293{
294 // Timing modeling
295
296 OperandList list;
297 fillOperandList(&list, ins);
298
299 Instruction *inst;
300
301 // branches
302 if (INS_IsBranch(ins) && INS_HasFallThrough(ins))
303 {
304 inst = new BranchInstruction(list);
305 }
306 // Now handle instructions which have a static cost
307 else
308 {
309 switch(INS_Opcode(ins))
310 {
311 case XED_ICLASS_DIV:
312 inst = new ArithInstruction(INST_DIV, list);
313 break;
314 case XED_ICLASS_MUL:
315 inst = new ArithInstruction(INST_MUL, list);
316 break;
317 case XED_ICLASS_FDIV:
318 inst = new ArithInstruction(INST_FDIV, list);
319 break;
320 case XED_ICLASS_FMUL:
321 inst = new ArithInstruction(INST_FMUL, list);
322 break;
323 default:
324 inst = new GenericInstruction(list);
325 }
326 }
327
328 ADDRINT addr = INS_Address(ins);
329
330 inst->setAddress(addr);
331 inst->setSize(INS_Size(ins));
332 inst->setAtomic(INS_IsAtomicUpdate(ins));
333 inst->setDisassembly(INS_Disassemble(ins).c_str());
334
335
336 const std::vector<const MicroOp *> * uops = InstructionDecoder::decode(INS_Address(ins), INS_XedDec(ins), inst);
337 inst->setMicroOps(uops);
338
339 return inst;
340}

342VOID InstructionModeling::countInstructions(THREADID thread_id, ADDRINT address, INT32 count)
343{
344 if (!Sim()->isRunning())
345 {
346 // Main thread has exited, but we still seem to be running.
347 // Don't touch any more simulator structure as they're being deallocated right now.
348 // Just wait here until the whole application terminates us.
349 while(1)
350 sched_yield();
351 }
352
353 Core* core = localStore[thread_id].thread->getCore();
354 assert(core);
355 bool check_rescheduled = core->countInstructions(address, count);
356
357 if (check_rescheduled)
358 {
359 // If countInstructions returns true, we may have been rescheduled (using the fast-forward performance model)
360 SubsecondTime time;
361 if (localStore[thread_id].thread->reschedule(time, core))
362 {
363 core = localStore[thread_id].thread->getCore();
364 core->getPerformanceModel()->queuePseudoInstruction(new SyncInstruction(time, SyncInstruction::UNSCHEDULED));
365 }
366 }
367}
