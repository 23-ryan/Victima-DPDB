99FastForwardPerformanceManager::step()
100{

102 m_target_sync_time = Sim()->getClockSkewMinimizationServer()->getGlobalTime() + m_sync_interval;
103
104 SubsecondTime barrier_next = SubsecondTime::Zero();

106 {
107 PerformanceModel *perf = Sim()->getCoreManager()->getCoreFromID(core_id)->getPerformanceModel();
108 perf->setFastForward(true);
109 barrier_next = std::max(barrier_next, perf->getElapsedTime());
110 recalibrateInstructionsCallback(core_id);
111 }
112 // Set barrier to fastforward, and update next_barrier_time to the maximum of all core times so we definitely release everyone
113 Sim()->getClockSkewMinimizationServer()->setFastForward(true, barrier_next);
