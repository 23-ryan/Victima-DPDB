48 //Determine the type of a given key by trying to cast it to different types
49 //and catching the error.
50 unsigned short Key::DetermineType(String value)
51 {
52 //strings are always valid
53 unsigned short valid = TYPE_STRING_VALID;
54
55 // Try for floats
56 if(parse(value.c_str(),real_p).full)
57 {
58 try
59 {
60 m_value_f = boost::lexical_cast<double>(value);
61 valid |= TYPE_FLOAT_VALID;
62 }
63 catch(const boost::bad_lexical_cast &)
64 {
65 }
66 }
67
68 // and ints
69 if(parse(value.c_str(),long_p).full)
70 {
71 try
72 {
73 m_value_i = boost::lexical_cast<SInt64>(value);
74 valid |= TYPE_INT_VALID;
75 }
76 catch(const boost::bad_lexical_cast &)
77 {
78 }
79 }
80
81 // and bools
82 String icase_value = m_value;
83 boost::to_lower(icase_value);
84 if(icase_value == "true" || icase_value == "yes" || icase_value == "1")
85 {
86 valid |= TYPE_BOOL_VALID;
87 m_value_b = true;
88 }
89 else if(icase_value == "false" || icase_value == "no" || icase_value == "0")
90 {
91 valid |= TYPE_BOOL_VALID;
92 m_value_b = false;
93 }
94 return valid;
95 }
