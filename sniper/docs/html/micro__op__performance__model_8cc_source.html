109void MicroOpPerformanceModel::doSquashing(std::vector<DynamicMicroOp*> &current_uops, uint32_t first_squashed)
110{
111 MicroOp::uop_type_t uop_type = MicroOp::UOP_INVALID;
112 uint32_t i, size, squashedCount = 0, microOpTypeOffset = 0;
113
114 // recalculate microOpTypeOffsets for dynamic uops and
115 // collect squashing info
116 for(i = 0, size = current_uops.size(); i < size; i++)
117 {
118 DynamicMicroOp* uop = current_uops[i];
119 if(uop->isSquashed())
120 {
121 squashedCount++;
122 }
123 else
124 {
125 if(uop->getMicroOp()->getType() != uop_type)
126 {
127 uop_type = uop->getMicroOp()->getType();
128 microOpTypeOffset = 0;
129 }
130 uop->setMicroOpTypeOffset(microOpTypeOffset++);
131 }
132 uop->setSquashedCount(squashedCount);
133 }
134
135 // recalculate intraInstructionDependancies
136 for(i = first_squashed; i < size; i++)
137 {
138 DynamicMicroOp* uop = current_uops[i];
139 if(!uop->isSquashed())
140 {
141 uint32_t intraDeps = uop->getIntraInstrDependenciesLength();
142 if(intraDeps != 0)
143 {
144 uint32_t iBase = i - uop->getMicroOp()->microOpTypeOffset;
145 LOG_ASSERT_ERROR(iBase >= intraDeps, "intraInstructionDependancies (%d) should be <= (%d)", intraDeps, iBase);
146 uop->setIntraInstrDependenciesLength(intraDeps - (current_uops[iBase]->getSquashedCount() -
147 current_uops[iBase-intraDeps]->getSquashedCount()));
148 }
149 }
150 }
