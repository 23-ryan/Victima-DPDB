266
267 const ThreadStatsManager::ThreadStatTypeList& types = Sim()->getThreadStatsManager()->getThreadStatTypes();
268
269 // header line

293 fprintf(fp, "%s\t%" PRId64 "\t%" PRId64 "\t%" PRId64,
294 s.str().c_str(), it->second->m_calls, it->second->m_bits_used, it->second->m_bits_total);

320UInt64 RoutineTracerFunctionStats::ThreadStatAggregates::callback(ThreadStatsManager::ThreadStatType type, thread_id_t thread_id, Core *core, UInt64 user)
321{
322 UInt64 result = 0;
323
324 switch(user)
325 {
326 case GLOBAL_INSTRUCTIONS:
