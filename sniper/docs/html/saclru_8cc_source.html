108uint64_t
109CheetahSACLRU::hits(unsigned sets_log2, unsigned assoc)
110{
111 assert(sets_log2 >= A);
112 assert(sets_log2 <= B);
113
114 if (hitarr0)
115 flush();
116
117 uint64_t hits = 0;
118 for(unsigned int j = 0; j < assoc; ++j)
119 hits += sac_hits[j][sets_log2-A];
120
121 return hits;
122}

125/**********************************************************************
126Initialization routine. Allocates space for the various arrays and
127initializes them.
128
129Input: None
130Output: None
131Side effects: Allocates space for the arrays and initializes the array
132 locations.
133**********************************************************************/
134void
135CheetahSACLRU::init_saclru(void)
136{
137 uint64_t i, j, k, l;
138 uint64_t init_value;
139 uint64_t *arr_ptr, *slot_ptr;
140
141 next_save_time = SAVE_INTERVAL;
142 TWO_PWR_N = (ONE << N);
143 MAX_DEPTH = B-A;
144 TWO_POWER_MAX_DEPTH = (ONE << MAX_DEPTH);
145 SET_MASK = ((ONE << A) - 1);
146 DIFF_SET_MASK = ((ONE << MAX_DEPTH) - 1);
147 BASE = (TWO_PWR_N+1);
148 SIZE_OF_TREE = (TWO_POWER_MAX_DEPTH * TWO * BASE);
149
150 assert(((uint64_t)(ONE << A)*TWO_POWER_MAX_DEPTH*TWO*BASE) < (uint64_t(1)<<32));
151 arr = (uint64_t *)calloc((uint64_t(ONE << A)*TWO_POWER_MAX_DEPTH*TWO*BASE), sizeof(uint64_t));
152 if (!arr)
153 fatal("out of virtual memory");
154
155 BASE_PWR_MAX_DEPTH_PLUS_ONE = power(BASE, MAX_DEPTH+1);
156 if ((BASE_PWR_MAX_DEPTH_PLUS_ONE * sizeof(uint64_t)) < MEM_AVAIL_HITARR)
157 {
158 hitarr = (int64_t *)calloc(BASE_PWR_MAX_DEPTH_PLUS_ONE, sizeof(int64_t));
159 if (!hitarr)
160 fatal("out of virtual memory");
161 }
162
163 base_pwr_array = (uint64_t *)calloc((MAX_DEPTH+1), sizeof(uint64_t));
164 if (!base_pwr_array)
165 fatal("out of virtual memory");
166
167 rm_arr = (uint64_t *)calloc((TWO_POWER_MAX_DEPTH), sizeof(uint64_t));
168 if (!rm_arr)
169 fatal("out of virtual memory");
170
171 sac_hits = idim2((TWO_PWR_N), (MAX_DEPTH + 2));
172
173 for (i=0; i < ONE << A; i++)
174 {
175 arr_ptr = arr + i * SIZE_OF_TREE;
176 init_value = B80000000;
177 slot_ptr = arr_ptr;
178 *slot_ptr = 1;
179 for (l=1; l<=TWO_PWR_N; l++)
180 *(slot_ptr + l) = init_value;
181 ++init_value;
182 for (j=1; j <= MAX_DEPTH; j++)
183 {
184 for (k=((ONE << j) - 1);
185 k < (((ONE << j) - 1) + (ONE << (j-1)));
186 k++)
187 {
188 *(arr_ptr + (k*(TWO_PWR_N+1))) = TWO_PWR_N+1;
189 }
190 for (k= (((ONE << j) - 1) + (ONE << (j-1)));
191 k < ((ONE << (j+1)) - 1);
192 k++)
193 {
194 slot_ptr = arr_ptr + (k * (TWO_PWR_N+1));
195 *slot_ptr = 1;
196 for (l=1; l<=TWO_PWR_N; l++)
197 *(slot_ptr + l) = init_value;
198 ++init_value;
199 }
200 }
201 }
202 rm_arr[0] = MAX_DEPTH;
203 for (i=1; i < (ONE << MAX_DEPTH); i++)
204 {
205 for (j=0; j<=MAX_DEPTH; j++)
206 {
207 if (i & (ONE << j))
208 {
209 rm_arr[i] = j;
210 break;
211 }
212 }
213 }
214 j = 1;
215 for (i=0; i<=MAX_DEPTH; i++)
216 {
217 base_pwr_array[i] = j;
218 j *= BASE;
219 }
220
221 depths = (uint64_t *)calloc((MAX_DEPTH+1), sizeof(uint64_t));
222 if (!depths)
223 fatal("out of virtual memory");
224
225 for (i=0;i<=MAX_DEPTH;i++)
226 depths[i] = 0;
227}

248#if 0

262#if 0

269 arr_ptr = arr + set_no * SIZE_OF_TREE;

271#ifdef PERF

273#endif
274 if (*(arr_ptr + 1) == orig_tag)
275 ++hitarr0;

278 atag = orig_tag;
279 depth = 0;

292 *(slot_ptr+i) = tag;

297 ++depths[t];
298 *(slot_ptr + i) = tag;
299 tag = t1;
300 }
301 ++*slot_ptr;
302 entry =
303 (((ONE << depth) + (atag & ((ONE << depth) - 1))) - 1) * BASE;
304 slot_ptr = arr_ptr + entry;

306 slot_ptr[*slot_ptr] = atag;
307 if (hit==1)
308 {
309 sum = 0;
310 i = MAX_DEPTH;
311 while (i >= 0)
312 {
313 sum += depths[i];
314 if (sum < TWO_PWR_N)
315 {
316 ++sac_hits[sum][i];
317 --i;
318 }
319 else
320 break;
321 }
322 break;
323 }
324 ++depth;
325 atag = tag;
326 entry =
327 (((ONE << depth) + (orig_tag & ((ONE << depth)-1)))-1) * BASE;

329 {
330 ++depth;

333 }
334 if (depth <= MAX_DEPTH)
335 {
336 slot_ptr = arr_ptr + entry;
337 fst = *slot_ptr;
338 }
339 }

341 for (i=0;i<=(int)MAX_DEPTH;i++)
342 depths[i] = 0;
343 } /* else */
344}
