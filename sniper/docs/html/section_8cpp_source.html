57 const Key & Section::addKeyInternal(const String & name_, const V & value, uint64_t index)
58 {
59 //Make the key all lower-case if not case sensitive
60 String iname(name_);
61 if(!m_case_sensitive)
62 boost::to_lower(iname);
63
64 // Check for non-index version
65 if (index == UINT64_MAX)
66 {
67 //Remove existing default key
68 m_keys.erase(iname);
69
70 //Remove overrides
71 m_array_keys.erase(iname);
72
73 m_keys.insert(std::pair< String, Key* >(iname, new Key(this->getFullPath(),name_,value)));
74 return *(m_keys[iname]);
75 }
76 else
77 {
78 //Do not remove existing default key for override case
79
80 UInt64 needed_size = index + 1;
81 KeyArrayList::iterator found = m_array_keys.find(iname);
82 if(found != m_array_keys.end())
83 {
84 std::vector<Key*> & arr = (*found).second;
85 if (arr.size() < needed_size)
86 {
87 arr.resize(needed_size);
88 }
89 else
90 {
91 arr[index] = NULL;
92 }
93 }
94 else
95 {
96 m_array_keys[iname] = std::vector<Key*>(needed_size);
97 }
98
99 m_array_keys[iname][index] = new Key(this->getFullPath(),name_,value);
100
101 return *(m_array_keys[iname][index]);
102 }

111 const Key & Section::getKey(const String & name_, uint64_t index)
112 {
113 String iname(name_);
114 if(!m_case_sensitive)
115 boost::to_lower(iname);
116
117 if (index == UINT64_MAX)
118 {
119 // Default to using non-index version
120 return *(m_keys[iname]);

122 else
123 {
124 if ( (m_array_keys.find(iname) != m_array_keys.end()) &&

126 (m_array_keys[iname][index] != NULL) )

131 else

133 // Otherwise, return the value requested from the non-indexed version
134 return *(m_keys[iname]);
