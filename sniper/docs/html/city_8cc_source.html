100// Hash 128 input bits down to 64 bits of output.
101// This is intended to be a reasonably good hash function.
102static inline uint64 Hash128to64(const uint128 x) {
103 // Murmur-inspired hashing.
104 const uint64 kMul = 0x9ddfea08eb382d69ULL;
105 uint64 a = (Uint128Low64(x) ^ Uint128High64(x)) * kMul;
106 a ^= (a >> 47);
107 uint64 b = (Uint128High64(x) ^ a) * kMul;
108 b ^= (b >> 47);
109 b *= kMul;
110 return b;
111}

113// Bitwise right rotate. Normally this will compile to a single
114// instruction, especially if the shift is a manifest constant.
115static uint64 Rotate(uint64 val, int shift) {
116 // Avoid shifting by 64: doing so yields an undefined result.
117 return shift == 0 ? val : ((val >> shift) | (val << (64 - shift)));
118}
119
120// Equivalent to Rotate(), but requires the second arg to be non-zero.
121// On x86-64, and probably others, it's possible for this to compile
122// to a single instruction if both args are already in registers.
123static uint64 RotateByAtLeast1(uint64 val, int shift) {
124 return (val >> shift) | (val << (64 - shift));
125}

136static uint64 HashLen0to16(const char *s, size_t len) {
137 if (len > 8) {
138 uint64 a = Fetch64(s);
139 uint64 b = Fetch64(s + len - 8);
140 return HashLen16(a, RotateByAtLeast1(b + len, len)) ^ b;
141 }
142 if (len >= 4) {
143 uint64 a = Fetch32(s);
144 return HashLen16(len + (a << 3), Fetch32(s + len - 4));
145 }
146 if (len > 0) {
147 uint8 a = s[0];
148 uint8 b = s[len >> 1];
149 uint8 c = s[len - 1];
150 uint32 y = (uint32)(a) + ((uint32)(b) << 8);
151 uint32 z = len + ((uint32)(c) << 2);
152 return ShiftMix(y * k2 ^ z * k3) * k2;
153 }
154 return k2;
155}

159static uint64 HashLen17to32(const char *s, size_t len) {
160 uint64 a = Fetch64(s) * k1;
161 uint64 b = Fetch64(s + 8);
162 uint64 c = Fetch64(s + len - 8) * k2;
163 uint64 d = Fetch64(s + len - 16) * k0;
164 return HashLen16(Rotate(a - b, 43) + Rotate(c, 30) + d,
165 a + Rotate(b ^ k3, 20) - c + len);
166}

193// Return an 8-byte hash for 33 to 64 bytes.
194static uint64 HashLen33to64(const char *s, size_t len) {
195 uint64 z = Fetch64(s + 24);
196 uint64 a = Fetch64(s) + (len + Fetch64(s + len - 16)) * k0;
197 uint64 b = Rotate(a + z, 52);
198 uint64 c = Rotate(a, 37);
199 a += Fetch64(s + 8);
200 c += Rotate(a, 7);
201 a += Fetch64(s + 16);
202 uint64 vf = a + z;
203 uint64 vs = b + Rotate(a, 31) + c;
204 a = Fetch64(s + 16) + Fetch64(s + len - 32);
205 z = Fetch64(s + len - 8);
206 b = Rotate(a + z, 52);
207 c = Rotate(a, 37);
208 a += Fetch64(s + len - 24);
209 c += Rotate(a, 7);
210 a += Fetch64(s + len - 16);
211 uint64 wf = a + z;
212 uint64 ws = b + Rotate(a, 31) + c;
213 uint64 r = ShiftMix((vf + ws) * k2 + (wf + vs) * k0);
214 return ShiftMix(r * k0 + vs) * k2;
215}

217uint64 CityHash64(const char *s, size_t len) {
218 if (len <= 32) {
219 if (len <= 16) {
220 return HashLen0to16(s, len);
221 } else {
222 return HashLen17to32(s, len);
223 }
224 } else if (len <= 64) {
225 return HashLen33to64(s, len);
226 }
227
228 // For strings over 64 bytes we hash the end first, and then as we
229 // loop we keep 56 bytes of state: v, w, x, y, and z.
230 uint64 x = Fetch64(s + len - 40);
231 uint64 y = Fetch64(s + len - 16) + Fetch64(s + len - 56);
232 uint64 z = HashLen16(Fetch64(s + len - 48) + len, Fetch64(s + len - 24));
233 uint64 temp;
234 uint128 v = WeakHashLen32WithSeeds(s + len - 64, len, z);
235 uint128 w = WeakHashLen32WithSeeds(s + len - 32, y + k1, x);
236 x = x * k1 + Fetch64(s);
237
238 // Decrease len to the nearest multiple of 64, and operate on 64-byte chunks.
239 len = (len - 1) & ~(size_t)(63);
240 do {
241 x = Rotate(x + y + v.first + Fetch64(s + 8), 37) * k1;
242 y = Rotate(y + v.second + Fetch64(s + 48), 42) * k1;
243 x ^= w.second;
244 y += v.first + Fetch64(s + 40);
245 z = Rotate(z + w.first, 33) * k1;
246 v = WeakHashLen32WithSeeds(s, v.second * k1, x + w.first);
247 w = WeakHashLen32WithSeeds(s + 32, z + w.second, y + Fetch64(s + 16));
248 temp = z;
249 z = x;
250 x = temp;
251 s += 64;
252 len -= 64;
253 } while (len != 0);
254 return HashLen16(HashLen16(v.first, w.first) + ShiftMix(y) * k1 + z,
255 HashLen16(v.second, w.second) + x);
256}

267// A subroutine for CityHash128(). Returns a decent 128-bit hash for strings
268// of any length representable in signed long. Based on City and Murmur.
269static uint128 CityMurmur(const char *s, size_t len, uint128 seed) {
270 uint64 a = Uint128Low64(seed);
271 uint64 b = Uint128High64(seed);
272 uint64 c = 0;
273 uint64 d = 0;
274 signed long l = len - 16;
275 if (l <= 0) { // len <= 16
276 a = ShiftMix(a * k1) * k1;
277 c = b * k1 + HashLen0to16(s, len);
278 d = ShiftMix(a + (len >= 8 ? Fetch64(s) : c));
279 } else { // len > 16
280 c = HashLen16(Fetch64(s + len - 8) + k1, a);
281 d = HashLen16(b + len, c + Fetch64(s + len - 16));
282 a += d;
283 do {
284 a ^= ShiftMix(Fetch64(s) * k1) * k1;
285 a *= k1;
286 b ^= a;
287 c ^= ShiftMix(Fetch64(s + 8) * k1) * k1;
288 c *= k1;
289 d ^= c;
290 s += 16;
291 l -= 16;
292 } while (l > 0);
293 }
294 a = HashLen16(a, c);
295 b = HashLen16(d, b);
296
297 uint128 result;
298 result.first = (uint64)(a ^ b);
299 result.second = (uint64)(HashLen16(b, a));
300 return result;
301}

303uint128 CityHash128WithSeed(const char *s, size_t len, uint128 seed) {
304 if (len < 128) {
305 return CityMurmur(s, len, seed);
306 }
307
308 // We expect len >= 128 to be the common case. Keep 56 bytes of state:
309 // v, w, x, y, and z.
310 uint128 v, w;
311 uint64 x = Uint128Low64(seed);
312 uint64 y = Uint128High64(seed);
313 uint64 z = len * k1;
314 uint64 temp;
315 v.first = Rotate(y ^ k1, 49) * k1 + Fetch64(s);
316 v.second = Rotate(v.first, 42) * k1 + Fetch64(s + 8);
317 w.first = Rotate(y + z, 35) * k1 + x;
318 w.second = Rotate(x + Fetch64(s + 88), 53) * k1;
319
320 // This is the same inner loop as CityHash64(), manually unrolled.
321 do {
322 x = Rotate(x + y + v.first + Fetch64(s + 8), 37) * k1;
323 y = Rotate(y + v.second + Fetch64(s + 48), 42) * k1;
324 x ^= w.second;
325 y += v.first + Fetch64(s + 40);
326 z = Rotate(z + w.first, 33) * k1;
327 v = WeakHashLen32WithSeeds(s, v.second * k1, x + w.first);
328 w = WeakHashLen32WithSeeds(s + 32, z + w.second, y + Fetch64(s + 16));
329 temp = z;
330 z = x;
331 x = temp;
332 s += 64;
333 x = Rotate(x + y + v.first + Fetch64(s + 8), 37) * k1;
334 y = Rotate(y + v.second + Fetch64(s + 48), 42) * k1;
335 x ^= w.second;
336 y += v.first + Fetch64(s + 40);
337 z = Rotate(z + w.first, 33) * k1;
338 v = WeakHashLen32WithSeeds(s, v.second * k1, x + w.first);
339 w = WeakHashLen32WithSeeds(s + 32, z + w.second, y + Fetch64(s + 16));
340 temp = z;
341 z = x;
342 x = temp;
343 s += 64;
344 len -= 128;
345 } while (LIKELY(len >= 128));
346 x += Rotate(v.first + z, 49) * k0;
347 z += Rotate(w.first, 37) * k0;
348 // If 0 < len < 128, hash up to 4 chunks of 32 bytes each from the end of s.
349 size_t tail_done;
350 for (tail_done = 0; tail_done < len;) {
351 tail_done += 32;
352 y = Rotate(x + y, 42) * k0 + v.second;
353 w.first += Fetch64(s + len - tail_done + 16);
354 x = x * k0 + w.first;
355 z += w.second + Fetch64(s + len - tail_done);
356 w.second += v.first;
357 v = WeakHashLen32WithSeeds(s + len - tail_done, v.first + z, v.second);
358 }
359 // At this point our 56 bytes of state should contain more than
360 // enough information for a strong 128-bit hash. We use two
361 // different 56-byte-to-8-byte hashes to get a 16-byte final result.
362 x = HashLen16(x, v.first);
363 y = HashLen16(y + z, w.first);
364
365 uint128 result;
366 result.first = (uint64)(HashLen16(x + v.second, w.second) + y);
367 result.second = (uint64)HashLen16(x + w.second, y + v.second);
368 return result;
369}

371uint128 CityHash128(const char *s, size_t len) {
372 uint128 r;
373 if (len >= 16) {
374 r.first = (uint64)(Fetch64(s) ^ k3);
375 r.second = (uint64)(Fetch64(s + 8));
376
377 return CityHash128WithSeed(s + 16, len - 16, r);
378
379 } else if (len >= 8) {
380 r.first = (uint64)(Fetch64(s) ^ (len * k0));
381 r.second = (uint64)(Fetch64(s + len - 8) ^ k1);
382
383 return CityHash128WithSeed(NULL, 0, r);
384 } else {
385 r.first = (uint64)k0;
386 r.second = (uint64)k1;
387 return CityHash128WithSeed(s, len, r);
388 }
389}

395// Requires len >= 240.
396static void CityHashCrc256Long(const char *s, size_t len, uint32 seed,
397 uint64 *result) {
398 uint64 a = Fetch64(s + 56) + k0;
399 uint64 b = Fetch64(s + 96) + k0;
400 uint64 c = result[0] = HashLen16(b, len);
401 uint64 d = result[1] = Fetch64(s + 120) * k0 + len;
402 uint64 e = Fetch64(s + 184) + seed;
403 uint64 f = seed;
404 uint64 g = 0;
405 uint64 h = 0;
406 uint64 i = 0;
407 uint64 j = 0;
408 uint64 t = c + d;
409
410 // 240 bytes of input per iter.
411 size_t iters = len / 240;
412 len -= iters * 240;
413 do {
414#define CHUNK(multiplier, z) \
415 { \
416 uint64 old_a = a; \
417 a = Rotate(b, 41 ^ z) * multiplier + Fetch64(s); \
418 b = Rotate(c, 27 ^ z) * multiplier + Fetch64(s + 8); \
419 c = Rotate(d, 41 ^ z) * multiplier + Fetch64(s + 16); \
420 d = Rotate(e, 33 ^ z) * multiplier + Fetch64(s + 24); \
421 e = Rotate(t, 25 ^ z) * multiplier + Fetch64(s + 32); \
422 t = old_a; \
423 } \
424 f = _mm_crc32_u64(f, a); \
425 g = _mm_crc32_u64(g, b); \
426 h = _mm_crc32_u64(h, c); \
427 i = _mm_crc32_u64(i, d); \
428 j = _mm_crc32_u64(j, e); \
429 s += 40
430
431 CHUNK(1, 1);
432 CHUNK(k0, 0);
433 CHUNK(1, 1);
434 CHUNK(k0, 0);
435 CHUNK(1, 1);
436 CHUNK(k0, 0);
437 } while (--iters > 0);
438
439 while (len >= 40) {
440 CHUNK(k0, 0);
441 len -= 40;
442 }
443 if (len > 0) {
444 s = s + len - 40;
445 CHUNK(k0, 0);
446 }
447 j += i << 32;
448 a = HashLen16(a, j);
449 h += g << 32;
450 b += h;
451 c = HashLen16(c, f) + i;
452 d = HashLen16(d, e + result[0]);
453 j += e;
454 i += HashLen16(h, t);
455 e = HashLen16(a, d) + j;
456 f = HashLen16(b, c) + a;
457 g = HashLen16(j, i) + c;
458 result[0] = e + f + g + h;
459 a = ShiftMix((a + g) * k0) * k0 + b;
460 result[1] += a + result[0];
461 a = ShiftMix(a * k0) * k0 + c;
462 result[2] = a + result[1];
463 a = ShiftMix((a + e) * k0) * k0;
464 result[3] = a + result[2];
465}
