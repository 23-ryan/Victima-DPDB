26CoreModelBoomV1::CoreModelBoomV1()
27{
28 // https://github.com/ucb-bar/riscv-boom/blob/master/src/main/scala/exu/execute.scala
29 int dfmaLatency = 4;
30 int imulLatency = 3;
31 for (unsigned int i = 0 ; i < rv_op_last ; i++)
32 {
33 instructionLatencies[i] = 1;
34 if (instrlist[i].has_fpu && instrlist[i].has_alu) {
35 instructionLatencies[i] = dfmaLatency;
36 }
37 else if (instrlist[i].has_alu && instrlist[i].has_mul) {
38 instructionLatencies[i] = imulLatency;
39 }
40 else if (instrlist[i].has_alu) {
41 instructionLatencies[i] = 1;
42 }
43 else {
44 instructionLatencies[i] = 1;
45 }
46 }
47

49}
