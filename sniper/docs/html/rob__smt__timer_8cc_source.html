110void RobSmtTimer::computeCurrentWindowsize()
111{
112 if (windowRepartition)
113 {
114 uint64_t num_threads_awake = 0;
115 for(smtthread_id_t robthread_id = 0; robthread_id < m_threads.size(); ++robthread_id)
116 if (m_threads[robthread_id]->running)
117 ++num_threads_awake;
118 if (num_threads_awake > 0)

123 else

295void RobSmtTimer::setDependencies(smtthread_id_t thread_id, RobEntry *entry)
296{
297 RobThread *thread = m_rob_threads[thread_id];
298
299 // Add = calculate dependencies, add yourself to list of depenants
300 // If no dependants in window: set ready = now()
301 uint64_t lowestValidSequenceNumber = thread->rob.size() > 0 ? thread->rob.front().uop->getSequenceNumber() : entry->uop->getSequenceNumber();
302
303 if (entry->uop->getMicroOp()->isStore())
304 setStoreAddressProducers(thread_id, entry, lowestValidSequenceNumber);
305
306 thread->registerDependencies->setDependencies(*entry->uop, lowestValidSequenceNumber);
307 thread->memoryDependencies->setDependencies(*entry->uop, lowestValidSequenceNumber);
308
309 if (m_store_to_load_forwarding && entry->uop->getMicroOp()->isLoad())
310 {
311 for(unsigned int i = 0; i < entry->uop->getDependenciesLength(); ++i)
312 {
313 RobEntry *prodEntry = this->findEntryBySequenceNumber(thread_id, entry->uop->getDependency(i));
314 // If we depend on a store
315 if (prodEntry->uop->getMicroOp()->isStore())
316 {
317 // Remove dependency on the store (which won't execute until it reaches the front of the ROB)
318 entry->uop->removeDependency(entry->uop->getDependency(i));
319
320 // Add dependencies to the producers of the value being stored instead
321 // Remark: one of these may be producing the store address, but because the store has to be
322 // disambiguated, it's correct to have the load depend on the address producers as well.
323 for(unsigned int j = 0; j < prodEntry->uop->getDependenciesLength(); ++j)
324 entry->uop->addDependency(prodEntry->uop->getDependency(j));
325
326 break;
327 }
328 }
329 }
330
331 // Add ourselves to the dependants list of the uops we depend on
332 uint64_t minProducerDistance = UINT64_MAX;
333 thread->m_totalConsumers += 1 ;
334 uint64_t deps_to_remove[8], num_dtr = 0;
335 for(unsigned int i = 0; i < entry->uop->getDependenciesLength(); ++i)
336 {
337 UInt64 dependencySequenceNumber = entry->uop->getDependency(i);
338 if (dependencySequenceNumber < lowestValidSequenceNumber)
339 {
340 // dependency is already done (this will be an intra-instruction dependency, register/memoryDependencies won't return anything < lowestValidSequenceNumber)
341 deps_to_remove[num_dtr++] = entry->uop->getDependency(i);
342 }
343 else
344 {
345 RobEntry *prodEntry = this->findEntryBySequenceNumber(thread_id, entry->uop->getDependency(i));
346 minProducerDistance = std::min( minProducerDistance, entry->uop->getSequenceNumber() - prodEntry->uop->getSequenceNumber() );
347 if (prodEntry->done != SubsecondTime::MaxTime())
348 {
349 // If producer is already done (but hasn't reached writeback stage), remove it from our dependency list
350 deps_to_remove[num_dtr++] = entry->uop->getDependency(i);
351 entry->readyMax = std::max(entry->readyMax, prodEntry->done);
352 }
353 else
354 {
355 prodEntry->addDependant(entry);
356 }
357 }
358 }
359
360 #ifdef DEBUG_PERCYCLE
361 // Make sure we are in the dependant list of all of our address producers
362 for(unsigned int i = 0; i < entry->getNumAddressProducers(); ++i)
363 {
364 if (thread->rob.size() && entry->getAddressProducer(i) >= thread->rob[0].uop->getSequenceNumber())
365 {
366 RobEntry *prodEntry = this->findEntryBySequenceNumber(thread_id, entry->getAddressProducer(i));
367 bool found = false;
368 for(unsigned int j = 0; j < prodEntry->getNumDependants(); ++j)
369 if (prodEntry->getDependant(j) == entry)
370 {
371 found = true;
372 break;
373 }
374 LOG_ASSERT_ERROR(found == true, "Store %ld depends on %ld for address production, but is not in its dependants list",
375 entry->uop->getSequenceNumber(), prodEntry->uop->getSequenceNumber());
376 }
377 }
378 #endif
379
380 if (minProducerDistance != UINT64_MAX)
381 {
382 thread->m_totalProducerInsDistance += minProducerDistance;
383 // KENZO: not sure why the distance can be larger than the windowSize, but it happens...
384 if (minProducerDistance >= thread->m_producerInsDistance.size())
385 minProducerDistance = thread->m_producerInsDistance.size()-1;
386 thread->m_producerInsDistance[ minProducerDistance ]++ ;
387 }
388 else
389 {
390 // Not depending on any instruction in the rob
391 thread->m_producerInsDistance[ 0 ] += 1 ;
392 }
393
394 // If there are any dependencies to be removed, do this after iterating over them (don't mess with the list we're reading)
395 LOG_ASSERT_ERROR(num_dtr < sizeof(deps_to_remove)/sizeof(8), "Have to remove more dependencies than I expected");
396 for(uint64_t i = 0; i < num_dtr; ++i)
397 entry->uop->removeDependency(deps_to_remove[i]);
398 if (entry->uop->getDependenciesLength() == 0)
399 {
400 // We have no dependencies in the ROB: mark ourselves as ready
401 entry->ready = entry->readyMax;
402 }
403}

405void RobSmtTimer::setStoreAddressProducers(smtthread_id_t thread_id, RobEntry *entry, uint64_t lowestValidSequenceNumber)
406{
407 RobThread *thread = m_rob_threads[thread_id];
408
409 for(unsigned int i = 0; i < entry->uop->getMicroOp()->getAddressRegistersLength(); ++i)
410 {
411 dl::Decoder::decoder_reg reg = entry->uop->getMicroOp()->getAddressRegister(i);
412 uint64_t addressProducer = thread->registerDependencies->peekProducer(reg, lowestValidSequenceNumber);
413 if (addressProducer != INVALID_SEQNR)
414 {
415 RobEntry *prodEntry = findEntryBySequenceNumber(thread_id, addressProducer);
416 if (prodEntry->done != SubsecondTime::MaxTime())
417 entry->addressReadyMax = std::max(entry->addressReadyMax, prodEntry->done);
418 else
419 entry->addAddressProducer(addressProducer);
420 }
421 }
422 if (entry->getNumAddressProducers() == 0)
423 entry->addressReady = entry->addressReadyMax;
424}

426RobSmtTimer::RobEntry *RobSmtTimer::findEntryBySequenceNumber(smtthread_id_t thread_id, UInt64 sequenceNumber)
427{
428 Rob &rob = m_rob_threads[thread_id]->rob;
429 // Assumption: MicroOps in the ROB are numbered sequentially, none of them are removed halfway
430 UInt64 first = rob[0].uop->getSequenceNumber();
431 UInt64 position = sequenceNumber - first;
432 LOG_ASSERT_ERROR(position < rob.size(), "Sequence number %ld outside of ROB", sequenceNumber);
433 RobEntry *entry = &rob[position];
434 LOG_ASSERT_ERROR(entry->uop->getSequenceNumber() == sequenceNumber, "Sequence number %ld unexpectedly not at ROB position %ld", sequenceNumber, position);
435 return entry;
436}

454void RobSmtTimer::pushInstructions(smtthread_id_t thread_id, const std::vector<DynamicMicroOp*>& insts)
455{
456 RobThread *thread = m_rob_threads[thread_id];
457
458 // Collect new microops
459
460 for (std::vector<DynamicMicroOp*>::const_iterator it = insts.begin(); it != insts.end(); it++ )
461 {
462 if ((*it)->isSquashed())
463 {
464 delete *it;
465 continue;
466 }
467
468 RobEntry *entry = &thread->rob.next();
469 entry->init(*it, thread->nextSequenceNumber++);
470
471 #ifdef DEBUG_PERCYCLE
472 std::cout<<"** ["<<int(thread_id)<<"] simulate: "<<entry->uop->getMicroOp()->toShortString(true)<<std::endl;
473 #endif
474
475 thread->m_uop_type_count[(*it)->getMicroOp()->getSubtype()]++;
476 thread->m_uops_total++;
477 if ((*it)->getMicroOp()->isX87()) thread->m_uops_x87++;
478 if ((*it)->getMicroOp()->isPause()) thread->m_uops_pause++;
479
480 if (thread->m_uops_total > 10000 && thread->m_uops_x87 > thread->m_uops_total / 20)
481 LOG_PRINT_WARNING_ONCE("Significant fraction of x87 instructions encountered, accuracy will be low. Compile without -mno-sse2 -mno-sse3 to avoid.");
482 }

485boost::tuple<uint64_t,SubsecondTime> RobSmtTimer::returnLatency(smtthread_id_t thread_id)
486{
487 RobThread *thread = m_rob_threads[thread_id];
488
489 // Return latency from previous execute() calls
490
491 SubsecondTime returnLat = SubsecondTime::Zero();
492 if (now > thread->now)
493 {
494 returnLat = now - thread->now;
495 thread->now = now;
496 }
497
498 #ifdef DEBUG_PERCYCLE
499 if (returnLat > SubsecondTime::Zero())
500 std::cout<<"** ["<<int(thread_id)<<"] return "<<returnLat<<" latency cycles"<<std::endl<<std::endl<<std::endl;
501 #endif
502
503 // Calculate number of instructions executed for this thread since last call
504 uint64_t instrs = thread->instrs - thread->instrs_returned;
505 thread->instrs_returned = thread->instrs;
506
507 return boost::tuple<uint64_t,SubsecondTime>(instrs, returnLat);
508}

543bool RobSmtTimer::tryDispatch(smtthread_id_t thread_num, SubsecondTime &next_event)
544{
545 // Dispatch up to dispatchWidth instructions, all from a single thread
546 RobThread *thread = m_rob_threads[thread_num];
547 uint32_t instrs_dispatched = 0, uops_dispatched = 0;
548
549 // Unless we set a new stall condition later on, assume no stall occured
550 thread->m_cpiCurrentFrontEndStall = NULL;
551
552 while(thread->m_num_in_rob < currentWindowSize)
553 {
554 LOG_ASSERT_ERROR(thread->m_num_in_rob < thread->rob.size(), "Expected sufficient uops for dispatching in pre-ROB buffer, but didn't find them");
555
556 RobEntry *entry = &thread->rob.at(thread->m_num_in_rob);
557 DynamicMicroOp &uop = *entry->uop;
558
559 // Dispatch up to 4 instructions
560 if (uops_dispatched == dispatchWidth)
561 break;
562
563 // This is actually in the decode stage, there's a buffer between decode and dispatch
564 // so we shouldn't do this here.
567 //if (instrs_dispatched > 0 && !uop.isLast())
568 // break;
569
570 bool iCacheMiss = (uop.getICacheHitWhere() != HitWhere::L1I);
571 if (iCacheMiss)
572 {
573 if (thread->in_icache_miss)
574 {
575 // We just took the latency for this instruction, now dispatch it
576 #ifdef DEBUG_PERCYCLE
577 std::cout<<"-- icache return"<<std::endl;
578 #endif
579 thread->in_icache_miss = false;
580 }
581 else
582 {
583 #ifdef DEBUG_PERCYCLE
584 std::cout<<"-- icache miss("<<uop.getICacheLatency()<<")"<<std::endl;
585 #endif
586 thread->frontend_stalled_until = now + uop.getICacheLatency();
587 thread->in_icache_miss = true;
588 // Don't dispatch this instruction yet
589 thread->m_cpiCurrentFrontEndStall = &thread->m_cpiInstructionCache[uop.getICacheHitWhere()];
590 break;
591 }
592 }
593
594 if (m_rs_entries_used == rsEntries)
595 {
596 thread->m_cpiCurrentFrontEndStall = &thread->m_cpiRSFull;
597 break;
598 }
599
600 setDependencies(thread_num, entry);
601
602 entry->dispatched = now;
603 ++thread->m_num_in_rob;
604 ++m_rs_entries_used;
605
606 uops_dispatched++;
607 if (uop.isLast())
608 instrs_dispatched++;
609
610 // If uop is already ready, we may need to issue it in the following cycle
611 entry->ready = std::max(entry->ready, (now + 1ul).getElapsedTime());
612 next_event = std::min(next_event, entry->ready);
613
614 thread->next_event = std::min(thread->next_event, next_event);
615
616 #ifdef DEBUG_PERCYCLE
617 std::cout<<"["<<int(thread_num)<<"] DISPATCH "<<uop.getMicroOp()->toShortString()<<std::endl;
618 #endif
619
620 #ifdef ASSERT_SKIP
621 LOG_ASSERT_ERROR(will_skip == false, "Cycle would have been skipped but stuff happened");
622 #endif
623
624 // Mispredicted branch
625 if (uop.getMicroOp()->isBranch() && uop.isBranchMispredicted())
626 {
627 thread->frontend_stalled_until = SubsecondTime::MaxTime();
628 #ifdef DEBUG_PERCYCLE
629 std::cout<<"-- branch mispredict"<<std::endl;
630 #endif
631 thread->m_cpiCurrentFrontEndStall = &thread->m_cpiBranchPredictor;
632 break;
633 }
634 }
635
636 if (thread->m_num_in_rob < currentWindowSize)
637 next_event = std::min(thread->frontend_stalled_until, next_event);
638
639 return uops_dispatched > 0;
640}

642SubsecondTime* RobSmtTimer::findCpiComponent(smtthread_id_t thread_num)
643{
644 RobThread *thread = m_rob_threads[thread_num];
645
646 // Determine the CPI component corresponding to the first non-committed instruction
647 for(uint64_t i = 0; i < thread->m_num_in_rob; ++i)
648 {
649 RobEntry *entry = &thread->rob.at(i);
650 DynamicMicroOp *uop = entry->uop;
651 // Skip over completed instructions
652 if (entry->done < now)
653 continue;
654 // Nothing currently executing, CPI component will be Base or Branch/Icache as determined by front-end
655 if (entry->issued >= now)
656 return NULL;
657 // This is the first instruction in the ROB which is still executing
658 // Assume everyone is blocked on this one
659 // Assign 100% of this cycle to this guy's CPI component
660 if (uop->getMicroOp()->isSerializing() || uop->getMicroOp()->isMemBarrier())
661 return &thread->m_cpiSerialization;
662 else if (uop->getMicroOp()->isLoad() || uop->getMicroOp()->isStore())
663 return &thread->m_cpiDataCache[uop->getDCacheHitWhere()];
664 else
665 return NULL;
666 }
667 // No instruction is currently executing
668 return NULL;
669}

671SubsecondTime RobSmtTimer::doDispatch()
672{
673 SubsecondTime next_event = SubsecondTime::MaxTime();
674 smtthread_id_t thread_first = dispatch_thread, thread_dispatched_from = dispatch_thread;
675 bool hasDispatched = false;
676
677 // Dispatch round-robin from threads that are not blocked due to I-cache miss, full ROB, idle, etc.
678
679 smtthread_id_t thread_num = thread_first;
680 do
681 {
682 SmtThread *smt_thread = m_threads[thread_num];
683 RobThread *thread = m_rob_threads[thread_num];
684 SubsecondTime *cpiComponent = NULL;
685 SubsecondTime *cpiRobHead = findCpiComponent(thread_num);
686
687 if (thread->frontend_stalled_until > now)
688 {
689 // front-end should not be stalled due to I-cache miss or branch misprediction
690 assert(thread->m_cpiCurrentFrontEndStall);
691 // canDispatch returned false, but not because the ROB was full: previous stall condition
692 // (I-cache miss or BP-misprediction) still applies
693 if (cpiRobHead)
694 // Prioritize back-end stalls over front-end stalls
695 cpiComponent = cpiRobHead;
696 else
697 cpiComponent = thread->m_cpiCurrentFrontEndStall;
698 }
699 else if (!smt_thread->running || thread->now > now)
700 {
701 // thread should not be idle, nor should it live way in the future because it just woke from idle
702 cpiComponent = &thread->m_cpiIdle;
703 }
704 else if (thread->m_num_in_rob >= currentWindowSize)
705 {
706 // Could not dispatch because of a full ROB: determine which instruction we're stalling on
707 cpiComponent = cpiRobHead ? cpiRobHead : &thread->m_cpiBase;
708 }
709 else
710 {
711 // Front-end is fine, lets try to dispatch new instructions
712
713 if (hasDispatched)
714 {
715 // Could have dispatched something, but a previous thread has already dispatched
716 cpiComponent = &thread->m_cpiSMT;
717 }
718 else
719 {
720 hasDispatched = tryDispatch(thread_num, next_event);
721 if (hasDispatched)
722 thread_dispatched_from = thread_num;
723
724 if (thread->m_cpiCurrentFrontEndStall)
725 // if tryDispatch failed, it will have set m_cpiCurrentFrontEndStall
726 cpiComponent = thread->m_cpiCurrentFrontEndStall;
727 else
728 // if it did dispatch something, add to base CPI
729 cpiComponent = &thread->m_cpiBase;
730 }
731 }
732
733 LOG_ASSERT_ERROR(cpiComponent != NULL, "We expected cpiComponent to be set, but it wasn't");
734 *cpiComponent += now.getPeriod();
735
736 thread_num = (thread_num + 1) % m_threads.size();
737 }
738 while (thread_num != thread_first);
739
740 // Either someone has issued, in the following cycle we'll start with the one following the thread that did issue
741 // Or no-one issued, then thread_dispatched_from == thread_first, so we'll continue with the one following the previous first thread
742 dispatch_thread = (thread_dispatched_from + 1) % m_threads.size();
743
744 return next_event;
745}

747void RobSmtTimer::issueInstruction(smtthread_id_t thread_num, uint64_t idx, SubsecondTime &next_event)
748{
749 SmtThread *smt_thread = m_threads[thread_num];
750 RobThread *thread = m_rob_threads[thread_num];
751 RobEntry *entry = &thread->rob[idx];
752 DynamicMicroOp &uop = *entry->uop;
753
754 if ((uop.getMicroOp()->isLoad() || uop.getMicroOp()->isStore())
755 && uop.getDCacheHitWhere() == HitWhere::UNKNOWN)
756 {
757 MemoryResult res = smt_thread->core->accessMemory(
758 Core::NONE,
759 uop.getMicroOp()->isLoad() ? Core::READ : Core::WRITE,
760 uop.getAddress().address,
761 NULL,
762 uop.getMicroOp()->getMemoryAccessSize(),
763 Core::MEM_MODELED_RETURN,
764 uop.getMicroOp()->getInstruction() ? uop.getMicroOp()->getInstruction()->getAddress() : static_cast<uint64_t>(NULL),
765 now.getElapsedTime()
766 );
767 uint64_t latency = SubsecondTime::divideRounded(res.latency, now.getPeriod());
768
769 uop.setExecLatency(uop.getExecLatency() + latency); // execlatency already contains bypass latency
770 uop.setDCacheHitWhere(res.hit_where);
771 }
772
773 if (uop.getMicroOp()->isLoad())
774 load_queue.getCompletionTime(now, uop.getExecLatency() * now.getPeriod(), uop.getAddress().address);
775 else if (uop.getMicroOp()->isStore())
776 store_queue.getCompletionTime(now, uop.getExecLatency() * now.getPeriod(), uop.getAddress().address);
777
778 ComponentTime cycle_depend = now + uop.getExecLatency(); // When result is available for dependent instructions
779 SubsecondTime cycle_done = cycle_depend + 1ul; // When the instruction can be committed
780
781 if (uop.getMicroOp()->isLoad())
782 {
783 thread->m_loads_count++;
784 thread->m_loads_latency += uop.getExecLatency() * now.getPeriod();
785 }
786 else if (uop.getMicroOp()->isStore())
787 {
788 thread->m_stores_count++;
789 thread->m_stores_latency += uop.getExecLatency() * now.getPeriod();
790 }
791
792 if (uop.getMicroOp()->isStore())
793 {
794 last_store_done = std::max(last_store_done, cycle_done);
795 cycle_depend = now + 1ul; // For stores, forward the result immediately
796 // Stores can be removed from the ROB once they're issued to the memory hierarchy
797 // Dependent operations such as SFENCE and synchronization instructions need to wait until last_store_done
798 cycle_done = now + 1ul;
799
800 LOG_ASSERT_ERROR(entry->addressReady <= entry->ready, "%ld: Store address cannot be ready (%ld) later than the whole uop is (%ld)",
801 entry->uop->getSequenceNumber(), entry->addressReady.getPS(), entry->ready.getPS());
802 }
803
804 if (m_rob_contention)
805 m_rob_contention->doIssue(uop);
806
807 entry->issued = now;
808 entry->done = cycle_done;
809 next_event = std::min(next_event, entry->done);
810
811 --m_rs_entries_used;
812
813 #ifdef DEBUG_PERCYCLE
814 std::cout<<"["<<int(thread_num)<<"] ISSUE "<<entry->uop->getMicroOp()->toShortString()<<" latency="<<uop.getExecLatency()<<std::endl;
815 #endif
816
817 for(size_t idx = 0; idx < entry->getNumDependants(); ++idx)
818 {
819 RobEntry *depEntry = entry->getDependant(idx);
820 LOG_ASSERT_ERROR(depEntry->uop->getDependenciesLength()> 0, "??");
821
822 // Remove uop from dependency list and update readyMax
823 depEntry->readyMax = std::max(depEntry->readyMax, cycle_depend.getElapsedTime());
824 depEntry->uop->removeDependency(uop.getSequenceNumber());
825
826 // If all dependencies are resolved, mark the uop ready
827 if (depEntry->uop->getDependenciesLength() == 0)
828 {
829 depEntry->ready = depEntry->readyMax;
830 //std::cout<<" ready @ "<<depEntry->ready<<std::endl;
831 }
832
833 // For stores, check if their address has been produced
834 if (depEntry->uop->getMicroOp()->isStore() && depEntry->addressReady == SubsecondTime::MaxTime())
835 {
836 bool ready = true;
837 for(unsigned int i = 0; i < depEntry->getNumAddressProducers(); ++i)
838 {
839 uint64_t addressProducer = depEntry->getAddressProducer(i);
840 RobEntry *prodEntry = addressProducer >= thread->rob.front().uop->getSequenceNumber()
841 ? this->findEntryBySequenceNumber(thread_num, addressProducer) : NULL;
842
843 if (prodEntry == entry)
844 {
845 // The instruction we just executed is producing an address. Update the store's addressReadyMax
846 depEntry->addressReadyMax = std::max(depEntry->addressReadyMax, cycle_depend.getElapsedTime());
847 }
848
849 if (prodEntry && prodEntry->done == SubsecondTime::MaxTime())
850 {
851 // An address producer has not yet been issued: address remains not ready
852 ready = false;
853 }
854 }
855
856 if (ready)
857 {
858 // We did not find any address producing instructions that have not yet been issued.
859 // Store address will be ready at addressReadyMax
860 depEntry->addressReady = depEntry->addressReadyMax;
861 }
862 }
863 }
864
865 // After issuing a mispredicted branch: allow the ROB to refill after flushing the pipeline
866 if (uop.getMicroOp()->isBranch() && uop.isBranchMispredicted())
867 {
868 thread->frontend_stalled_until = now + (misprediction_penalty - 2); // The frontend needs to start 2 cycles earlier to get a total penalty of <misprediction_penalty>
869 #ifdef DEBUG_PERCYCLE
870 std::cout<<"-- branch resolve"<<std::endl;
871 #endif
872 }
873}

875bool RobSmtTimer::tryIssue(smtthread_id_t thread_num, SubsecondTime &next_event, bool would_have_skipped)
876{
877 uint64_t num_issued = 0;
878 bool head_of_queue = true, no_more_load = false, no_more_store = false, have_unresolved_store = false;
879 RobThread *thread = m_rob_threads[thread_num];
880 SubsecondTime thread_next_event = SubsecondTime::MaxTime();
881
882 for(uint64_t i = 0; i < thread->m_num_in_rob; ++i)
883 {
884 RobEntry *entry = &thread->rob.at(i);
885 DynamicMicroOp *uop = entry->uop;
886
887
888 if (entry->done != SubsecondTime::MaxTime())
889 {
890 thread_next_event = std::min(thread_next_event, entry->done);
891 continue; // already done
892 }
893
894 thread_next_event = std::min(thread_next_event, entry->ready);
895
896
897 // See if we can issue this instruction
898
899 bool canIssue = false;
900
901 if (entry->ready > now)
902 canIssue = false; // blocked by dependency
903
904 else if ((no_more_load && uop->getMicroOp()->isLoad()) || (no_more_store && uop->getMicroOp()->isStore()))
905 canIssue = false; // blocked by mfence
906
907 else if (uop->getMicroOp()->isSerializing())
908 {
909 if (head_of_queue && last_store_done <= now)
910 canIssue = true;
911 else
912 break;
913 }
914
915 else if (uop->getMicroOp()->isMemBarrier())
916 {
917 if (head_of_queue && last_store_done <= now)
918 canIssue = true;
919 else
920 // Don't issue any memory operations following a memory barrier
921 no_more_load = no_more_store = true;
922 // FIXME: L/SFENCE
923 }
924
925 else if (!m_rob_contention && num_issued == dispatchWidth)
926 canIssue = false; // no issue contention: issue width == dispatch width
927
928 else if (uop->getMicroOp()->isLoad() && !load_queue.hasFreeSlot(now))
929 canIssue = false; // load queue full
930
931 else if (uop->getMicroOp()->isLoad() && m_no_address_disambiguation && have_unresolved_store)
932 canIssue = false; // preceding store with unknown address
933
934 else if (uop->getMicroOp()->isStore() && (!head_of_queue || !store_queue.hasFreeSlot(now)))
935 canIssue = false; // store queue full
936
937 else
938 canIssue = true; // issue!
939
940
941 // canIssue already marks issue ports as in use, so do this one last
942 if (canIssue && m_rob_contention && ! m_rob_contention->tryIssue(*uop))
943 canIssue = false; // blocked by structural hazard
944
945
946 if (canIssue)
947 {
948 num_issued++;
949 issueInstruction(thread_num, i, thread_next_event);
950
951
952 // Calculate memory-level parallelism (MLP) for long-latency loads (but ignore overlapped misses)
953 if (uop->getMicroOp()->isLoad() && uop->isLongLatencyLoad() && uop->getDCacheHitWhere() != HitWhere::L1_OWN)
954 {
955 if (thread->m_lastAccountedMemoryCycle < now) thread->m_lastAccountedMemoryCycle = now;
956
957 SubsecondTime done = std::max( now.getElapsedTime(), entry->done );
958 // Ins will be outstanding for until it is done. By account beforehand I don't need to
959 // worry about fast-forwarding simulations
960 thread->m_outstandingLongLatencyInsns += (done - now);
961
962 // Only account for the cycles that have not yet been accounted for by other long
963 // latency misses (don't account cycles twice).
964 if ( done > thread->m_lastAccountedMemoryCycle )
965 {
966 thread->m_outstandingLongLatencyCycles += done - thread->m_lastAccountedMemoryCycle;
967 thread->m_lastAccountedMemoryCycle = done;
968 }
969
970 #ifdef ASSERT_SKIP
971 LOG_ASSERT_ERROR( thread->m_outstandingLongLatencyInsns >= thread->m_outstandingLongLatencyCycles, "MLP calculation is wrong: MLP cannot be < 1!" );
972 #endif
973 }
974
975
976 #ifdef ASSERT_SKIP
977 LOG_ASSERT_ERROR(will_skip == false, "Cycle would have been skipped but stuff happened");
978 LOG_ASSERT_ERROR(would_have_skipped == false, "Cycle would have been skipped but stuff happened");
979 #endif
980 }
981 else
982 {
983 head_of_queue = false; // Subsequent instructions are not at the head of the ROB
984
985 if (uop->getMicroOp()->isStore() && entry->addressReady > now)
986 have_unresolved_store = true;
987
988 if (inorder)
989 // In-order: only issue from head of the ROB
990 break;
991 }
992
993
994 if (m_rob_contention)
995 {
996 if (m_rob_contention->noMore())
997 break;
998 }
999 else
1000 {
1001 if (num_issued == dispatchWidth)
1002 break;
1003 }
1004 }
1005
1006 thread->next_event = thread_next_event;
1007 next_event = std::min(next_event, thread_next_event);
1008
1009 return num_issued > 0;
1010}

1012SubsecondTime RobSmtTimer::doIssue()
1013{
1014 SubsecondTime next_event = SubsecondTime::MaxTime();
1015 smtthread_id_t thread_first = issue_thread;
1016
1017 if (m_rob_contention)
1018 m_rob_contention->initCycle(now);
1019
1020 // Issue round-robin, from a single-thread only (interleaved multithreading, SMT would need to check all ROBs)
1021
1022 smtthread_id_t thread_num = thread_first;
1023 do
1024 {
1025 RobThread *thread = m_rob_threads[thread_num];
1026
1027 if (thread->next_event > now)
1028 {
1029 #ifdef ASSERT_SKIP
1030 tryIssue(thread_num, next_event, true);
1031 #endif
1032 }
1033 else if (m_rob_threads[thread_num]->m_num_in_rob > 0)
1034 {
1035 bool hasIssued = tryIssue(thread_num, next_event);
1036 if (hasIssued && !simultaneousIssue)
1037 break;
1038 }
1039 thread_num = (thread_num + 1) % m_threads.size();
1040 } while (thread_num != thread_first);
1041
1042 // Either someone has issued, in the following cycle we'll start with the one following the thread that did issue
1043 // Or no-one issued, then thread_num == thread_first, so we'll continue with the one following the previous first thread
1044 issue_thread = (thread_num + 1) % m_threads.size();
1045
1046 return next_event;
1047}

1049SubsecondTime RobSmtTimer::doCommit()
1050{
1051 // Commit all done instructions from all threads
1052 for(smtthread_id_t thread_num = 0; thread_num < m_threads.size(); ++thread_num)
1053 {
1054 uint64_t num_committed = 0;
1055 RobThread *thread = m_rob_threads[thread_num];
1056
1057 while(thread->rob.size() && (thread->rob.front().done <= now))
1058 {
1059 RobEntry *entry = &thread->rob.front();
1060
1061 #ifdef DEBUG_PERCYCLE
1062 std::cout<<"["<<int(thread_num)<<"] COMMIT "<<entry->uop->getMicroOp()->toShortString()<<std::endl;
1063 #endif
1064
1065 InstructionTracer::uop_times_t times = {
1066 entry->dispatched,
1067 entry->issued,
1068 entry->done,
1069 now
1070 };
1071 thread->core->getPerformanceModel()->traceInstruction(entry->uop, &times);
1072
1073 if (entry->uop->isLast())
1074 thread->instrs++;
1075
1076 entry->free();
1077 thread->rob.pop();
1078 thread->m_num_in_rob--;
1079
1080 #ifdef ASSERT_SKIP
1081 LOG_ASSERT_ERROR(will_skip == false, "Cycle would have been skipped but stuff happened");
1082 #endif
1083
1084 ++num_committed;
1085 if (num_committed == commitWidth)
1086 break;
1087 }
1088 }
1089
1090 if (m_rob_threads[0]->rob.size()) // NOTE: for SMT, skip is ignored anyway
1091 return m_rob_threads[0]->rob.front().done;
1092 else
1093 return SubsecondTime::MaxTime();
1094}

1099 RobThread *thread = m_rob_threads[thread_num];
1100

1104 || thread->m_num_in_rob >= currentWindowSize // ROB is full
1105 || !smt_thread->running // thread is not running
1106 )

1108 return false;

1137SubsecondTime RobSmtTimer::executeCycle()
1138{
1139 SubsecondTime latency = SubsecondTime::Zero();
1140
1141 #ifdef DEBUG_PERCYCLE
1142 std::cout<<std::endl<<"###################################################################################"<<std::endl<<std::endl;
1143 std::cout<<"Running cycle "<<now<<std::endl;
1144 #endif
1145
1146 // Model dispatch, issue and commit stages
1147 // Decode stage is not modeled, assumes the decoders can keep up with (up to) dispatchWidth uops per cycle
1148
1149 SubsecondTime next_dispatch = doDispatch();
1150 SubsecondTime next_issue = doIssue();
1151 SubsecondTime next_commit = doCommit();
1152
1153
1154 #ifdef DEBUG_PERCYCLE
1155 #ifdef ASSERT_SKIP
1156 if (! will_skip)
1157 {
1158 #endif
1159 printRob();
1160 #ifdef ASSERT_SKIP
1161 }
1162 #endif
1163 #endif
1164
1165
1166 #ifdef DEBUG_PERCYCLE
1167 std::cout<<"Next event: D("<<next_dispatch<<") I("<<next_issue<<") C("<<next_commit<<")"<<std::endl;
1168 #endif
1169 SubsecondTime next_event = std::min(next_dispatch, std::min(next_issue, next_commit));
1170 SubsecondTime skip;
1171 if (m_threads.size() > 1)
1172 {
1173 // SMT: we would need to check all ROBs at all cycles. For now, don't bother skipping
1174 will_skip = false;
1175 skip = now.getPeriod();
1176 // Also, CPI stack code is concentrated at dispatch and wouldn't know about skipping
1177 }
1178 else if (next_event != SubsecondTime::MaxTime() && next_event > now + 1ul)
1179 {
1180 #ifdef DEBUG_PERCYCLE
1181 std::cout<<"++ Skip "<<(next_event - now)<<std::endl;
1182 #endif
1183 will_skip = true;
1184 skip = next_event - now;
1185 }
1186 else
1187 {
1188 will_skip = false;
1189 skip = now.getPeriod();
1190 }
1191
1192 #ifdef ASSERT_SKIP
1193 now += now.getPeriod();
1194 latency += now.getPeriod();
1195 if (will_skip)
1196 time_skipped += now.getPeriod();
1197 #else
1198 now += skip;
1199 latency += skip;
1200 if (skip > now.getPeriod())
1201 time_skipped += (skip - now.getPeriod());
1202 #endif
1203
1204 if (m_mlp_histogram)
1205 {
1206 for(smtthread_id_t thread_num = 0; thread_num < m_threads.size(); ++thread_num)
1207 countOutstandingMemop(thread_num, skip);
1208 }
1209
1210 return latency;
1211}

1213void RobSmtTimer::countOutstandingMemop(smtthread_id_t thread_num, SubsecondTime time)
1214{
1215 RobThread *thread = m_rob_threads[thread_num];
1216 UInt64 counts[HitWhere::NUM_HITWHERES] = {0}, total = 0;
1217
1218 for(unsigned int i = 0; i < thread->m_num_in_rob; ++i)
1219 {
1220 RobEntry *e = &thread->rob.at(i);
1221 if (e->done != SubsecondTime::MaxTime() && e->done > now && e->uop->getMicroOp()->isLoad())
1222 {
1223 ++counts[e->uop->getDCacheHitWhere()];
1224 ++total;
1225 }
1226 }
1227
1228 for(unsigned int h = 0; h < HitWhere::NUM_HITWHERES; ++h)
1229 if (counts[h] > 0)
1230 thread->m_outstandingLoads[h][counts[h] >= MAX_OUTSTANDING ? MAX_OUTSTANDING-1 : counts[h]] += time;
1231 if (total > 0)
1232 thread->m_outstandingLoadsAll[total >= MAX_OUTSTANDING ? MAX_OUTSTANDING-1 : total] += time;
