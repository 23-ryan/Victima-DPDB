40template<unsigned ElemSize, unsigned MaxElem = 0, typename TypeToAlloc = UnspecifiedType>
41class FSBAllocator_ElemAllocator
42{
43 typedef std::size_t Data_t;
44 static const Data_t BlockElements = 512;
45
46 static const Data_t DSize = sizeof(Data_t);
47 static const Data_t ElemSizeInDSize = (ElemSize + (DSize-1)) / DSize;
48 static const Data_t UnitSizeInDSize = ElemSizeInDSize + 1;
49 static const Data_t BlockSize = BlockElements*UnitSizeInDSize;
50
51 class MemBlock
52 {
53 Data_t* block;
54 Data_t firstFreeUnitIndex, allocatedElementsAmount, endIndex;
55
56 public:
57 MemBlock():
58 block(new Data_t[BlockSize]),
59 firstFreeUnitIndex(Data_t(-1)),
60 allocatedElementsAmount(0)
61 {}
62
63 bool isFull() const
64 {
65 return allocatedElementsAmount == BlockElements;
66 }
67
68 void clear()
69 {
70 firstFreeUnitIndex = Data_t(-1);
71 }
72
73 void* allocate(Data_t vectorIndex)
74 {
75 if(firstFreeUnitIndex == Data_t(-1))
76 {
77 if(allocatedElementsAmount == 0)
78 {
79 endIndex = 0;
80 }
81
82 Data_t* retval = block + endIndex;
83 endIndex += UnitSizeInDSize;
84 retval[ElemSizeInDSize] = vectorIndex;
85 ++allocatedElementsAmount;
86 return retval;
87 }
88 else
89 {
90 Data_t* retval = block + firstFreeUnitIndex;
91 firstFreeUnitIndex = *retval;
92 ++allocatedElementsAmount;
93 return retval;
94 }
95 }
96
97 void deallocate(Data_t* ptr)
98 {
99 *ptr = firstFreeUnitIndex;
100 firstFreeUnitIndex = ptr - block;
101
102 if(--allocatedElementsAmount == 0)
103 clear();
104 }
105 };
106
107 struct BlocksVector
108 {
109 std::vector<MemBlock> data;
110
111 BlocksVector() { data.reserve(1024); }
112
113 ~BlocksVector()
114 {
115 for(std::size_t i = 0; i < data.size(); ++i)
116 data[i].clear();
117 }
118 };
119
120 BlocksVector blocksVector;
121 std::vector<Data_t> blocksWithFree;
122
123 public:
124 void* allocate()
125 {
126 if(blocksWithFree.empty())
127 {
128 blocksWithFree.push_back(blocksVector.data.size());
129 blocksVector.data.push_back(MemBlock());
130 if (MaxElem)
131 {
132 int status;
133 char *nameoftype = abi::__cxa_demangle(typeid(TypeToAlloc).name(), 0, 0, &status);
134 LOG_ASSERT_ERROR(blocksVector.data.size() * BlockElements <= MaxElem, "Maximum number of blocks exceeded for allocator of %d-sized objects of %s", ElemSize, nameoftype);
135 free(nameoftype);
136 }
137 }
138
139 const Data_t index = blocksWithFree.back();
140 MemBlock& block = blocksVector.data[index];
141 void* retval = block.allocate(index);
142
143 if(block.isFull())
144 blocksWithFree.pop_back();
145
146 return retval;
147 }
148
149 void deallocate(void* ptr)
150 {
151 if(!ptr) return;
152
153 Data_t* unitPtr = (Data_t*)ptr;
154 const Data_t blockIndex = unitPtr[ElemSizeInDSize];
155 MemBlock& block = blocksVector.data[blockIndex];
156
157 if(block.isFull())
158 blocksWithFree.push_back(blockIndex);
159 block.deallocate(unitPtr);
160 }
161};
