38 Section & Config::getSection_unsafe(const String & path)
39 {
40 //Handle the base case
41 if(isLeaf(path))
42 {
43 if(!m_root.hasSection(path))
44 m_root.addSubsection(path);
45 return m_root.getSection_unsafe(path);
46 }
47
48 //split up the path on "/", and loop through each entry of this
49 //split to obtain the actual section
50 PathElementList path_elements;
51 Config::splitPathElements(path, path_elements);
52
53 Section * current = &m_root;
54 for(PathElementList::iterator path_element = path_elements.begin();
55 path_element != path_elements.end(); path_element++)
56 {
57 //add the section if it doesn't already exist
58 if(!current->hasSection(*path_element))
59 {
60 current->addSubsection(*path_element);
61 }
62
63 //Find the current element name as a sub section of the current section
64 current = &(current->getSection_unsafe(*path_element));
65 }
66 return *current;
67 }

150 std::pair<String,String> Config::splitPathElements(const String & path, PathElementList & path_elements)
151 {
152 //split up the path on "/", the last entry is the name of the key
153 //Everything up to the last "/" is the 'base_path' (which will specify a section)
154
155 boost::split(path_elements, path, boost::is_any_of("/"));
156
157 //Grab the appropriate pieces from the split
158 String key_name = path_elements[path_elements.size() - 1];
159 String base_path = "";
160 if(path.rfind("/") != String::npos)
161 base_path = path.substr(0, path.rfind("/"));
162
163 return PathPair(base_path,key_name);
164 }
