159void TraceManager::signalDone(TraceThread *thread, SubsecondTime time, bool aborted)
160{
161 ScopedLock sl(m_lock);
162
163 // Make sure threads don't call signalDone twice (once through endApplication,
164 // and once the regular way), as this would throw off our counts
165 if (thread->m_stopped)
166 {
167 return;
168 }
169 thread->m_stopped = true;
170
171 app_id_t app_id = thread->getThread()->getAppId();
172 m_app_info[app_id].num_threads--;
173
174 if (!aborted)
175 {
176 if (m_app_info[app_id].num_threads == 0)
177 {
178 m_app_info[app_id].num_runs++;
179 Sim()->getHooksManager()->callHooks(HookType::HOOK_APPLICATION_EXIT, (UInt64)app_id);
180 Sim()->getStatsManager()->logEvent(StatsManager::EVENT_APP_EXIT, SubsecondTime::MaxTime(), INVALID_CORE_ID, INVALID_THREAD_ID, (UInt64)app_id, 0, "");
181
182 if (m_app_info[app_id].num_runs == 1)
183 m_num_apps_nonfinish--;
184
185 if (m_stop_with_first_app)
186 {
187 // First app has ended: stop
188 stop();
189 }
190 else if (m_num_apps_nonfinish == 0)
191 {
192 // All apps have completed at least once: stop
193 stop();
194 }
195 else
196 {
197 // Stop condition not met. Restart app?
198 if (m_app_restart)
199 {
200 newThread(app_id, true /*first*/, false /*init_fifo*/, true /*spawn*/, time, INVALID_THREAD_ID);
201 }
202 }
203 }
204 }
205
206 m_num_threads_running--;
207}

209void TraceManager::endApplication(TraceThread *thread, SubsecondTime time)
210{
211 for(std::vector<TraceThread *>::iterator it = m_threads.begin(); it != m_threads.end(); ++it)
212 {
213 // Abort all threads in this application, except ourselves (we should end normally soon)
214 if ((*it)->getThread()->getAppId() == thread->getThread()->getAppId() && *it != thread)
215 {
216 // Ask thread to stop
217 (*it)->stop();
218 // Threads are often blocked on a futex in this case, so call signalDone in their place
219 signalDone(*it, time, true /* aborted */);
220 }

251void TraceManager::stop()
252{
253 // End of region-of-interest when running Sniper inside Sniper
254 SimRoiEnd();
255
256 // Signal threads to stop.
257 for(std::vector<TraceThread *>::iterator it = m_threads.begin(); it != m_threads.end(); ++it)
258 (*it)->stop();
259 // Give threads some time to end.
260 sleep(1);
261 // Some threads may be blocked (SIFT reader, syscall, etc.). Don't wait for them or we'll deadlock.

290UInt64 TraceManager::getProgressValue()
291{
292 UInt64 value = 0;
293 for(std::vector<TraceThread *>::iterator it = m_threads.begin(); it != m_threads.end(); ++it)
294 {
295 uint64_t expect = (*it)->getProgressExpect();
296 if (expect)
297 {
298 UInt64 this_value = 1000000 * (*it)->getProgressValue() / expect;
299
300 if (m_stop_with_first_app)
301 // Stop with first app
302 value = std::max(value, this_value);
303 else
304 // Stop with last app
305 value = std::min(value, this_value);
306 }
307 }
308 return value;
309}
