141void Windows::add(DynamicMicroOp* micro_op)
142{
143 LOG_ASSERT_ERROR(!wIsFull(), "Window is full");
144
145 WindowEntry& entry = getInstructionByIndex(m_window_tail);
146 m_window_tail = incrementIndex(m_window_tail);
147 m_window_length++;
148 micro_op->setSequenceNumber(m_next_sequence_number);
149 m_next_sequence_number++;
150
151 entry.initialize(micro_op);
152
153 uint64_t lowestValidSequenceNumber = getInstructionByIndex(m_old_window_head).getSequenceNumber();
154 m_register_dependencies->setDependencies(*micro_op, lowestValidSequenceNumber);
155 m_memory_dependencies->setDependencies(*micro_op, lowestValidSequenceNumber);
156}

164Windows::WindowEntry& Windows::getInstruction(uint64_t sequenceNumber) const
165{
166 Windows::WindowEntry& windowHead = getInstructionByIndex(m_window_head__old_window_tail);
167 int distance = windowHead.getSequenceNumber() - sequenceNumber;
168 int index = windowIndex(windowHead.getWindowIndex() - distance);
169
170 Windows::WindowEntry& ret = getInstructionByIndex(index);
171 if (ret.getSequenceNumber() == sequenceNumber)
172 {
173 return ret;
174 }
175 else
176 {
177 std::cout << "Getting an instruction that is not in the window !!! That's it, I'm out of here !" << std::endl;
178 exit(0);
179 }

209void Windows::dispatchInstruction()
210{
211 if (m_old_window_length == m_window_size)
212 {
213 // No longer keep track of instructions leaving the old window
214 removeFunctionalUnitStats(getInstructionByIndex(m_old_window_head));
215
216 /* Recalculate critical path: new m_old_window_head */
217 m_critical_path_head = std::max(m_critical_path_head, getInstructionByIndex(m_old_window_head).getExecTime());
218 m_old_window_head = incrementIndex(m_old_window_head);
219 m_old_window_length--;
220 }
221
222 // Keep track of the just dispatched instruction
223 // Determine if this new instruction will cause the use of too many resources
224 addFunctionalUnitStats(getInstructionByIndex(m_window_head__old_window_tail));
225
226 m_window_head__old_window_tail = incrementIndex(m_window_head__old_window_tail);
227 m_window_length--;
228 m_old_window_length++;
229}

279uint64_t Windows::longLatencyOperationLatency(WindowEntry& uop)
280{
281 if (uop.getExecTime() < m_critical_path_tail)
282 {
283 // No critical path extension
284 return 0;
285 }
286 else if (uop.getExecTime() - m_critical_path_tail < m_core_model->getLongLatencyCutoff())
287 {
288 // Extension smaller than LLL cutoff
289 return 0;
290 }
291 else
292 {
293 // Critical path extension longer than LLL cutoff
294 return true;
295 }

298uint64_t Windows::updateCriticalPathTail(WindowEntry& uop)
299{
300 uint64_t execTime = uop.getExecTime();
301 if (execTime > m_critical_path_tail) {
302 if (execTime - m_critical_path_tail > m_core_model->getLongLatencyCutoff()) {
303 // Extending the critical path by huge amounts isn't healthy (see Redmine #113).
304 // Instead, the caller should add the long latency directly (and attribute it) and call clearOldWindow()
305 LOG_PRINT_WARNING_ONCE("Warning: Updating the critical path by %lu > %u cycles", execTime - m_critical_path_tail, m_core_model->getLongLatencyCutoff());
306 }
307 uop.setCpContr(execTime - m_critical_path_tail);
308 m_critical_path_tail = execTime;
309 }
310 return m_critical_path_tail;
311}

351int Windows::calculateBranchResolutionLatency()
352{
353 Windows::WindowEntry& micro_op = getInstructionToDispatch();
354 uint32_t br_resolution_latency = 0;
355
356 // The m_exec_time_map is empty when the algorithm starts !
357
358 // Mark direct producers of this instruction
359 for(uint32_t i = 0; i < micro_op.getDynMicroOp()->getDependenciesLength(); i++)
360 {
361 if (oldWindowContains(micro_op.getDynMicroOp()->getDependency(i)))
362 {
363 Windows::WindowEntry& producer = getInstruction(micro_op.getDynMicroOp()->getDependency(i));
364 m_exec_time_map[producer.getWindowIndex()] = producer.getDynMicroOp()->getExecLatency();
365 }
366 }
367
368 // Find/mark producers of producers
369 for (int i = windowIndex(m_window_head__old_window_tail - 1), j = 0; j < m_old_window_length; i = windowIndex(i - 1), j++)
370 {
371 if (m_exec_time_map[i])
372 {
373 // There is a path to the committed branch: check the dependencies
374 Windows::WindowEntry& op = getInstructionByIndex(i);
375 for (uint32_t k = 0; k < op.getDynMicroOp()->getDependenciesLength(); k++)
376 {
377 if (oldWindowContains(op.getDynMicroOp()->getDependency(k)))
378 {
379 Windows::WindowEntry& producer = getInstruction(op.getDynMicroOp()->getDependency(k));
380 m_exec_time_map[producer.getWindowIndex()] = std::max((producer.getDynMicroOp()->getExecLatency() + m_exec_time_map[i]), m_exec_time_map[producer.getWindowIndex()]);
381 }
382 }
383
384 br_resolution_latency = std::max(br_resolution_latency, m_exec_time_map[i]);
385 // Reset m_exec_time_map entry
386 m_exec_time_map[i] = 0;
387 }
388 }
389
390 return br_resolution_latency;
391}

417CpContrType getCpContrType(const Windows::WindowEntry& entry)
418{
419 const MicroOp &uop = *entry.getMicroOp();
420 switch(uop.getSubtype()) {
421 case MicroOp::UOP_SUBTYPE_FP_ADDSUB:
422 return CPCONTR_TYPE_FP_ADDSUB;
423 case MicroOp::UOP_SUBTYPE_FP_MULDIV:
424 return CPCONTR_TYPE_FP_MULDIV;
425 case MicroOp::UOP_SUBTYPE_LOAD:
426 switch(entry.getDynMicroOp()->getDCacheHitWhere()) {
427 case HitWhere::L1_OWN:
428 return CPCONTR_TYPE_LOAD_L1;
429 case HitWhere::L2_OWN:
430 case HitWhere::L1_SIBLING:
431 case HitWhere::L2_SIBLING:
432 return CPCONTR_TYPE_LOAD_L2;
433 case HitWhere::L3_OWN:
434 case HitWhere::L3_SIBLING:
435 return CPCONTR_TYPE_LOAD_L3;
436 default:
437 return CPCONTR_TYPE_LOAD_LX;
438 }
439 case MicroOp::UOP_SUBTYPE_STORE:
440 return CPCONTR_TYPE_STORE;
441 case MicroOp::UOP_SUBTYPE_GENERIC:
442 return CPCONTR_TYPE_GENERIC;
443 case MicroOp::UOP_SUBTYPE_BRANCH:
444 return CPCONTR_TYPE_BRANCH;
445 default:
446 LOG_PRINT_ERROR("Unknown uop_subtype %d", uop.getSubtype());
447 }
448 assert(false);
449}
