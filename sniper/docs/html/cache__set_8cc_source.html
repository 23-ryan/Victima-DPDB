171 case CacheBase::LRU:
172 case CacheBase::LRU_QBS:

175 case CacheBase::NRU:
176 return new CacheSetNRU(cache_type, associativity, blocksize, is_tlb_set);
177
178 case CacheBase::MRU:
179 return new CacheSetMRU(cache_type, associativity, blocksize, is_tlb_set);
180
181 case CacheBase::NMRU:
182 return new CacheSetNMRU(cache_type, associativity, blocksize, is_tlb_set);
183
184 case CacheBase::PLRU:
185 return new CacheSetPLRU(cache_type, associativity, blocksize, is_tlb_set);
186
187 case CacheBase::SRRIP:
188 case CacheBase::SRRIP_QBS:
189 return new CacheSetSRRIP(cfgname, core_id, cache_type, associativity, blocksize, dynamic_cast<CacheSetInfoLRU*>(set_info), getNumQBSAttempts(policy, cfgname, core_id), is_tlb_set);
190
191 case CacheBase::RANDOM:
192 return new CacheSetRandom(cache_type, associativity, blocksize, is_tlb_set);
193
194 default:
195 LOG_PRINT_ERROR("Unrecognized Cache Replacement Policy: %i",
196 policy);
197 break;
198 }
