74void ThreadStatsManager::calculateWaitingCosts(SubsecondTime time)
75{
76 // Calculate a waiting cost for all threads.
77 // For fully-subscribed systems, the cost is equal to each thread's waiting time.
78 // On over-subscribed systems, waiting is Ok as long as there are other threads that can execute,
79 // so the cost becomes equal to the number of unused core*cycles, which is spread out over all waiting threads.
80
81 if (time > m_waiting_time_last)
82 {
83 SubsecondTime time_delta = time - m_waiting_time_last;
84 UInt32 n_running = 0, n_stalled = 0, n_total = Sim()->getConfig()->getApplicationCores();
85 for(thread_id_t thread_id = 0; thread_id < (thread_id_t)Sim()->getThreadManager()->getNumThreads(); ++thread_id)
86 {
87 if (Sim()->getThreadManager()->isThreadRunning(thread_id))
88 ++n_running;
89 else if (Sim()->getThreadManager()->getThreadStallReason(thread_id) == ThreadManager::STALL_UNSCHEDULED)
90 ;
91 else
92 ++n_stalled;
93 }
94
95 if (n_stalled && n_running <= n_total)
96 {
97 SubsecondTime cost = (n_total - n_running) * time_delta / n_stalled;
98 for(thread_id_t thread_id = 0; thread_id < (thread_id_t)Sim()->getThreadManager()->getNumThreads(); ++thread_id)
99 {
100 if (!Sim()->getThreadManager()->isThreadRunning(thread_id)
101 && Sim()->getThreadManager()->getThreadStallReason(thread_id) != ThreadManager::STALL_UNSCHEDULED)
102 {
103 m_threads_stats[thread_id]->m_counts[WAITING_COST] += cost.getFS();
104 }
105 }
106 }
107 m_waiting_time_last = time;
108 }
109}

111UInt64 ThreadStatsManager::metricCallback(ThreadStatType type, thread_id_t thread_id, Core *core, UInt64 user)
112{
113 switch(type)
114 {
115 case INSTRUCTIONS:
116 return core->getPerformanceModel()->getInstructionCount();
117 case ELAPSED_NONIDLE_TIME:
118 return core->getPerformanceModel()->getNonIdleElapsedTime().getFS();
119 case WAITING_COST:
120 // Waiting cost is added directly to m_counts[], as it needs to be applied even (especially!) when the thread is not on a core
121 return 0;
122 default:
123 LOG_PRINT_ERROR("Invalid ThreadStatType(%d) for this callback", type);
124 }

197void ThreadStatsManager::ThreadStats::update(SubsecondTime time, bool init)
198{
199 if (Sim()->getThreadManager()->getThreadState(m_thread->getId()) == Core::IDLE
200 || Sim()->getThreadManager()->getThreadState(m_thread->getId()) == Core::INITIALIZING)
201 return;
202
203 // Increment per-thread statistics based on the progress our core has made since last time
204 SubsecondTime time_delta = init || m_time_last > time ? SubsecondTime::Zero() : time - m_time_last;
205 if (m_core_id == INVALID_CORE_ID)
206 {
207 m_elapsed_time += time_delta;
208 m_unscheduled_time += time_delta;
209 }
210 else
211 {
212 Core *core = Sim()->getCoreManager()->getCoreFromID(m_core_id);
213 m_elapsed_time += time_delta;
214 time_by_core[core->getId()] += core->getPerformanceModel()->getNonIdleElapsedTime().getFS() - m_last[ELAPSED_NONIDLE_TIME];
215 insn_by_core[core->getId()] += core->getPerformanceModel()->getInstructionCount() - m_last[INSTRUCTIONS];
216 for(std::unordered_map<ThreadStatType, UInt64>::iterator it = m_counts.begin(); it != m_counts.end(); ++it)
217 {
218 m_counts[it->first] += Sim()->getThreadStatsManager()->callThreadStatCallback(it->first, m_thread->getId(), core) - m_last[it->first];
219 }
220 }
221 // Take a snapshot of our current core's statistics for later comparison
222 Core *core = m_thread->getCore();
223 if (core)
224 {
225 m_core_id = core->getId();
226 for(std::unordered_map<ThreadStatType, UInt64>::iterator it = m_counts.begin(); it != m_counts.end(); ++it)
227 {
228 m_last[it->first] = Sim()->getThreadStatsManager()->callThreadStatCallback(it->first, m_thread->getId(), core);
229 }
230 }
231 else
232 m_core_id = INVALID_CORE_ID;
233
234 if (time > m_time_last)
235 m_time_last = time;
236}
