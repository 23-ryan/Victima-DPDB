40unsigned int InstructionDecoder::getNumExecs(const xed_decoded_inst_t *ins, int numLoads, int numStores)
41{
42 if (xed_decoded_inst_get_category(ins) == XED_CATEGORY_DATAXFER || xed_decoded_inst_get_category(ins) == XED_CATEGORY_CMOV
43 || xed_decoded_inst_get_iclass(ins) == XED_ICLASS_PUSH || xed_decoded_inst_get_iclass(ins) == XED_ICLASS_POP)
44 {
45 unsigned int numExecs = 0;
46
47 // Move instructions with additional microops to process the load and store information
48 switch(xed_decoded_inst_get_iclass(ins))
49 {
50 case XED_ICLASS_MOVLPS:
51 case XED_ICLASS_MOVLPD:
52 case XED_ICLASS_MOVHPS:
53 case XED_ICLASS_MOVHPD:
54 case XED_ICLASS_SHUFPS:
55 case XED_ICLASS_SHUFPD:
56 case XED_ICLASS_BLENDPS:
57 case XED_ICLASS_BLENDPD:
58 case XED_ICLASS_EXTRACTPS:
59 case XED_ICLASS_ROUNDSS:
60 case XED_ICLASS_ROUNDPS:
61 case XED_ICLASS_ROUNDSD:
62 case XED_ICLASS_ROUNDPD:
63 numExecs += 1;
64 break;
65 case XED_ICLASS_INSERTPS:
66 numExecs += 2;
67 break;
68 default:
69 break;
70 }
71
72 // Explicit register moves. Normal loads and stores do not require this.
73 if ((numLoads + numStores) == 0)
74 {
75 numExecs += 1;
76 }
77
78 return numExecs;
79 }
80 else
81 {
82 return 1;
83 }

94
95 std::vector<std::set<xed_reg_enum_t> > regs_loads, regs_stores;
96 std::set<xed_reg_enum_t> regs_mem, regs_src, regs_dst;
97 std::vector<uint16_t> memop_load_size, memop_store_size;
98
99 int numLoads = 0;
100 int numExecs = 0;
101 int numStores = 0;
102
103 // Ignore memory-referencing operands in NOP instructions
104 if (!xed_decoded_inst_get_attribute(ins, XED_ATTRIBUTE_NOP))
105 {

109 regs.insert(xed_decoded_inst_get_base_reg(ins, mem_idx));
110 regs.insert(xed_decoded_inst_get_index_reg(ins, mem_idx));
111
112 if (xed_decoded_inst_mem_read(ins, mem_idx)) {
113 regs_loads.push_back(regs);
114 memop_load_size.push_back(xed_decoded_inst_get_memory_operand_length(ins, mem_idx));
115 numLoads++;
116 }
117
118 if (xed_decoded_inst_mem_written(ins, mem_idx)) {
119 regs_stores.push_back(regs);
120 memop_store_size.push_back(xed_decoded_inst_get_memory_operand_length(ins, mem_idx));
121 numStores++;

123
124 regs_mem.insert(regs.begin(), regs.end());

132
133 const xed_inst_t *inst = xed_decoded_inst_inst(ins);
134 for(uint32_t idx = 0; idx < xed_inst_noperands(inst); ++idx)
135 {

137 xed_operand_enum_t name = xed_operand_name(op);

143 }
144 else if (xed_operand_is_register(name))
145 {
146 xed_reg_enum_t reg = xed_decoded_inst_get_reg(ins, name);
147
148 if (xed_operand_read(op) && regs_mem.count(reg) == 0)
149 regs_src.insert(reg);
150 if (xed_operand_written(op))
151 regs_dst.insert(reg);
152 }
153 }

155 // Not sure if this is needed, the hardware is doing a lot of optimizations on stack operations
156 #if 0

158 {

160 // (It's already listed as a source memory operand for the load/store, but the modified rsp needs to be written back)
161 // (It should be one of the implicit operands, but those are rather difficult to parse correctly)
162 regs_dst.insert(XED_REG_RSP);
163 }
164 #endif

169
170 // Determine instruction operand width

191 }

195

198 // TODO: There may be more (newer) instructions, but they are all kernel only

200 case XED_ICLASS_CALL_FAR:
201 case XED_ICLASS_RET_FAR:

247 }

299 || xed_decoded_inst_get_iclass(ins) == XED_ICLASS_SFENCE
300 )
301 currentMicroOp->setMemBarrier(true);
302
303 // Special cases
304 if ((xed_decoded_inst_get_iclass(ins) == XED_ICLASS_MOVHPD)
305 || (xed_decoded_inst_get_iclass(ins) == XED_ICLASS_MOVHPS)
306 || (xed_decoded_inst_get_iclass(ins) == XED_ICLASS_MOVLPD)
307 || (xed_decoded_inst_get_iclass(ins) == XED_ICLASS_MOVLPS)
308 || (xed_decoded_inst_get_iclass(ins) == XED_ICLASS_MOVSD_XMM) // EXEC exists only for reg-to-reg moves

312 addSrcs(regs_dst, currentMicroOp);
313 }
314 }
315
316 else /* STORE */
317 {
318 size_t storeIndex = index - numLoads - numExecs;
319 addSrcs(regs_stores[storeIndex], currentMicroOp);
320 addAddrs(regs_stores[storeIndex], currentMicroOp);
321
322 if (numExecs == 0) {
323 // No execute microop: we inherit its write operands
324 addDsts(regs_dst, currentMicroOp);
325 if (numLoads == 0)
326 // No load microops either: we also inherit its read operands
327 addSrcs(regs_src, currentMicroOp);
328 }
329 if (is_atomic)
330 currentMicroOp->setMemBarrier(true);
331 }
332
333
334 /* Extra information on all micro ops */
335
336 currentMicroOp->setOperandSize(operand_size);
337 currentMicroOp->setInstruction(ins_ptr);
338
339
340 /* Extra information of first micro op */
341
342 if (index == 0)
343 {
344 currentMicroOp->setFirst(true);
345
346 // Use of x87 FPU?
347 if (toupper(xed_iclass_enum_t2str(xed_decoded_inst_get_iclass(ins))[0]) == 'F')
348 currentMicroOp->setIsX87(true);
349 }
350
351
352 /* Extra information on last micro op */
353
354 if (index == totalMicroOps - 1) {
355 currentMicroOp->setLast(true);
356
357 // Interrupts and serializing instructions are added to the last microOp
358 // Since we're user-level, there are no interrupts
359 #if 0
360 // Check if the instruction is an interrupt, place the interrupt flag
361 if (disassembler_instruction->insType != IT_BRANCH
362 && disassembler_instruction->insType != IT_CALL
363 && disassembler_instruction->insType != IT_RETURN
364 && jump) {

367 #endif
368
369 // Check if the instruction is serializing, place the serializing flag
370 if (is_serializing)
371 currentMicroOp->setSerializing(true);
372 }
373
374 uops->push_back(currentMicroOp);
375 }
