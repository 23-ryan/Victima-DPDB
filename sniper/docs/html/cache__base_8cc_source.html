51void
52CacheBase::splitAddress(const IntPtr addr, IntPtr& tag, UInt32& set_index) const
53{
54 tag = addr >> m_log_blocksize;
55
56 IntPtr linearAddress = m_ahl ? m_ahl->getLinearAddress(addr) : addr;
57 IntPtr block_num = linearAddress >> m_log_blocksize;
58
59 switch(m_hash)
60 {
61 case CacheBase::HASH_MASK:
62 set_index = block_num & (m_num_sets-1);
63 break;
64 case CacheBase::HASH_MOD:
65 set_index = block_num % m_num_sets;
66 break;
67 case CacheBase::HASH_RNG1_MOD:
68 {
69 UInt64 state = rng_seed(block_num);
70 set_index = rng_next(state) % m_num_sets;
71 break;
72 }
73 case CacheBase::HASH_RNG2_MOD:
74 {
75 UInt64 state = rng_seed(block_num);
76 rng_next(state);
77 set_index = rng_next(state) % m_num_sets;
78 break;
79 }
80 case CacheBase::HASH_PRIME_DIS:
81 {
82
83 //Prime Displacement Hashing Function (HPCA04)
84 UInt64 si = block_num % m_num_sets;
85 UInt64 Ti = block_num >> m_log_num_sets;
86 UInt64 rho = 3;
87 set_index = (rho * Ti + si) % m_num_sets;
88 break;
89 }
90 case CacheBase::HASH_XOR_MOD:
91 {
92 //Based on related work of "Eliminating Conflict Misses Using Prime Number-Based Cache Indexing" (TC may 2005)
93 //XOR based hash function
94 UInt64 si = block_num % m_num_sets;
95 UInt64 ti = (block_num >> m_log_num_sets) % m_num_sets;
96 set_index = (si ^ ti); // ^ -> bitwise XOR
97 break;
98 }
99 case CacheBase::HASH_MER_MOD:
100 {
101 //Based on related work of "Eliminating Conflict Misses Using Prime Number-Based Cache Indexing" (TC may 2005)
102 //Mersenne based hash function
103 //Disadvantage of this mod is that we will not use one set in our cache.
104 set_index = block_num % (m_num_sets - 1);
105 break;
106 }
107 default:
108 LOG_PRINT_ERROR("Invalid hash function %d", m_hash);
109 }
110}
