16{
17 // history_list queuing model does not play nice with the interval core model:
18 // The interval model issues memory operations in big batches, even when it later decides to serialize those,
19 // The history_list queue model will in this case return large, increasing latencies which are then also serialized.
20 // Also, this queuing model does not work very well with threads that are out-of-order.
21 // In general, in a simulation environment with loose time synchronization (or fluffy time), you don't want
22 // to look at exact times (such as this model does) since those are wrong, but use averages only (which are fine)
23 // -- this is exactly what the Windowed M/G/1 queue model is supposed to do.
24 //LOG_PRINT_WARNING_ONCE("history_list queuing model is deprecated. Please consider using windowed_mg1 instead.");

33 max_list_size = Sim()->getCfg()->getInt("queue_model/history_list/max_list_size");
34 }

37 LOG_PRINT_ERROR("Could not read parameters from cfg");

160SubsecondTime
161QueueModelHistoryList::computeUsingAnalyticalModel(SubsecondTime pkt_time, SubsecondTime processing_time)
162{
163 // WH: Old code used general queuing model to estimate delay based on historical utilization rate:
164 // queue_delay = (rho * processing_time) / (2 * (1 - rho))) + 1
165 // For rho near or over 1, which does happen due to other approximations (skew between cores), this yields nonsense.
166 // Yet, everyone knows that in a system with feedback, the arrival rate is no longer an independent, exponential process
167 // (i.e., more delay causes upstream buffer blockages which throttle the actual request rate, avoiding rho from ever reaching 1)
168 // Current best guess: return average of delays computed using history list model
169 return m_average_delay->compute();
170}

173QueueModelHistoryList::computeUsingHistoryList(SubsecondTime pkt_time, SubsecondTime processing_time)
174{
175 LOG_ASSERT_ERROR(m_free_interval_list.size() <= m_max_free_interval_list_size,
176 "Free Interval list size(%u) > %u", m_free_interval_list.size(), m_max_free_interval_list_size);
177 SubsecondTime queue_delay = SubsecondTime::MaxTime();
178
179 FreeIntervalList::iterator curr_it;
180 for (curr_it = m_free_interval_list.begin(); curr_it != m_free_interval_list.end(); curr_it ++)
181 {
182 std::pair<SubsecondTime,SubsecondTime> interval = (*curr_it);
183
184 if ((pkt_time >= interval.first) && ((pkt_time + processing_time) <= interval.second))
185 {
186 queue_delay = SubsecondTime::Zero();
187 // Adjust the data structure accordingly
188 curr_it = m_free_interval_list.erase(curr_it);
189 if ((pkt_time - interval.first) >= m_min_processing_time)
190 {
191 m_free_interval_list.insert(curr_it, std::pair<SubsecondTime,SubsecondTime>(interval.first, pkt_time));
192 }
193 if ((interval.second - (pkt_time + processing_time)) >= m_min_processing_time)
194 {
195 m_free_interval_list.insert(curr_it, std::pair<SubsecondTime,SubsecondTime>(pkt_time + processing_time, interval.second));
196 }
197 break;
198 }
199 // WH: The request comes before this free part, but doesn't fit. It doesn't make sense to me to
200 // demand a fit and move this request down even further. In reality, this request would have most
201 // likely executed at interval.first, while later request would/could be delayed. But it's too late
202 // for that now.
203 // (If we assume all wait times are additive then the average works out by shifting it down,
204 // but since this is an interactive simulation all delays propagate through the system
205 // so this won't be accurate.)
206 else if ((pkt_time < interval.first) /*&& ((interval.first + processing_time) <= interval.second)*/)
207 {
208 queue_delay = interval.first - pkt_time;
209 // Adjust the data structure accordingly
210 curr_it = m_free_interval_list.erase(curr_it);
211 if ((interval.second - (interval.first + processing_time)) >= m_min_processing_time)
212 {
213 m_free_interval_list.insert(curr_it, std::pair<SubsecondTime,SubsecondTime>(interval.first + processing_time, interval.second));
214 }
215 break;
216 }
217 }
218
219 LOG_ASSERT_ERROR(queue_delay != SubsecondTime::MaxTime(), "queue delay(%s), free interval not found", itostr(queue_delay).c_str());
220
221 if (m_free_interval_list.size() > m_max_free_interval_list_size)
222 {
223 m_free_interval_list.erase(m_free_interval_list.begin());
224 }
225
226 LOG_PRINT("HistoryList: pkt_time(%s), processing_time(%s), queue_delay(%s)", itostr(pkt_time).c_str(), itostr(processing_time).c_str(), itostr(queue_delay).c_str());
