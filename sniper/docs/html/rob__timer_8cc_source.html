242RobTimer::RobEntry *RobTimer::findEntryBySequenceNumber(UInt64 sequenceNumber)
243{
244 // Assumption: MicroOps in the ROB are numbered sequentially, none of them are removed halfway
245 UInt64 first = rob[0].uop->getSequenceNumber();
246 UInt64 position = sequenceNumber - first;
247 LOG_ASSERT_ERROR(position < rob.size(), "Sequence number %ld outside of ROB", sequenceNumber);
248 RobEntry *entry = &rob[position];
249 LOG_ASSERT_ERROR(entry->uop->getSequenceNumber() == sequenceNumber, "Sequence number %ld unexpectedly not at ROB position %ld", sequenceNumber, position);
250 return entry;
251}

253boost::tuple<uint64_t,SubsecondTime> RobTimer::simulate(const std::vector<DynamicMicroOp*>& insts)
254{
255 uint64_t totalInsnExec = 0;
256 SubsecondTime totalLat = SubsecondTime::Zero();
257
258 for (std::vector<DynamicMicroOp*>::const_iterator it = insts.begin(); it != insts.end(); it++ )
259 {
260 if ((*it)->isSquashed())
261 {
262 delete *it;
263 continue;
264 }
265
266 RobEntry *entry = &this->rob.next();
267 entry->init(*it, nextSequenceNumber++);
268
269 // Add = calculate dependencies, add yourself to list of depenants
270 // If no dependants in window: set ready = now()
271 uint64_t lowestValidSequenceNumber = this->rob.size() > 0 ? this->rob.front().uop->getSequenceNumber() : 0;
272 if (entry->uop->getMicroOp()->isStore())
273 {
274 for(unsigned int i = 0; i < entry->uop->getMicroOp()->getAddressRegistersLength(); ++i)
275 {
276 dl::Decoder::decoder_reg reg = entry->uop->getMicroOp()->getAddressRegister(i);
277 uint64_t addressProducer = this->registerDependencies->peekProducer(reg, lowestValidSequenceNumber);
278 if (addressProducer != INVALID_SEQNR)
279 {
280 RobEntry *prodEntry = this->findEntryBySequenceNumber(addressProducer);
281 if (prodEntry->done != SubsecondTime::MaxTime())
282 entry->addressReadyMax = std::max(entry->addressReadyMax, prodEntry->done);
283 else
284 entry->addAddressProducer(addressProducer);
285 }
286 }
287 if (entry->getNumAddressProducers() == 0)
288 entry->addressReady = entry->addressReadyMax;
289 }
290 this->registerDependencies->setDependencies(*entry->uop, lowestValidSequenceNumber);
291 this->memoryDependencies->setDependencies(*entry->uop, lowestValidSequenceNumber);
292
293 if (m_store_to_load_forwarding && entry->uop->getMicroOp()->isLoad())
294 {
295 for(unsigned int i = 0; i < entry->uop->getDependenciesLength(); ++i)
296 {
297 RobEntry *prodEntry = this->findEntryBySequenceNumber(entry->uop->getDependency(i));
298 // If we depend on a store
299 if (prodEntry->uop->getMicroOp()->isStore())
300 {
301 // Remove dependency on the store (which won't execute until it reaches the front of the ROB)
302 entry->uop->removeDependency(entry->uop->getDependency(i));
303
304 // Add dependencies to the producers of the value being stored instead
305 // Remark: one of these may be producing the store address, but because the store has to be
306 // disambiguated, it's correct to have the load depend on the address producers as well.
307 for(unsigned int j = 0; j < prodEntry->uop->getDependenciesLength(); ++j)
308 entry->uop->addDependency(prodEntry->uop->getDependency(j));
309
310 break;
311 }
312 }
313 }
314
315 // Add ourselves to the dependants list of the uops we depend on
316 uint64_t minProducerDistance = UINT64_MAX;
317 m_totalConsumers += 1 ;
318 uint64_t deps_to_remove[8], num_dtr = 0;
319 for(unsigned int i = 0; i < entry->uop->getDependenciesLength(); ++i)
320 {
321 RobEntry *prodEntry = this->findEntryBySequenceNumber(entry->uop->getDependency(i));
322 minProducerDistance = std::min( minProducerDistance, entry->uop->getSequenceNumber() - prodEntry->uop->getSequenceNumber() );
323 if (prodEntry->done != SubsecondTime::MaxTime())
324 {
325 // If producer is already done (but hasn't reached writeback stage), remove it from our dependency list
326 deps_to_remove[num_dtr++] = entry->uop->getDependency(i);
327 entry->readyMax = std::max(entry->readyMax, prodEntry->done);
328 }
329 else
330 {
331 prodEntry->addDependant(entry);
332 }
333 }
334
335 #ifdef DEBUG_PERCYCLE
336 // Make sure we are in the dependant list of all of our address producers
337 for(unsigned int i = 0; i < entry->getNumAddressProducers(); ++i)
338 {
339 if (rob.size() && entry->getAddressProducer(i) >= rob[0].uop->getSequenceNumber())
340 {
341 RobEntry *prodEntry = this->findEntryBySequenceNumber(entry->getAddressProducer(i));
342 bool found = false;
343 for(unsigned int j = 0; j < prodEntry->getNumDependants(); ++j)
344 if (prodEntry->getDependant(j) == entry)
345 {
346 found = true;
347 break;
348 }
349 LOG_ASSERT_ERROR(found == true, "Store %ld depends on %ld for address production, but is not in its dependants list",
350 entry->uop->getSequenceNumber(), prodEntry->uop->getSequenceNumber());
351 }
352 }
353 #endif
354
355 if (minProducerDistance != UINT64_MAX)
356 {
357 m_totalProducerInsDistance += minProducerDistance;
358 // KENZO: not sure why the distance can be larger than the windowSize, but it happens...
359 if (minProducerDistance >= m_producerInsDistance.size())
360 minProducerDistance = m_producerInsDistance.size()-1;
361 m_producerInsDistance[ minProducerDistance ]++ ;
362 }
363 else
364 {
365 // Not depending on any instruction in the rob
366 m_producerInsDistance[ 0 ] += 1 ;
367 }
368
369 // If there are any dependencies to be removed, do this after iterating over them (don't mess with the list we're reading)
370 LOG_ASSERT_ERROR(num_dtr < sizeof(deps_to_remove)/sizeof(8), "Have to remove more dependencies than I expected");
371 for(uint64_t i = 0; i < num_dtr; ++i)
372 entry->uop->removeDependency(deps_to_remove[i]);
373 if (entry->uop->getDependenciesLength() == 0)
374 {
375 // We have no dependencies in the ROB: mark ourselves as ready
376 entry->ready = entry->readyMax;
377 }
378
379 #ifdef DEBUG_PERCYCLE
380 std::cout<<"** simulate: "<<entry->uop->getMicroOp()->toShortString(true)<<std::endl;
381 #endif
382
383 m_uop_type_count[(*it)->getMicroOp()->getSubtype()]++;
384 m_uops_total++;
385 if ((*it)->getMicroOp()->isX87()) m_uops_x87++;
386 if ((*it)->getMicroOp()->isPause()) m_uops_pause++;
387
388 if (m_uops_total > 10000 && m_uops_x87 > m_uops_total / 20)
389 LOG_PRINT_WARNING_ONCE("Significant fraction of x87 instructions encountered, accuracy will be low. Compile without -mno-sse2 -mno-sse3 to avoid.");
390 }
391
392 while (true)
393 {
394 uint64_t instructionsExecuted;
395 SubsecondTime latency;
396 execute(instructionsExecuted, latency);
397 totalInsnExec += instructionsExecuted;
398 totalLat += latency;
399 if (latency == SubsecondTime::Zero())
400 break;
401 }
402
403 return boost::tuple<uint64_t,SubsecondTime>(totalInsnExec, totalLat);
404}

414SubsecondTime* RobTimer::findCpiComponent()
415{
416 // Determine the CPI component corresponding to the first non-committed instruction
417 for(uint64_t i = 0; i < m_num_in_rob; ++i)
418 {
419 RobEntry *entry = &rob.at(i);
420 DynamicMicroOp *uop = entry->uop;
421 // Skip over completed instructions
422 if (entry->done < now)
423 continue;
424 // This is the first instruction in the ROB which is still executing
425 // Assume everyone is blocked on this one
426 // Assign 100% of this cycle to this guy's CPI component
427 if (uop->getMicroOp()->isSerializing() || uop->getMicroOp()->isMemBarrier())
428 return &m_cpiSerialization;
429 else if (uop->getMicroOp()->isLoad() || uop->getMicroOp()->isStore())
430 return &m_cpiDataCache[uop->getDCacheHitWhere()];
431 else
432 return NULL;
433 }
434 // No instruction is currently executing
435 return NULL;
436}

438SubsecondTime RobTimer::doDispatch(SubsecondTime **cpiComponent)
439{
440 SubsecondTime next_event = SubsecondTime::MaxTime();
441 SubsecondTime *cpiFrontEnd = NULL;
442
443 if (frontend_stalled_until <= now)
444 {
445 uint32_t instrs_dispatched = 0, uops_dispatched = 0;
446
447 while(m_num_in_rob < windowSize)
448 {
449 LOG_ASSERT_ERROR(m_num_in_rob < rob.size(), "Expected sufficient uops for dispatching in pre-ROB buffer, but didn't find them");
450 RobEntry *entry = &rob.at(m_num_in_rob);
451 DynamicMicroOp &uop = *entry->uop;
452
453 // Dispatch up to 4 instructions
454 if (uops_dispatched == dispatchWidth)
455 break;
456
457 // This is actually in the decode stage, there's a buffer between decode and dispatch
458 // so we shouldn't do this here.
461 //if (instrs_dispatched > 0 && !uop.isLast())
462 // break;
463
464 bool iCacheMiss = (uop.getICacheHitWhere() != HitWhere::L1I);
465 if (iCacheMiss)
466 {
467 if (in_icache_miss)
468 {
469 // We just took the latency for this instruction, now dispatch it
470 #ifdef DEBUG_PERCYCLE
471 std::cout<<"-- icache return"<<std::endl;
472 #endif
473 in_icache_miss = false;
474 }
475 else
476 {
477 #ifdef DEBUG_PERCYCLE
478 std::cout<<"-- icache miss("<<uop.getICacheLatency()<<")"<<std::endl;
479 #endif
480 frontend_stalled_until = now + uop.getICacheLatency();
481 in_icache_miss = true;
482 // Don't dispatch this instruction yet
483 cpiFrontEnd = &m_cpiInstructionCache[uop.getICacheHitWhere()];
484 break;
485 }
486 }
487
488 if (m_rs_entries_used == rsEntries)
489 {
490 cpiFrontEnd = &m_cpiRSFull;
491 break;
492 }
493
494 entry->dispatched = now;
495 ++m_num_in_rob;
496 ++m_rs_entries_used;
497
498 uops_dispatched++;
499 if (uop.isLast())
500 instrs_dispatched++;
501
502 // If uop is already ready, we may need to issue it in the following cycle
503 entry->ready = std::max(entry->ready, (now + 1ul).getElapsedTime());
504 next_event = std::min(next_event, entry->ready);
505
506 #ifdef DEBUG_PERCYCLE
507 std::cout<<"DISPATCH "<<entry->uop->getMicroOp()->toShortString()<<std::endl;
508 #endif
509
510 #ifdef ASSERT_SKIP
511 LOG_ASSERT_ERROR(will_skip == false, "Cycle would have been skipped but stuff happened");
512 #endif
513
514 // Mispredicted branch
515 if (uop.getMicroOp()->isBranch() && uop.isBranchMispredicted())
516 {
517 frontend_stalled_until = SubsecondTime::MaxTime();
518 #ifdef DEBUG_PERCYCLE
519 std::cout<<"-- branch mispredict"<<std::endl;
520 #endif
521 cpiFrontEnd = &m_cpiBranchPredictor;
522 break;
523 }
524 }
525
526 m_cpiCurrentFrontEndStall = cpiFrontEnd;
527 }
528 else
529 {
530 // Front-end is still stalled: re-use last CPI component
531 cpiFrontEnd = m_cpiCurrentFrontEndStall;
532 }
533
534
535 // Find CPI component corresponding to the first executing instruction
536 SubsecondTime *cpiRobHead = findCpiComponent();
537
538 if (cpiFrontEnd)
539 {
540 // Front-end is stalled
541 if (cpiRobHead)
542 {
543 // Have memory/serialization components take precendence over front-end stalls
544 *cpiComponent = cpiRobHead;
545 }
546 else
547 {
548 *cpiComponent = cpiFrontEnd;
549 }
550 }
551 else if (m_num_in_rob == windowSize)
552 {
553 *cpiComponent = cpiRobHead ? cpiRobHead : &m_cpiBase;
554 }
555 else
556 {
557 *cpiComponent = &m_cpiBase;
558 }
559
560
561 if (m_num_in_rob == windowSize)
562 return next_event; // front-end is effectively stalled so wait for another event
563 else
564 return std::min(frontend_stalled_until, next_event);
565}

567void RobTimer::issueInstruction(uint64_t idx, SubsecondTime &next_event)
568{
569 RobEntry *entry = &rob[idx];
570 DynamicMicroOp &uop = *entry->uop;
571
572 if ((uop.getMicroOp()->isLoad() || uop.getMicroOp()->isStore())
573 && uop.getDCacheHitWhere() == HitWhere::UNKNOWN)
574 {
575 MemoryResult res = m_core->accessMemory(
576 Core::NONE,
577 uop.getMicroOp()->isLoad() ? Core::READ : Core::WRITE,
578 uop.getAddress().address,
579 NULL,
580 uop.getMicroOp()->getMemoryAccessSize(),
581 Core::MEM_MODELED_RETURN,
582 uop.getMicroOp()->getInstruction() ? uop.getMicroOp()->getInstruction()->getAddress() : static_cast<uint64_t>(NULL),
583 now.getElapsedTime()
584 );
585 uint64_t latency = SubsecondTime::divideRounded(res.latency, now.getPeriod());
586
587 uop.setExecLatency(uop.getExecLatency() + latency); // execlatency already contains bypass latency
588 uop.setDCacheHitWhere(res.hit_where);
589 }
590
591 if (uop.getMicroOp()->isLoad())
592 {
593 load_queue.getCompletionTime(now, uop.getExecLatency() * now.getPeriod(), uop.getAddress().address);
594 }
595 else if (uop.getMicroOp()->isStore())
596 {
597 store_queue.getCompletionTime(now, uop.getExecLatency() * now.getPeriod(), uop.getAddress().address);
598 }
599
600 ComponentTime cycle_depend = now + uop.getExecLatency(); // When result is available for dependent instructions
601 SubsecondTime cycle_done = cycle_depend + 1ul; // When the instruction can be committed
602
603 if (uop.getMicroOp()->isLoad())
604 {
605 m_loads_count++;
606 m_loads_latency += uop.getExecLatency() * now.getPeriod();
607 }
608 else if (uop.getMicroOp()->isStore())
609 {
610 m_stores_count++;
611 m_stores_latency += uop.getExecLatency() * now.getPeriod();
612 }
613
614 if (uop.getMicroOp()->isStore())
615 {
616 last_store_done = std::max(last_store_done, cycle_done);
617 cycle_depend = now + 1ul; // For stores, forward the result immediately
618 // Stores can be removed from the ROB once they're issued to the memory hierarchy
619 // Dependent operations such as SFENCE and synchronization instructions need to wait until last_store_done
620 cycle_done = now + 1ul;
621
622 LOG_ASSERT_ERROR(entry->addressReady <= entry->ready, "%ld: Store address cannot be ready (%ld) later than the whole uop is (%ld)",
623 entry->uop->getSequenceNumber(), entry->addressReady.getPS(), entry->ready.getPS());
624 }
625
626 if (m_rob_contention)
627 m_rob_contention->doIssue(uop);
628
629 entry->issued = now;
630 entry->done = cycle_done;
631 next_event = std::min(next_event, entry->done);
632
633 --m_rs_entries_used;
634
635 #ifdef DEBUG_PERCYCLE
636 std::cout<<"ISSUE "<<entry->uop->getMicroOp()->toShortString()<<" latency="<<uop.getExecLatency()<<std::endl;
637 #endif
638
639 for(size_t idx = 0; idx < entry->getNumDependants(); ++idx)
640 {
641 RobEntry *depEntry = entry->getDependant(idx);
642 LOG_ASSERT_ERROR(depEntry->uop->getDependenciesLength()> 0, "??");
643
644 // Remove uop from dependency list and update readyMax
645 depEntry->readyMax = std::max(depEntry->readyMax, cycle_depend.getElapsedTime());
646 depEntry->uop->removeDependency(uop.getSequenceNumber());
647
648 // If all dependencies are resolved, mark the uop ready
649 if (depEntry->uop->getDependenciesLength() == 0)
650 {
651 depEntry->ready = depEntry->readyMax;
652 //std::cout<<" ready @ "<<depEntry->ready<<std::endl;
653 }
654
655 // For stores, check if their address has been produced
656 if (depEntry->uop->getMicroOp()->isStore() && depEntry->addressReady == SubsecondTime::MaxTime())
657 {
658 bool ready = true;
659 for(unsigned int i = 0; i < depEntry->getNumAddressProducers(); ++i)
660 {
661 uint64_t addressProducer = depEntry->getAddressProducer(i);
662 RobEntry *prodEntry = addressProducer >= this->rob.front().uop->getSequenceNumber()
663 ? this->findEntryBySequenceNumber(addressProducer) : NULL;
664
665 if (prodEntry == entry)
666 {
667 // The instruction we just executed is producing an address. Update the store's addressReadyMax
668 depEntry->addressReadyMax = std::max(depEntry->addressReadyMax, cycle_depend.getElapsedTime());
669 }
670
671 if (prodEntry && prodEntry->done == SubsecondTime::MaxTime())
672 {
673 // An address producer has not yet been issued: address remains not ready
674 ready = false;
675 }
676 }
677
678 if (ready)
679 {
680 // We did not find any address producing instructions that have not yet been issued.
681 // Store address will be ready at addressReadyMax
682 depEntry->addressReady = depEntry->addressReadyMax;
683 }
684 }
685 }
686
687 // After issuing a mispredicted branch: allow the ROB to refill after flushing the pipeline
688 if (uop.getMicroOp()->isBranch() && uop.isBranchMispredicted())
689 {
690 frontend_stalled_until = now + (misprediction_penalty - 2); // The frontend needs to start 2 cycles earlier to get a total penalty of <misprediction_penalty>
691 #ifdef DEBUG_PERCYCLE
692 std::cout<<"-- branch resolve"<<std::endl;
693 #endif
694 }
695}

697SubsecondTime RobTimer::doIssue()
698{
699 uint64_t num_issued = 0;
700 SubsecondTime next_event = SubsecondTime::MaxTime();
701 bool head_of_queue = true, no_more_load = false, no_more_store = false, have_unresolved_store = false;
702
703 if (m_rob_contention)
704 m_rob_contention->initCycle(now);
705
706 for(uint64_t i = 0; i < m_num_in_rob; ++i)
707 {
708 RobEntry *entry = &rob.at(i);
709 DynamicMicroOp *uop = entry->uop;
710
711
712 if (entry->done != SubsecondTime::MaxTime())
713 {
714 next_event = std::min(next_event, entry->done);
715 continue; // already done
716 }
717
718 next_event = std::min(next_event, entry->ready);
719
720
721 // See if we can issue this instruction
722
723 bool canIssue = false;
724
725 if (entry->ready > now)
726 canIssue = false; // blocked by dependency
727
728 else if ((no_more_load && uop->getMicroOp()->isLoad()) || (no_more_store && uop->getMicroOp()->isStore()))
729 canIssue = false; // blocked by mfence
730
731 else if (uop->getMicroOp()->isSerializing())
732 {
733 if (head_of_queue && last_store_done <= now)
734 canIssue = true;
735 else
736 break;
737 }
738
739 else if (uop->getMicroOp()->isMemBarrier())
740 {
741 if (head_of_queue && last_store_done <= now)
742 canIssue = true;
743 else
744 // Don't issue any memory operations following a memory barrier
745 no_more_load = no_more_store = true;
746 // FIXME: L/SFENCE
747 }
748
749 else if (!m_rob_contention && num_issued == dispatchWidth)
750 canIssue = false; // no issue contention: issue width == dispatch width
751
752 else if (uop->getMicroOp()->isLoad() && !load_queue.hasFreeSlot(now))
753 canIssue = false; // load queue full
754
755 else if (uop->getMicroOp()->isLoad() && m_no_address_disambiguation && have_unresolved_store)
756 canIssue = false; // preceding store with unknown address
757
758 else if (uop->getMicroOp()->isStore() && (!head_of_queue || !store_queue.hasFreeSlot(now)))
759 canIssue = false; // store queue full
760
761 else
762 canIssue = true; // issue!
763
764
765 // canIssue already marks issue ports as in use, so do this one last
766 if (canIssue && m_rob_contention && ! m_rob_contention->tryIssue(*uop))
767 canIssue = false; // blocked by structural hazard
768
769
770 if (canIssue)
771 {
772 num_issued++;
773 issueInstruction(i, next_event);
774
775 // Calculate memory-level parallelism (MLP) for long-latency loads (but ignore overlapped misses)
776 if (uop->getMicroOp()->isLoad() && uop->isLongLatencyLoad() && uop->getDCacheHitWhere() != HitWhere::L1_OWN)
777 {
778 if (m_lastAccountedMemoryCycle < now) m_lastAccountedMemoryCycle = now;
779
780 SubsecondTime done = std::max( now.getElapsedTime(), entry->done );
781 // Ins will be outstanding for until it is done. By account beforehand I don't need to
782 // worry about fast-forwarding simulations
783 m_outstandingLongLatencyInsns += (done - now);
784
785 // Only account for the cycles that have not yet been accounted for by other long
786 // latency misses (don't account cycles twice).
787 if ( done > m_lastAccountedMemoryCycle )
788 {
789 m_outstandingLongLatencyCycles += done - m_lastAccountedMemoryCycle;
790 m_lastAccountedMemoryCycle = done;
791 }
792
793 #ifdef ASSERT_SKIP
794 LOG_ASSERT_ERROR( m_outstandingLongLatencyInsns >= m_outstandingLongLatencyCycles, "MLP calculation is wrong: MLP cannot be < 1!" );
795 #endif
796 }
797
798
799 #ifdef ASSERT_SKIP
800 LOG_ASSERT_ERROR(will_skip == false, "Cycle would have been skipped but stuff happened");
801 #endif
802 }
803 else
804 {
805 head_of_queue = false; // Subsequent instructions are not at the head of the ROB
806
807 if (uop->getMicroOp()->isStore() && entry->addressReady > now)
808 have_unresolved_store = true;
809
810 if (inorder)
811 // In-order: only issue from head of the ROB
812 break;
813 }
814
815
816 if (m_rob_contention)
817 {
818 if (m_rob_contention->noMore())
819 break;
820 }
821 else
822 {
823 if (num_issued == dispatchWidth)
824 break;
825 }
826 }
827
828 return next_event;
829}

831SubsecondTime RobTimer::doCommit(uint64_t& instructionsExecuted)
832{
833 uint64_t num_committed = 0;
834
835 while(rob.size() && (rob.front().done <= now))
836 {
837 RobEntry *entry = &rob.front();
838
839 #ifdef DEBUG_PERCYCLE
840 std::cout<<"COMMIT "<<entry->uop->getMicroOp()->toShortString()<<std::endl;
841 #endif
842
843 // Send instructions to loop tracer, in-order, once we know their issue time
844 InstructionTracer::uop_times_t times = {
845 entry->dispatched,
846 entry->issued,
847 entry->done,
848 now
849 };
850 m_core->getPerformanceModel()->traceInstruction(entry->uop, &times);
851
852 if (entry->uop->isLast())
853 instructionsExecuted++;
854
855 entry->free();
856 rob.pop();
857 m_num_in_rob--;
858
859 #ifdef ASSERT_SKIP
860 LOG_ASSERT_ERROR(will_skip == false, "Cycle would have been skipped but stuff happened");
861 #endif
862
863 ++num_committed;
864 if (num_committed == commitWidth)
865 break;
866 }
867
868 if (rob.size())
869 return rob.front().done;
870 else
871 return SubsecondTime::MaxTime();
872}

874void RobTimer::execute(uint64_t& instructionsExecuted, SubsecondTime& latency)
875{
876 latency = SubsecondTime::Zero();
877 instructionsExecuted = 0;
878 SubsecondTime *cpiComponent = NULL;
879
880 #ifdef DEBUG_PERCYCLE
881 std::cout<<std::endl;
882 std::cout<<"Running cycle "<<now<<std::endl;
883 #endif
884
885
886 // If frontend not stalled
887 if (frontend_stalled_until <= now)
888 {
889 if (rob.size() < m_num_in_rob + 2*dispatchWidth)
890 {
891 // We don't have enough instructions to dispatch <dispatchWidth> new ones. Ask for more before doing anything this cycle.
892 return;
893 }
894 }
895
896
897 // Model dispatch, issue and commit stages
898 // Decode stage is not modeled, assumes the decoders can keep up with (up to) dispatchWidth uops per cycle
899
900 SubsecondTime next_dispatch = doDispatch(&cpiComponent);
901 SubsecondTime next_issue = doIssue();
902 SubsecondTime next_commit = doCommit(instructionsExecuted);
903
904
905 #ifdef DEBUG_PERCYCLE
906 #ifdef ASSERT_SKIP
907 if (! will_skip)
908 {
909 #endif
910 printRob();
911 #ifdef ASSERT_SKIP
912 }
913 #endif
914 #endif
915
916
917 #ifdef DEBUG_PERCYCLE
918 std::cout<<"Next event: D("<<next_dispatch<<") I("<<next_issue<<") C("<<next_commit<<")"<<std::endl;
919 #endif
920 SubsecondTime next_event = std::min(next_dispatch, std::min(next_issue, next_commit));
921 SubsecondTime skip;
922 if (next_event != SubsecondTime::MaxTime() && next_event > now + 1ul)
923 {
924 #ifdef DEBUG_PERCYCLE
925 std::cout<<"++ Skip "<<(next_event - now)<<std::endl;
926 #endif
927 will_skip = true;
928 skip = next_event - now;
929 }
930 else
931 {
932 will_skip = false;
933 skip = now.getPeriod();
934 }
935
936 #ifdef ASSERT_SKIP
937 now += now.getPeriod();
938 latency += now.getPeriod();
939 if (will_skip)
940 time_skipped += now.getPeriod();
941 #else
942 now += skip;
943 latency += skip;
944 if (skip > now.getPeriod())
945 time_skipped += skip - now.getPeriod();
946 #endif
947
948 if (m_mlp_histogram)
949 countOutstandingMemop(skip);
950
951 LOG_ASSERT_ERROR(cpiComponent != NULL, "We expected cpiComponent to be set by doDispatch, but it wasn't");
952 *cpiComponent += latency;
953}

955void RobTimer::countOutstandingMemop(SubsecondTime time)
956{
957 UInt64 counts[HitWhere::NUM_HITWHERES] = {0}, total = 0;
958
959 for(unsigned int i = 0; i < m_num_in_rob; ++i)
960 {
961 RobEntry *e = &rob.at(i);
962 if (e->done != SubsecondTime::MaxTime() && e->done > now && e->uop->getMicroOp()->isLoad())
963 {
964 ++counts[e->uop->getDCacheHitWhere()];
965 ++total;
966 }
967 }
968
969 for(unsigned int h = 0; h < HitWhere::NUM_HITWHERES; ++h)
970 if (counts[h] > 0)
971 m_outstandingLoads[h][counts[h] >= MAX_OUTSTANDING ? MAX_OUTSTANDING-1 : counts[h]] += time;
972 if (total > 0)
973 m_outstandingLoadsAll[total >= MAX_OUTSTANDING ? MAX_OUTSTANDING-1 : total] += time;
974}
