102void Log::parseModules(std::set<String> &mods, String list)
103{
104 String delimiters = " ";
105
106 String::size_type lastPos = list.find_first_not_of(delimiters, 0);
107 String::size_type pos = list.find_first_of(delimiters, lastPos);
108
109 std::set<String> unformatted;
110
111 while (String::npos != pos || String::npos != lastPos)
112 {
113 unformatted.insert(list.substr(lastPos, pos - lastPos));
114 lastPos = list.find_first_not_of(delimiters, pos);
115 pos = list.find_first_of(delimiters, lastPos);
116 }
117
118 for (std::set<String>::iterator it = unformatted.begin();
119 it != unformatted.end();
120 it++)
121 {
122 String formatted;
123
124 for (unsigned int i = 0; i < std::min(MODULE_LENGTH, it->length()); i++)
125 {
126 formatted.push_back((*it)[i]);
127 }
128
129 for (unsigned int i = formatted.length(); i < MODULE_LENGTH; i++)
130 {
131 formatted.push_back(' ');
132 }
133
134 assert(formatted.length() == MODULE_LENGTH);
135 mods.insert(formatted);
136 }
137}
