28bool RobContentionBoomV1::tryIssue(const DynamicMicroOp &uop)
29{
30 // Port contention
31 // TODO: Maybe the scheduler is more intelligent and doesn't just assing the first uop in the ROB
32 // that fits a particular free port. We could give precedence to uops that have dependants, etc.
33 // NOTE: mixes canIssue and doIssue, in the sense that ports* are incremented immediately.
34 // This works as long as, if we return true, this microop is indeed issued
35
36 const DynamicMicroOpBoomV1 *core_uop_info = uop.getCoreSpecificInfo<DynamicMicroOpBoomV1>();
37 DynamicMicroOpBoomV1::uop_port_t uop_port = core_uop_info->getPort();
38
39 if (uop_port == DynamicMicroOpBoomV1::UOP_PORT012)
40 {
41 if (ports_generic012 >= 3)
42 return false;
43 else
44 ports_generic012++;
45 }
46 else
47 { // PORT0, PORT1 or PORT2
48 if (ports[uop_port])
49 return false;
50 else if (ports_generic012 >= 3)
51 return false;
52 else
53 {
54 ports[uop_port] = true;
55 ports_generic012++;
56 }
57 }
58
59 // ALU contention
60 if (DynamicMicroOpBoomV1::uop_alu_t alu = core_uop_info->getAlu())
61 {
62 if (alu_used_until[alu] > m_now)
63 return false;
64 }
65
66 return true;
67}

69void RobContentionBoomV1::doIssue(DynamicMicroOp &uop)
70{
71 const DynamicMicroOpBoomV1 *core_uop_info = uop.getCoreSpecificInfo<DynamicMicroOpBoomV1>();
72 DynamicMicroOpBoomV1::uop_alu_t alu = core_uop_info->getAlu();
73 if (alu)
74 alu_used_until[alu] = m_now + m_core_model->getAluLatency(uop.getMicroOp());
75}

77bool RobContentionBoomV1::noMore()
78{
79 // When we issued something to all ports in this cycle, stop walking the rest of the ROB
80 if (ports[DynamicMicroOpBoomV1::UOP_PORT2] && ports_generic012 >= 3)
81 return true;
82 else
83 return false;
