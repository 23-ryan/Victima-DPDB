80 definition(config_parser const& self) {
81 //Grammar Definition
82
83 // Complex rules
84 r_file = r_config_node >> end_p;
85
86 r_config_node = r_config;
87
88 r_config = *r_section_node ;
89
90 r_section_node = access_node_d[ r_section ][Name(sectionNameID)];
91
92 r_section = root_node_d[r_section_name_node] >> *r_key_node ;
93
94 r_key_node = access_node_d[ r_key ][Name(keyID)];
95
96 r_key = r_key_name
97 >> lexeme_d[ no_node_d[ *r_key_array ] >> no_node_d[ *space_p ] >> root_node_d[ ch_p('=') ] >> no_node_d[ *space_p ] ]
98 >> !r_value ;
99
100 r_key_name = access_node_d[ !r_string ][Name(keyNameID)];
101
102 r_key_array = ch_p('[') >> ch_p(']');
103
104 r_value_array = access_node_d[ r_value_array2 ][Name(keyValueArrayID)];
105
106 r_value_array2 = r_value_single_or_empty >> +(access_node_d[ ch_p(',') ][Name(keySeparatorID)] >> r_value_single_or_empty);
107
108 r_value_span = access_node_d[ r_value_long >> root_node_d[ ch_p('-') ] >> r_value_long ][Name(keyValueSpanID)];
109
110 r_value_long = access_node_d[ long_p ][Name(keyValueID)];
111
112 r_value_single = access_node_d[ r_string ][Name(keyValueID)];
113
114 r_value_single_or_empty = r_value_single | !r_value_single;
115
116 r_value = r_value_array | r_value_span | r_value_single;
117
118 // Terminals
119 // Note, the lexeme directive tells the parser to act character-by-character (don't skip spaces),
120 // the token_node_d says to make 1 node out of the matched results (instead of 1 node per character)
121 // inner_node directive states to throw away each of the '[' and ']'
122 // finally config_p looks for matching [ with ]
123 r_section_name_node = access_node_d[ r_section_name ][Name(sectionNameID)] ;
124
125 //HACK:
126 //
127 //This is a total hack below. Because we are using Boost 1.34, we cannot use the gen_pt_node_d
128 //in certain places (notably the one commented below). The gen_pt_node_d function allows us to
129 //generate an AST node, even if there is no data contained within the node. This happens when we
130 //get a section that doesn't have a name in it (i.e. []). In Boost 1.35 we can use the following
131 //gent_pt_node_d with the inner_node_d to eliminate the braces at parse-time. Instead until then
132 //we have to leave the braces in due to null-node-culling and strip them at the time we walk the
133 //tree.
134
135 r_section_name = lexeme_d[ token_node_d[ confix_p('[', (*lex_escape_ch_p), ']') ] ] ;
136
137 // Strings and names may either be "quoted" or not.
138 r_string =
139 (
140 ( no_node_d[ *space_p ] >> lexeme_d[ token_node_d[ inner_node_d[ confix_p('"', *lex_escape_ch_p, '"') ] ] ] >> no_node_d[ *space_p ] )
141 |
142 ( lexeme_d[ token_node_d[ +(alnum_p | (punct_p - ch_p('=') - ch_p(',') - ch_p('[') - ch_p(']') ) ) ] ] )
143 )
144 ;
145 }
