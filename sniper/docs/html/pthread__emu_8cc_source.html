37void pthreadCount(pthread_enum_t function, Core *core, SubsecondTime delay_sync, SubsecondTime delay_mem)
38{
39 pthread_counters[core->getId()].pthread_count[function]++;
40 pthread_counters[core->getId()].pthread_total_delay_sync[function] += delay_sync;
41 pthread_counters[core->getId()].pthread_total_delay_mem[function] += delay_mem;
42}

44/* Model the kernel's hash_bucket lock used in the futex syscall.
45 Contended pthread_mutex_[un]lock calls should bring this address into the cache in exclusive state.
46 Some mutexes may collide if the hash function maps to the same value, but let's assume this is uncommon.
47 Instead, give each mutex (more or less) its own cache line. Allocate these for the real process as well.
48*/
49static std::unordered_map<pthread_mutex_t*, IntPtr> futex_map;
50static Lock futex_map_lock;
51IntPtr futexHbAddress(pthread_mutex_t *mux) {
52 ScopedLock sl(futex_map_lock);
53 if (futex_map.count(mux) == 0)
54 futex_map[mux] = (IntPtr)memalign(64, 64);
55 return futex_map[mux];
56}

116IntPtr MutexLock (pthread_mutex_t *mux)
117{
118 Core *core = Sim()->getCoreManager()->getCurrentCore();
119 assert (core);
120
121 pthread_mutex_t _mux;
122
123 /* Model the lock cmpxchg(mux) inside the real pthread_mutex_lock/lll_lock */
124 MemoryResult lat = core->accessMemory(Core::NONE, Core::READ_EX, (IntPtr) mux, (char *) &_mux, sizeof (pthread_mutex_t), Core::MEM_MODELED_FENCED);
125
126 updateState(core, STATE_WAITING);
127 SubsecondTime delay = CarbonMutexLock((carbon_mutex_t*) mux, lat.latency);
128
129 MemoryResult lat1 = makeMemoryResult(HitWhere::UNKNOWN, SubsecondTime::Zero());
130 if (delay > SubsecondTime::Zero()) { /* Assume in the uncontended case, nothing (not the (system) network, nor the MCPs SyncServer) adds any delay */
131 /* Model the lock addw(hb->spinlock) inside the futex_wake syscall */
132 lat1 = core->accessMemory(Core::NONE, Core::READ_EX, (IntPtr) futexHbAddress(mux), NULL, sizeof (UInt32), Core::MEM_MODELED_FENCED);
133 pthread_counters[core->getId()].pthread_mutex_lock_contended++;
134 }
135
136 /* Delay and lat will be pushed as dynamic instructions, but have not been processed yet so we need to tell updateState to add them to core->getCycleCount(). */
137 updateState(core, STATE_INREGION, delay + lat.latency + lat1.latency);
138
139 pthreadCount(PTHREAD_MUTEX_LOCK, core, delay, lat.latency + lat1.latency);
140 return 0;
141}

144{
145 Core *core = Sim()->getCoreManager()->getCurrentCore();

149
150 /* Model the lock cmpxchg(mux) inside the real pthread_mutex_trylock/lll_trylock */

162IntPtr MutexUnlock (pthread_mutex_t *mux)
163{
164 Core *core = Sim()->getCoreManager()->getCurrentCore();
165 assert (core);

167 pthread_mutex_t _mux;
168
169 /* Model the lock sub(mux) inside the real pthread_mutex_unlock/lll_unlock */
170 MemoryResult lat = core->accessMemory(Core::NONE, Core::READ_EX, (IntPtr) mux, (char *) &_mux, sizeof (pthread_mutex_t), Core::MEM_MODELED_FENCED);

174 MemoryResult lat1 = makeMemoryResult(HitWhere::UNKNOWN, SubsecondTime::Zero());
175 if (delay > SubsecondTime::Zero()) {
176 /* Model the lock addw(hb->spinlock) inside the futex_wait syscall */
177 // TODO: the latency hit for this should actually be while still holding the lock.
178 // But we can't request the latency until we've contacted the server (which already releases the lock) to tell us whether it's contended
179 // Also, no-one is currently spinning on this (and keeping the line in shared state) -- in fact, we may have even been the last ones to have used it in our matching pthread_mutex_lock call

181 pthread_counters[core->getId()].pthread_mutex_unlock_contended++;

186 pthreadCount(PTHREAD_MUTEX_UNLOCK, core, delay, lat.latency + lat1.latency);
