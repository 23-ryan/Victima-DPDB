118{
119 ScopedLock sl(m_lock);
120
121 core_id_t requester = INVALID_CORE_ID;
122
123 if (pkt.type == SHARED_MEM_1)
124 requester = getNetwork()->getCore()->getMemoryManager()->getShmemRequester(pkt.data);
125 else // Other Packet types
126 requester = pkt.sender;
127
128 LOG_ASSERT_ERROR((requester >= 0) && (requester < (core_id_t) Config::getSingleton()->getTotalCores()),
129 "requester(%i)", requester);
130
131 UInt32 pkt_length = getNetwork()->getModeledLength(pkt);
132
133 LOG_PRINT("pkt length(%u)", pkt_length);
134
135 if (pkt.sender == m_core_id)
136 {
137 m_total_packets_sent ++;
138 m_total_bytes_sent += pkt_length;
139 }
140
141 if (pkt.receiver == NetPacket::BROADCAST)
142 {
143 if (m_fake_node)
144 {
145 for (core_id_t i = 0; i < (core_id_t) Config::getSingleton()->getTotalCores(); i++)
146 {
147 addHop(DESTINATION, i, i, pkt.time, pkt_length, nextHops, requester);
148 }
149 }
150 else if (m_broadcast_tree_enabled)
151 {
152 // Injection Port Modeling
153 SubsecondTime injection_port_queue_delay = SubsecondTime::Zero();
154 if (pkt.sender == m_core_id)
155 injection_port_queue_delay = computeInjectionPortQueueDelay(pkt.receiver, pkt.time, pkt_length);
156 SubsecondTime curr_time = pkt.time + injection_port_queue_delay;
157
158 // Broadcast tree is enabled
159 // Build the broadcast tree
160 SInt32 sx, sy, cx, cy;
161
162 computePosition(pkt.sender, sx, sy);
163 computePosition(m_core_id, cx, cy);
164
165 if (cy >= sy)
166 addHop(UP, NetPacket::BROADCAST, computeCoreId(cx,cy+1), curr_time, pkt_length, nextHops, requester);
167 if (cy <= sy)
168 addHop(DOWN, NetPacket::BROADCAST, computeCoreId(cx,cy-1), curr_time, pkt_length, nextHops, requester);
169 if (cy == sy)
170 {
171 if (cx >= sx)
172 addHop(RIGHT, NetPacket::BROADCAST, computeCoreId(cx+1,cy), curr_time, pkt_length, nextHops, requester);
173 if (cx <= sx)
174 addHop(LEFT, NetPacket::BROADCAST, computeCoreId(cx-1,cy), curr_time, pkt_length, nextHops, requester);
175 if (cx == sx)
176 addHop(SELF, m_core_id, m_core_id, curr_time, pkt_length, nextHops, requester);
177 }
178 }
179 else
180 {
181 // Broadcast tree is not enabled
182 // Here, broadcast messages are sent as a collection of unicast messages
183 LOG_ASSERT_ERROR(pkt.sender == m_core_id,
184 "BROADCAST message to be sent at (%i), original sender(%i), Tree not enabled",
185 m_core_id, pkt.sender);
186
187 for (core_id_t i = 0; i < (core_id_t) Config::getSingleton()->getTotalCores(); i++)
188 {
189 // Injection Port Modeling
190 SubsecondTime injection_port_queue_delay = computeInjectionPortQueueDelay(i, pkt.time, pkt_length);
191 SubsecondTime curr_time = pkt.time + injection_port_queue_delay;
192
193 // Unicast message to each core
194 OutputDirection direction;
195 core_id_t next_dest = getNextDest(i, direction);
196
197 addHop(direction, i, next_dest, curr_time, pkt_length, nextHops, requester);
198 }
199 }
200 }
201 else if (m_fake_node)
202 {
203 addHop(DESTINATION, pkt.receiver, pkt.receiver, pkt.time, pkt_length, nextHops, requester);
204 }
205 else
206 {
207 // Injection Port Modeling
208 SubsecondTime injection_port_queue_delay = SubsecondTime::Zero();
209 if (pkt.sender == m_core_id)
210 {
211 injection_port_queue_delay = computeInjectionPortQueueDelay(pkt.receiver, pkt.time, pkt_length);
212 *(subsecond_time_t*)&pkt.queue_delay += injection_port_queue_delay;
213 }
214 SubsecondTime curr_time = pkt.time + injection_port_queue_delay;
215
216 // A Unicast packet
217 OutputDirection direction;
218 core_id_t next_dest = getNextDest(pkt.receiver, direction);
219
220 addHop(direction, pkt.receiver, next_dest, curr_time, pkt_length, nextHops, requester, (subsecond_time_t*)&pkt.queue_delay);

231 core_id_t requester = INVALID_CORE_ID;
232
233 if (pkt.type == SHARED_MEM_1)
234 requester = getNetwork()->getCore()->getMemoryManager()->getShmemRequester(pkt.data);

252
253 packet_latency += (ejection_port_queue_delay + processing_time);
254 contention_delay += ejection_port_queue_delay;
255 pkt.time += (ejection_port_queue_delay + processing_time);
256 pkt.queue_delay += ejection_port_queue_delay;
257 }
258
259 m_total_packets_received ++;
260 m_total_bytes_received += pkt_length;
261 m_total_packet_latency += packet_latency;
262 m_total_contention_delay += contention_delay;

379SInt32
380NetworkModelEMeshHopByHop::getNextDest(SInt32 final_dest, OutputDirection& direction)
381{
382 // Do dimension-order routing
383 // Curently, do store-and-forward routing
384 // FIXME: Should change this to wormhole routing eventually
385
386 if (final_dest >= (core_id_t)Config::getSingleton()->getApplicationCores())
387 {
388 // We are a fake core (not an application core): warp straight to the destination
389 direction = DESTINATION;
390 return final_dest;
391 }
392 else if (m_core_id / m_concentration == final_dest / m_concentration)
393 {
394 // Destination is self, a peer on our concentrated node
395 direction = DESTINATION;
396 return final_dest;
397 }
398 else if (m_fake_node)
399 {
400 // We are a concentrated node but not the master: first send to master
401 direction = PEER;
402 return m_core_id - m_core_id % m_concentration;
403 }
404
405 SInt32 sx, sy, dx, dy;
406
407 computePosition(m_core_id, sx, sy);
408 computePosition(final_dest, dx, dy);
409
410 if ((sx > dx) ^ (m_wrap_around && abs(sx - dx) > (m_mesh_width+1) / 2))
411 {
412 direction = LEFT;
413 return computeCoreId(sx-1,sy);
414 }
415 else if (sx != dx)
416 {
417 direction = RIGHT;
418 return computeCoreId(sx+1,sy);
419 }
420 else if ((sy > dy) ^ (m_wrap_around && abs(sy - dy) > (m_mesh_height+1) / 2))
421 {
422 direction = DOWN;
423 return computeCoreId(sx,sy-1);
424 }
425 else if (sy != dy)
426 {
427 direction = UP;
428 return computeCoreId(sx,sy+1);
429 }
430 else
431 {
432 // A send to itself
433 direction = SELF;
434 return m_core_id;
435 }

482 }

484 {
485 // ":"-separated, "," is for heterogeneity
