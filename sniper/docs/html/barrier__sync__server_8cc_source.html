76 LOG_ASSERT_ERROR(m_barrier_acquire_list[master_core_id] == false, "Core(%i) or its sibling is already in the barrier (this is thread %d, we have thread %d)", master_core_id, thread_me, m_core_thread[master_core_id]);

83 return;

98

192bool
193BarrierSyncServer::isBarrierReached()
194{
195 bool single_core_barrier_reached = false;
196
197 // Check if all cores have reached the barrier
198 // All least one core must have (sync_time > m_next_barrier_time)
199 for (core_id_t core_id = 0; core_id < (core_id_t) Sim()->getConfig()->getApplicationCores(); core_id++)
200 {
201 // In fastforward mode, it's enough that a core is waiting. In detailed mode, it needs to have advanced up to the predefined barrier time
202 if (m_fastforward)
203 {
204 if (m_barrier_acquire_list[core_id])
205 {
206 // At least one core has reached the barrier
207 single_core_barrier_reached = true;
208 }
209 else if (isCoreRunning(core_id))
210 {
211 // Core is running but hasn't checked in yet. Wait for it to sync.
212 return false;
213 }
214 }
215 else if (m_core_group[core_id] != INVALID_CORE_ID)
216 {
217 // Only consider group masters
218 continue;
219 }
220 else if (isCoreRunning(core_id))
221 {
222 if (m_local_clock_list[core_id] < m_next_barrier_time)
223 {
224 // Core running on this core has not reached the barrier
225 // Wait for it to sync
226 return false;
227 }
228 else
229 {
230 // At least one core has reached the barrier
231 single_core_barrier_reached = true;
232 }
233 }
234 }
235
236 return single_core_barrier_reached;
237}

239bool
240BarrierSyncServer::barrierRelease(thread_id_t caller_id, bool continue_until_release)
241{
242 CLOG("barrier", "Release (caller thread %d)", caller_id);
243 LOG_PRINT("Sending 'BARRIER_RELEASE'");
244
245 // All cores have reached the barrier
246 // Advance m_next_barrier_time
247 // Release the Barrier
248
249 LOG_ASSERT_ERROR(m_to_release.size() == 0, "Reached the barrier while some threads haven't even restarted?");
250
251 if (m_fastforward)
252 {
253 for (core_id_t core_id = 0; core_id < (core_id_t) Sim()->getConfig()->getApplicationCores(); core_id++)
254 {
255 // In fast-forward mode, skip over (potentially very many) timeslots
256 if (m_local_clock_list[core_id] > m_next_barrier_time)
257 m_next_barrier_time = m_local_clock_list[core_id];
258 }
259 }
260
261 // If a core cannot be resumed, we have to advance the sync
262 // time till a core can be resumed. Then only, will we have
263 // forward progress
264
265 bool core_resumed = false;
266 bool must_wait = true;
267 while (!core_resumed)
268 {
269 m_global_time = m_next_barrier_time;
270 CLOG("barrier", "Barrier %" PRId64 "ns", m_next_barrier_time.getNS());
271 Sim()->getHooksManager()->callHooks(HookType::HOOK_PERIODIC, static_cast<subsecond_time_t>(m_next_barrier_time).m_time);
272
273 if (continue_until_release)
274 {
275 // If HOOK_PERIODIC woke someone up, this thread can safely go to sleep
276 if (Sim()->getThreadManager()->anyThreadRunning())
277 return false;
278 else
279 LOG_ASSERT_ERROR(Sim()->getSyscallServer()->getNextTimeout(m_global_time) < SubsecondTime::MaxTime(), "No threads running, no timeout. Application has deadlocked...");
280 }
281
282 // If the barrier was disabled from HOOK_PERIODIC (for instance, if roi-end was triggered from a script), break
283 if (m_disable)
284 return false;
285
286 m_next_barrier_time += m_barrier_interval;
287 LOG_PRINT("m_next_barrier_time updated to (%s)", itostr(m_next_barrier_time).c_str());
288
289 for (core_id_t core_id = 0; core_id < (core_id_t) Sim()->getConfig()->getApplicationCores(); core_id++)
290 {
291 if (m_local_clock_list[core_id] < m_next_barrier_time)
292 {
293 // Check if this core was running. If yes, release that core
294 if (m_barrier_acquire_list[core_id] == true)
295 {
296 //Core *core = Sim()->getCoreManager()->getCoreFromID(core_id);
297 //LOG_ASSERT_ERROR(core->getState() == Core::RUNNING || core->getState() == Core::INITIALIZING, "(%i) has acquired barrier, local_clock(%s), m_next_barrier_time(%s), but not initializing or running", core_id, itostr(m_local_clock_list[core_id]).c_str(), itostr(m_next_barrier_time).c_str());
298
299 m_barrier_acquire_list[core_id] = false;
300 core_resumed = true;
301
302 if (m_core_thread[core_id] == caller_id)
303 must_wait = false;
304 else
305 {
306 Core *core = Sim()->getCoreManager()->getCoreFromID(core_id);
307 core->getPerformanceModel()->barrierExit();
308 m_to_release.push_back(core_id);
309 }
310 }
311 }
312 }
313 }
314
315 // To avoid overwhelming the OS scheduler, we only release N threads at a time (N ~= host cores).
316 // Once a thread is done (stops executing because it completed the next barrier quantum, or due to thread stall),
317 // one more thread is released so we always have at most N running threads.
318 std::random_shuffle(m_to_release.begin(), m_to_release.end());
319 doRelease(m_fastforward ? -1 : Sim()->getConfig()->getNumHostCores());
320
321 return must_wait;
322}

325BarrierSyncServer::doRelease(int n)
