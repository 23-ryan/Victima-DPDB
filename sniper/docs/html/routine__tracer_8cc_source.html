82void RoutineTracerThread::routineAssert(IntPtr eip, IntPtr esp)
83{
84 ScopedLock sl(m_lock);
85
86 if (m_stack.size() == 0)
87 {
88 // Newly created thread just jumps into the first routine
89 routineEnter_unlocked(eip, esp, 0);
90 }
91 else if (m_stack.back() == eip)
92 {
93 // We are where we think we are, no action
94 }
95 else if (esp <= m_last_esp)
96 {
97 // Stack grew (downwards), or stayed constant (tail call): we entered a new function
98 routineEnter_unlocked(eip, esp, 0);
99 }
100 else
101 {
102 bool found = unwindTo(eip);
103
104 if (!found)
105 {
106 // We now seem to be in a function we haven't been before, enter it (tail call elimination?)
107 routineEnter_unlocked(eip, esp, 0);
108 }
109 else
110 {
111 // Jumped back into a function further down the stack (longjmp?)
112 // unwindTo has already popped the stack
113 m_last_esp = esp;
114 }
115
116 // After all this, the current function should be at the top of the stack
117 LOG_ASSERT_ERROR(m_stack.back() == eip, "Expected to be in function %lx but now in %lx", eip, m_stack.back());
118 }
119}

156void RoutineTracerThread::hookRoiEnd()
157{
158 ScopedLock sl(m_lock);
159
160 // Call functionExit for all functions that are left on the stack.
161 // Since functionExit might use m_stack we need to keep it up-to-date by popping items off,
162 // we'll use stack_save to remember them and restore m_stack to its original state on exit.
163 CallStack stack_save;
164 IntPtr eip_child = 0;
165
166 while(m_stack.size())
167 {
168 if (eip_child)
169 functionChildExit(m_stack.back(), eip_child);
170 functionExit(m_stack.back());
171 eip_child = m_stack.back();
172 stack_save.push_back(m_stack.back());
173 m_stack.pop_back();
174 }
175 while(stack_save.size())
176 {
177 m_stack.push_back(stack_save.back());
178 stack_save.pop_back();
179 }
180}
