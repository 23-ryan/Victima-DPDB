145UInt32 Config::computeCoreIDLength(UInt32 core_count)
146{
147 UInt32 num_bits = ceilLog2(core_count);
148 if ((num_bits % 8) == 0)
149 return (num_bits / 8);
150 else
151 return (num_bits / 8) + 1;
152}

205UInt32 Config::getNearestAcceptableCoreCount(UInt32 core_count)
206{
207 UInt32 nearest_acceptable_core_count = 0;
208
209 UInt32 l_models[NUM_STATIC_NETWORKS];
210 try
211 {
212 config::Config *cfg = Sim()->getCfg();
213 l_models[STATIC_NETWORK_MEMORY_1] = NetworkModel::parseNetworkType(cfg->getString("network/memory_model_1"));
214 l_models[STATIC_NETWORK_SYSTEM] = NetworkModel::parseNetworkType(cfg->getString("network/system_model"));
215 }
216 catch (...)
217 {
218 config::Error("Exception while reading network model types.");
219 }
220
221 for (UInt32 i = 0; i < NUM_STATIC_NETWORKS; i++)
222 {
223 std::pair<bool,SInt32> core_count_constraints = NetworkModel::computeCoreCountConstraints(l_models[i], (SInt32) core_count);
224 if (core_count_constraints.first)
225 {
226 // Network Model has core count constraints
227 if ((nearest_acceptable_core_count != 0) &&
228 (core_count_constraints.second != (SInt32) nearest_acceptable_core_count))
229 {
230 config::Error("Problem using the network models specified in the configuration file.");
231 }
232 else
233 {
234 nearest_acceptable_core_count = core_count_constraints.second;
235 }
236 }
237 }
238
239 if (nearest_acceptable_core_count == 0)
240 nearest_acceptable_core_count = core_count;
241
242 return nearest_acceptable_core_count;
243}
