229 void BuddyAllocator::deallocate(UInt64 region_begin)
230 {
231 if(allocated_map.find(region_begin) == allocated_map.end())
232 {
233 //std::cout<<"[ERROR] FAILED TO FIND A BUDDY TO DEALLOCATE AT RANGE "<<region_begin<<std::endl;
234 return;
235 }
236
237 int ind = ceil(log(allocated_map[region_begin]) / log(2));
238 int i, buddyNumber, buddyAddress;
239
240 free_list[ind].push_back(std::make_pair(region_begin, region_begin + pow(2, ind) - 1));
241 //std::cout<<"Returned memory block from "<<region_begin<<" to "<<region_begin + pow(2, ind)-1<<std::endl;
242 m_free_pages += pow(2, ind);
243
244 //If we are at the max order, no need to try mergning
245 if(ind == m_max_order)
246 {
247 allocated_map.erase(region_begin);
248 return;
249 }
250
251 buddyNumber = region_begin / allocated_map[region_begin];
252 buddyAddress = (buddyNumber % 2 != 0) ? (region_begin - pow(2, ind)) : (region_begin + pow(2, ind));
253
254 //Begin trying to merge the buddies
255 while(true)
256 {
257 bool merged = false;
258 UInt64 new_region_begin;
259 for(i = 0; i<free_list[ind].size(); i++)
260 {
261 //If the buddy is found and free
262 if(free_list[ind][i].first == buddyAddress)
263 {
264 if(buddyNumber % 2 == 0)
265 {
266 free_list[ind + 1].push_back(std::make_pair(region_begin, region_begin + 2*pow(2, ind) - 1));
267 //std::cout<<"Mergning blocks starting at "<<region_begin<<" and "<<buddyAddress<<std::endl;
268 new_region_begin = region_begin;
269 }
270 else
271 {
272 free_list[ind+1].push_back(std::make_pair(buddyAddress, buddyAddress + 2*pow(2, ind)));
273 //std::cout<<"Mergning blocks starting at "<<buddyAddress<<" and "<<region_begin<<std::endl;
274 new_region_begin = buddyAddress;
275 }
276
277 free_list[ind].erase(free_list[ind].begin() + i);
278 free_list[ind].erase(free_list[ind].begin() + free_list[ind].size() - 1);
279 merged = true;
280 break;
281 }
282 }
283 //If we did not merge buddies, no point in continuing to try with next level
284 if(!merged)
285 break;
286
287 ind++;
288 //If we have the largest buddy, we cannot merge it further
289 if(ind == m_max_order)
290 break;
291
292 allocated_map.erase(region_begin);
293 region_begin = new_region_begin;
294 buddyNumber = 2;
295 buddyAddress = region_begin + pow(2, ind);
296 }
